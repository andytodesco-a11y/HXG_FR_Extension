Scan Planes

The Plane object is a very important one in ESPRIT EDGE. Plane encompasses both work planes and view planes. View planes define the angle from which you view your drawing in the graphics Window. Work planes determine how certain graphic objects like Arc and Circle are oriented, and are used by features to determine tool orientations for machining. When we start ESPRIT EDGE, we have certain work planes and view planes defined by default. This tutorial will introduce all of the properties associated with the Plane object by looking at those predefined planes.

Sub ScanPlanes()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
' loop through all planes
Dim PL As Esprit.Plane
For Each PL In Document.Planes
Call ScanPlane(PL)
Next
End Sub
Sub ScanPlane(PlaneObj As Esprit.Plane)
' check the plane type, note that a plane can be both a work plane and a view plane
If PlaneObj.IsWork Then
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & " is a WorkPlane")
Else
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & " is a ViewPlane")
End If
' display the origin and UVW vector information
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & ": X = " & FO(PlaneObj.X) & ", Y = " & FO(PlaneObj.Y) & ", Z = " & FO(PlaneObj.Z))
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & ": Ux = " & FO(PlaneObj.Ux) & ", Uy = " & FO(PlaneObj.Uy) & ", Uz = " & FO(PlaneObj.Uz))
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & ": Vx = " & FO(PlaneObj.Vx) & ", Vy = " & FO(PlaneObj.Vy) & ", Vz = " & FO(PlaneObj.Vz))
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanPlanesTutorial", "Plane " & PlaneObj.Name & ": Wx = " & FO(PlaneObj.Wx) & ", Wy = " & FO(PlaneObj.Wy) & ", Wz = " & FO(PlaneObj.Wz))
End Sub
Private Function FO(Num As Double) As String
' format numeric output to at most 5 decimal places
FO = Format(Num, "0.0####")
End Function
