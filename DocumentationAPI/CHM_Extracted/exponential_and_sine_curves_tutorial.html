Exponential and Sine Wave Curves

The Tutorial: Creating Curves Using FeaturePtop Commands have shown various ways of creating curves from points. What if we wanted to create a curve along a typically equation, such as an exponential or sine wave curve? All we need to do is plot sufficient points along the equation, and convert those points to a curve.

private void CreateExponentialCurve()
{
var coefA = 1.0;
var coefB = 1.0;
if (!RequestUserInput("For the equation y = Ae^Bx, enter the coefficient A.", "A", ref coefA) ||
!RequestUserInput("For the equation y = Ae^Bx, enter the coefficient B.", "B", ref coefB) ||
(coefA == 0 && coefB == 0))
{
return;
}
var startX = 0.0;
var endX = 1.0;
var stepX = 0.1;
if (!RequestUserInput("Enter the starting X value.", "Start X", ref startX) ||
!RequestUserInput("Enter the ending X value.", "End X", ref endX) ||
!RequestUserInput("Enter the X step value.", "Step X", ref stepX) ||
startX == endX)
{
return;
}
stepX = Math.Abs(stepX) * Math.Sign(endX - startX);
Esprit.FeaturePtop featurePtop = null;
double x = 0;
for (x = startX; x <= endX; x += stepX)
{
var y = coefA * Math.Exp(coefB * x);
var point = Document.GetPoint(x, y, 0);
if (featurePtop == null)
{
featurePtop = Document.FeaturePtops.Add(point);
}
else
{
featurePtop.Add(point);
}
}
if (featurePtop == null)
{
System.Windows.MessageBox.Show("There is no featurePtop");
return;
}
if (x != endX)
{
var y = coefA * Math.Exp(coefB * endX);
var point = Document.GetPoint(endX, y, 0);
featurePtop.Add(point);
}
try
{
var curve = Document.Curves.AddFromElement(featurePtop);
if (curve != null)
{
GraphicObjectHelper.SetDefaultAttributes(curve as Esprit.GraphicObject, _application.Configuration);
}
}
catch (System.Runtime.InteropServices.COMException)
{ }
Document.FeaturePtops.Remove(featurePtop.Key);
Document.Refresh();
}

Private Sub CreateExponentialCurve()
Dim coefA As Double = 1.0
Dim coefB As Double = 1.0
If Not RequestUserInput("For the equation y = Ae^Bx, enter the coefficient A.", "A", coefA) OrElse
Not RequestUserInput("For the equation y = Ae^Bx, enter the coefficient B.", "B", coefB) OrElse
(coefA = 0 And coefB = 0) Then
Return
End If
Dim startX As Double = 0.0
Dim endX As Double = 1.0
Dim stepX As Double = 0.1
If Not RequestUserInput("Enter the starting X value.", "Start X", startX) OrElse
Not RequestUserInput("Enter the ending X value.", "End X", endX) OrElse
Not RequestUserInput("Enter the X step value.", "Step X", stepX) OrElse
startX = endX Then
Return
End If
stepX = Math.Abs(stepX) * Math.Sign(endX - startX)
Dim featurePtop As Esprit.FeaturePtop = Nothing
Dim x As Double = 0.0
For x = startX To endX Step stepX
Dim y As Double = coefA * Math.Exp(coefB * x)
Dim point As Esprit.Point = Document.GetPoint(x, y, 0)
If featurePtop Is Nothing Then
featurePtop = Document.FeaturePtops.Add(point)
Else
featurePtop.Add(point)
End If
Next
If featurePtop Is Nothing Then
System.Windows.MessageBox.Show("There is no featurePtop")
Return
End If
If x <> endX Then
Dim y As Double = coefA * Math.Exp(coefB * endX)
Dim point As Esprit.Point = Document.GetPoint(endX, y, 0)
featurePtop.Add(point)
End If
Try
Dim curve As Esprit.Curve = Document.Curves.AddFromElement(featurePtop)
If Not curve Is Nothing Then
GraphicObjectHelper.SetDefaultAttributes(CType(curve, Esprit.GraphicObject), _application.Configuration)
End If
Catch ex As System.Runtime.InteropServices.COMException
End Try
Document.FeaturePtops.Remove(featurePtop.Key)
Document.Refresh()
End Sub

Sub CreateExponentialCurve()
' prompt the user for the paramters to the curve
Dim StartX As Double, EndX As Double, StepX As Double
Dim CoefficientA As Double, CoefficientB As Double
CoefficientA = Val(InputBox("For the equation y = Ae^Bx, enter the coefficient A.", "A"))
If CoefficientA = 0 Then Exit Sub
CoefficientB = Val(InputBox("For the equation y = Ae^Bx, enter the coefficient B.", "B"))
If CoefficientB = 0 Then Exit Sub
StartX = Val(InputBox("Enter the starting X value.", "Start X"))
EndX = Val(InputBox("Enter the ending X value.", "End X"))
If EndX = StartX Then Exit Sub
StepX = Val(InputBox("Enter the X step value.", "Step X"))
StepX = Abs(StepX) * Sgn(EndX - StartX) ' to make sure we go the right direction
' loop from start to end and evaluate the equation
Dim X As Double, Y As Double, P As Esprit.Point, FP As Esprit.FeaturePtop
For X = StartX To EndX Step StepX
Y = CoefficientA * Exp(CoefficientB * X)
Set P = Document.GetPoint(X, Y, 0)
If FP Is Nothing Then
Set FP = Document.FeaturePtops.Add(P)
Else
Call FP.Add(P)
End If
Next
' make sure last iteration is done at EndX value
If X <> EndX Then
Y = CoefficientA * Exp(CoefficientB * EndX)
Set P = Document.GetPoint(EndX, Y, 0)
Call FP.Add(P)
End If
' convert the feature ptop to a curve and remove
Dim Crv As Esprit.Curve
Set Crv = DefaultAttributes(Document.Curves.AddFromElement(FP))
Call Document.FeaturePtops.Remove(FP.Key)
Document.Refresh
End Sub

Enter in some appropriate values. As one example, try A = 1, B = 1, going from X0 to X1 with a step of 0.1. The curve should start at Y = 1, because Ae^(Bx) = 1e^(0x) = 1e^0 = 1 for X = 0 and it should end at Y = 2.71828, because Ae^(Bx) = 1e^(1x) = 1e^1 = e for X = 1, and e = 2.71828.

Enter in some appropriate values. As one example, try A = 1, B = 0, going from X0 to X = 6.28 (= 2 * PI) to get one cycle of just a sine wave. The sine wave will be 1 at X = 1.57 (= PI / 2), cross 0 at X = 3.14 (= PI) and -1 at X = 3 * PI / 2. Remember that the X value argument to the sin and cos functions is in radians. Try A = 0, B = 1 going from X0 to X6.28 (2*PI). That will result in a cosine wave, 90 degrees out of phase with the sine wave.
