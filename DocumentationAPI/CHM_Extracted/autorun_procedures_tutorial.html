public override void Execute()
{
if (!_isSubscribed)
{
_isSubscribed = true;
_application.BeforeQuit += OnBeforeApplicationQuit;
_application.AfterDocumentOpen += OnAfterDocumentOpen;
_application.AfterDocumentClose += OnAfterDocumentClose;
_application.AfterNewDocumentOpen += OnAfterNewDocumentOpen;
}
}

Public Overrides Sub Execute()
If Not _isSubscribed Then
_isSubscribed = True
AddHandler _application.BeforeQuit, AddressOf OnBeforeApplicationQuit
AddHandler _application.AfterDocumentOpen, AddressOf OnAfterDocumentOpen
AddHandler _application.AfterDocumentClose, AddressOf OnAfterDocumentClose
AddHandler _application.AfterNewDocumentOpen, AddressOf OnAfterNewDocumentOpen
End If
End Sub
