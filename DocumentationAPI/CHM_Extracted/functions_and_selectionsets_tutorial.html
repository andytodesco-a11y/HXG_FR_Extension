More Fun With Functions and SelectionSets

The Tutorial: Fun with Functions expanded upon functions and showed functions that returned values such as Long integer and Boolean, as well as functions which returned objects such as Point and Layer. In this tutorial we will write functions that create many GraphicObject objects at once. For them, we will want to return a SelectionSet object.

public static SelectionSet AddUniqueSelectionSet(ISelectionSets selectionSets, string baseName)
{
var suffix = 0;
foreach (SelectionSet set in selectionSets)
{
var setName = set.Name;
if (setName.Length > baseName.Length && setName.StartsWith(baseName) && setName[baseName.Length] == '_')
{
if (Int32.TryParse(setName.Substring(baseName.Length + 1), out var index))
{
if (index >= suffix)
{
suffix = index + 1;
}
}
}
}
var name = $"{baseName}_{suffix}";
return selectionSets.Add(name);
}
Public Function AddUniqueSelectionSet(document As Esprit.Document, baseName As String) As Esprit.SelectionSet
Dim baseNameLen As Long = Len(baseName)
Dim highestSSNumber As Long = 0
' loop through all SelectionSets to find ones with the same BaseName
For Each selectionSet As Esprit.SelectionSet In document.SelectionSets
' compare the position of the BaseName within the SelectionSet Name
If (InStr(1, selectionSet.Name, baseName) = 1) Then
' find the number at the end of the name
Dim sSNumber As Long = Val(Mid$(selectionSet.Name, baseNameLen + 1))
' keep track of the highest number so that it can be incremented later
If sSNumber > highestSSNumber Then highestSSNumber = sSNumber
End If
Next
' increment the highest number and return the resulting SelectionSet
baseName &= CStr(highestSSNumber + 1)
Return document.SelectionSets.Add(baseName)
End Function
Public Function AddUniqueSelectionSet(BaseName As String) As Esprit.SelectionSet
Dim BaseNameLen As Long, HighestSSNumber As Long
BaseNameLen = Len(BaseName)
HighestSSNumber = 0
' loop through all SelectionSets to find ones with the same BaseName
Dim SS As Esprit.SelectionSet, SSNumber As Long
For Each SS In Document.SelectionSets
' compare the position of the BaseName within the SelectionSet Name
If (InStr(1, SS.Name, BaseName) = 1) Then
' find the number at the end of the name
SSNumber = Val(Mid$(SS.Name, BaseNameLen + 1))
' keep track of the highest number so that it can be incremented later
If SSNumber > HighestSSNumber Then HighestSSNumber = SSNumber
End If
Next
' increment the highest number and return the resulting SelectionSet
BaseName = BaseName & CStr(HighestSSNumber + 1)
Set AddUniqueSelectionSet = Document.SelectionSets.Add(BaseName)
End Function

public static SelectionSet BoltHolePattern(Document document, Point center, double patternRadius, double holeRadius, int numberOfHoles)
{
var boltHolePattern = SelectionSetHelper.AddUniqueSelectionSet(document.SelectionSets, "BoltHole");
FeaturePtop featurePtop = null;
for (var i = 0; i < numberOfHoles; i++)
{
var angle = i * 2 * Math.PI / numberOfHoles;
var x = patternRadius * Math.Cos(angle) + center.X;
var y = patternRadius * Math.Sin(angle) + center.Y;
var hole = document.Circles.Add(document.GetPoint(x, y, center.Z), holeRadius);
boltHolePattern.Add(hole);
if (featurePtop == null)
{
featurePtop = document.FeaturePtops.Add(hole.CenterPoint);
}
else
{
featurePtop.Add(hole.CenterPoint);
}
}
if (featurePtop != null)
{
boltHolePattern.Add(featurePtop);
}
return boltHolePattern;
}

Public Function BoltHolePattern(document As Esprit.Document, center As Esprit.Point, patternRadius As Double, holeRadius As Double, numberOfHoles As Integer) As Esprit.SelectionSet
Dim boltHole As Esprit.SelectionSet = SelectionSetHelper.AddUniqueSelectionSet(document, "BoltHole")
Dim featurePtop As Esprit.FeaturePtop = Nothing
' loop through all of the holes in the pattern
For holeNumber As Integer = 0 To (numberOfHoles - 1)
' calculate the angle of the center of this hole
' with respect to the center of the pattern
Dim angle As Double = holeNumber * 2 * Math.PI / numberOfHoles
' calculate the coordinates of the center of the
' hole relative the center of the pattern
Dim x As Double = patternRadius * Math.Cos(angle) + center.X
Dim y As Double = patternRadius * Math.Sin(angle) + center.Y
' add a circle on those coordinates (without adding a point at the center)
Dim hole As Esprit.Circle = document.Circles.Add(document.GetPoint(x, y, center.Z), holeRadius)
Call boltHole.Add(hole)
If holeNumber = 0 Then
featurePtop = document.FeaturePtops.Add(hole.CenterPoint)
Else
Call featurePtop.Add(hole.CenterPoint)
End If
Next
' add the FeaturePtop to the SelectSet last
If Not featurePtop Is Nothing Then
Call boltHole.Add(featurePtop)
End If
Return boltHole
End Function

public static SelectionSet MoldHolePattern(Document document, Point center, double boundaryRadius, double holeRadius, double spacing)
{
var moldHolePattern = SelectionSetHelper.AddUniqueSelectionSet(document.SelectionSets, "MoldHole");
if (boundaryRadius < holeRadius)
{
return moldHolePattern;
}
var hole = document.Circles.Add(center, holeRadius);
moldHolePattern.Add(hole);
var centerDistance = spacing + 2 * holeRadius;
uint ring = 0;
var holesOnRing = 0;
do
{
ring++;
holesOnRing = 0;
var x = center.X + ring * centerDistance;
var y = center.Y;
for (var sideAngleInDegrees = 120; sideAngleInDegrees <= 420; sideAngleInDegrees += 60)
{
for (var holeOnSide = 1; holeOnSide <= ring; holeOnSide++)
{
x = x + centerDistance * Math.Cos(sideAngleInDegrees * Math.PI / 180);
y = y + centerDistance * Math.Sin(sideAngleInDegrees * Math.PI / 180);
var outerTangent = Geometry.Distance(x, y, center.X, center.Y) + holeRadius + spacing;
if (outerTangent < boundaryRadius)
{
hole = document.Circles.Add(document.GetPoint(x, y, center.Z), holeRadius);
moldHolePattern.Add(hole);
holesOnRing++;
}
}
}
} while (holesOnRing != 0);
return moldHolePattern;
}

Public Function MoldHolePattern(document As Esprit.Document, center As Esprit.Point, boundaryRadius As Double, holeRadius As Double, spacing As Double) As Esprit.SelectionSet
Dim moldHole As Esprit.SelectionSet = SelectionSetHelper.AddUniqueSelectionSet(document, "MoldHole")
If boundaryRadius < holeRadius Then
Return moldHole
End If
' first make hole in center
Dim hole As Esprit.Circle = document.Circles.Add(center, holeRadius)
Call moldHole.Add(hole)
Dim centerDistance As Double = spacing + 2 * holeRadius
Dim ring As Long = 0
Dim holesOnRing As Long = 0
Do
ring += 1
holesOnRing = 0
' start with the hole at 0 degrees on each ring
Dim x As Double = center.X + ring * centerDistance
Dim y As Double = center.Y
' most effecient pattern is along sides of hexagon
' follow the sides starting from 0 going counterclockwise
For sideAngleInDegrees As Integer = 120 To 420 Step 60
' make all of the holes on that side
For holeOnSide = 1 To ring
' find coordinates of next hole center, remember to convert degrees to radians
x += centerDistance * Math.Cos(sideAngleInDegrees * Math.PI / 180)
y += centerDistance * Math.Sin(sideAngleInDegrees * Math.PI / 180)
' make sure hole will fit, with spacing to spare
Dim outerTangent As Double = Math.Sqrt((x - center.X) ^ 2 + (y - center.Y) ^ 2) + holeRadius + spacing
If outerTangent < boundaryRadius Then
hole = document.Circles.Add(document.GetPoint(x, y, center.Z), holeRadius)
Call moldHole.Add(hole)
' keep track of the number of holes created
holesOnRing += 1
End If
Next
Next
' if no holes can fit on ring, we are done
Loop Until holesOnRing = 0
Return moldHole
End Function

Public Function MoldHolePattern(Center As Esprit.Point, BoundaryRadius As Double, HoleRadius As Double, Spacing As Double) As Esprit.SelectionSet
Set MoldHolePattern = AddUniqueSelectionSet("MoldHole")
If BoundaryRadius < HoleRadius Then Exit Function
Dim Hole As Esprit.Circle
' first make hole in center
Set Hole = Document.Circles.Add(Center, HoleRadius)
Call MoldHolePattern.Add(Hole)
' setup loop to make remaining holes
Dim PI As Double, CenterDistance As Double, X As Double, Y As Double
PI = 4 * Atn(1)
CenterDistance = Spacing + 2 * HoleRadius
Dim Ring As Long, HolesOnRing As Long
Ring = 0
Do
Ring = Ring + 1
HolesOnRing = 0
' start with the hole at 0 degrees on each ring
X = Center.X + Ring * CenterDistance
Y = Center.Y
' most effecient pattern is along sides of hexagon
' follow the sides starting from 0 going counterclockwise
Dim SideAngleInDegrees As Integer
For SideAngleInDegrees = 120 To 420 Step 60
' make all of the holes on that side
Dim HoleOnSide As Long
For HoleOnSide = 1 To Ring
' find coordinates of next hole center, remember to convert degrees to radians
X = X + CenterDistance * Cos(SideAngleInDegrees * PI / 180)
Y = Y + CenterDistance * Sin(SideAngleInDegrees * PI / 180)
' make sure hole will fit, with spacing to spare
Dim OuterTangent As Double
OuterTangent = Sqr((X - Center.X) ^ 2 + (Y - Center.Y) ^ 2) + HoleRadius + Spacing
If OuterTangent < BoundaryRadius Then
Set Hole = Document.Circles.Add(Document.GetPoint(X, Y, Center.Z), HoleRadius)
Call MoldHolePattern.Add(Hole)
' keep track of the number of holes created
HolesOnRing = HolesOnRing + 1
End If
Next
Next
' if no holes can fit on ring, we are done
Loop Until HolesOnRing = 0
End Function

public static SelectionSet ChainCircles(Document document, SelectionSet selectionSet, double startAngle)
{
var chainCircles = SelectionSetHelper.AddUniqueSelectionSet(document.SelectionSets, "ChainCircles");
for (var i = 1; i <= selectionSet.Count; i++)
{
var graphicObject = selectionSet[i] as IGraphicObject;
if (graphicObject.GraphicObjectType != espGraphicObjectType.espCircle)
{
continue;
}
var circle = graphicObject as Circle;
var featureChain = document.FeatureChains.Add(document.GetPoint(circle.CenterPoint.X + circle.Radius * Math.Cos(startAngle), circle.CenterPoint.Y + circle.Radius * Math.Sin(startAngle), circle.CenterPoint.Z));
featureChain.Add(document.GetArc(circle.CenterPoint, circle.Radius, startAngle, startAngle + 2 * Math.PI));
chainCircles.Add(featureChain);
}
return chainCircles;
}

Public Function ChainCircles(document As Esprit.Document, selectionSetObj As Esprit.SelectionSet, startAngle As Double) As Esprit.SelectionSet
Dim chainCrcl As Esprit.SelectionSet = SelectionSetHelper.AddUniqueSelectionSet(document, "ChainCircles")
' loop through all GraphicObjects in the SelectionSet
For i As Integer = 1 To selectionSetObj.Count
If selectionSetObj.Item(i).GraphicObjectType = espGraphicObjectType.espCircle Then
' put a FeatureChain on the Circles
Dim circle As Esprit.Circle = selectionSetObj.Item(i)
' start at the appropriate angle, and and a full 360 degree arc
Dim featureChain As Esprit.FeatureChain =
document.FeatureChains.Add(document.GetPoint(circle.CenterPoint.X + circle.Radius * Math.Cos(startAngle), circle.CenterPoint.Y + circle.Radius * Math.Sin(startAngle), circle.CenterPoint.Z))
Call featureChain.Add(document.GetArc(circle.CenterPoint, circle.Radius, startAngle, startAngle + 2 * Math.PI))
Call chainCrcl.Add(featureChain)
End If
Next
Return chainCrcl
End Function

Essentially, any combination of graphic objects can be associated as a SelectionSet, and by giving each one a unique name, you are labeling them in the same way that other graphic objects are labeled. This means that later on, there is the potential for someone to pick one of these items (such as a BoltHole) and view or even edit the parameters that were used to create it.
