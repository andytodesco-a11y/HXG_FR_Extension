Application Events

An advantage of using classes is that they can also contain event procedures. Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

For example, for the Application object, some of the events are AfterDocumentClose, AfterDocumentOpen and BeforeQuit.

The Tutorial demostrates common Application object events.

Define a class and add the event procedures:

class ApplicationEventsHandler
{
private readonly Esprit.Application _application;
private readonly string _messageBoxTitle;
public ApplicationEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "ApplicationEventsTutorial";
_application.BeforeQuit += BeforeQuit;
_application.OnError += OnError;
_application.OnMessageBox += OnMessageBox;
_application.BeforePromptDisplayed += BeforePromptDisplayed;
_application.OnPaneNotifyChange += OnPaneNotifyChange;
_application.OnActivate += OnActivate;
_application.AfterOptionsChanged += AfterOptionsChanged;
}
private void AfterOptionsChanged()
{
MessageBox.Show("AfterOptionsChanged called on clicking \"OK\" on Options dialog", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnActivate(int State, bool Minimized)
{
if (State == 1)
{
MessageBox.Show("OnActivate called on modifying main window state", _messageBoxTitle, MessageBoxButton.OK);
}
}
private void OnPaneNotifyChange(string PaneKey, EspritConstants.espPaneNotifyChangeType Action)
{
if (Action == EspritConstants.espPaneNotifyChangeType.espPaneNotifyChangeTypeFloated)
{
MessageBox.Show("OnPaneNotifyChange called on moving floated pane", _messageBoxTitle, MessageBoxButton.OK);
}
}
private void BeforePromptDisplayed(int CmdId, int SubCmdId, ref string PromptStr)
{
MessageBox.Show("BeforePromptDisplayed called on changing command state string", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnMessageBox(string Description, ref bool Override)
{
MessageBox.Show("OnMessageBox called before Application notification/error message box", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnError(int ErrorCode, string Description, ref bool Override)
{
MessageBox.Show("OnError called before Application notification/error message box", _messageBoxTitle, MessageBoxButton.OK);
}
private void BeforeQuit()
{
MessageBox.Show("BeforeQuit called before closing Application", _messageBoxTitle, MessageBoxButton.OK);
}
}

Dim WithEvents MyApp As Esprit.Application
Dim MessageBoxTitle As String
Private Sub Class_Initialize()
Set MyApp = Application
MessageBoxTitle = "ApplicationEventsTutorial"
End Sub
Private Sub MyApp_BeforeQuit()
Call MsgBox("BeforeQuit called before closing Application", vbOKOnly, MessageBoxTitle)
End Sub
Private Sub MyApp_OnError(ByVal ErrorCode As Long, ByVal Description As String, ByRef Override As Boolean)

End Sub

End Sub
