Exporting and Importing Point Data

Some other tutorials show how to make curves from point data. That’s fine if you already have the point data or can import it from another system via IGES or another CAD format. But what if you have a plain text file ASCII file of point data. How could you import that?

We can use the following code to export point data:

private void ExportPoints()
{
var filePath = Path.Combine(_application.TempDir, _fileName + ".csv");
using (var outStream = new StreamWriter(filePath))
{
foreach (Esprit.Point point in Document.Points)
{
outStream.WriteLine($"{point.X} {point.Y} {point.Z}");
}
}
}

private void ImportPoints()
{
var filePath = Path.Combine(_application.TempDir, _fileName + ".csv");
if (!File.Exists(filePath))
{
System.Windows.MessageBox.Show("Import file does not exist");
return;
}
using (var inStream = new StreamReader(filePath))
{
var charSeparators = new char[] { ' ' };
var data = string.Empty;
while ((data = inStream.ReadLine()) != null)
{
var pointData = data.Split(charSeparators, StringSplitOptions.RemoveEmptyEntries);
if (pointData.Length != 3)
{
continue;
}
if (!double.TryParse(pointData[0], out var x))
{
continue;
}
if (!double.TryParse(pointData[1], out var y))
{
continue;
}
if (!double.TryParse(pointData[2], out var z))
{
continue;
}
if (x != 0 || y != 0 || z != 0)
{
GraphicObjectHelper.SetDefaultAttributes(Document.Points.Add(x, y, z) as Esprit.GraphicObject, _application.Configuration);
}
}
}
Document.Refresh();
}
Private Sub ImportPoints()
Dim filePath As String = Path.Combine(_application.TempDir, _fileName + ".csv")
If Not File.Exists(filePath) Then
System.Windows.MessageBox.Show("Import file does not exist")
Return
End If
Dim inStream = New StreamReader(filePath)
Dim charSeparators() = New Char() {" "}
Dim data As String = inStream.ReadLine()
While data <> Nothing
Dim pointData = data.Split(charSeparators, StringSplitOptions.RemoveEmptyEntries)
If pointData.Length <> 3 Then
Continue While
End If
Dim x As Double = 0.0
Dim y As Double = 0.0
Dim z As Double = 0.0
If Not Double.TryParse(pointData(0), x) Then
Continue While
End If
If Not Double.TryParse(pointData(1), y) Then
Continue While
End If
If Not Double.TryParse(pointData(2), z) Then
Continue While
End If
If x <> 0.0 OrElse y <> 0.0 OrElse z <> 0.0 Then
GraphicObjectHelper.SetDefaultAttributes(Document.Points.Add(x, y, z), _application.Configuration)
End If
data = inStream.ReadLine()
End While
inStream.Close()
Document.Refresh()
End Sub
Sub ImportPoints()
Dim PointDataFilename As String
PointDataFilename = Application.TempDir & "points.csv"
If Dir(PointDataFilename) = "" Then Exit Sub
Open PointDataFilename For Input As #1
Dim X As Double, Y As Double, Z As Double
Do Until EOF(1)
Input #1, X, Y, Z
If (X <> 0) Or (Y <> 0) Or (Z <> 0) Then
Call DefaultAttributes(Document.Points.Add(X, Y, Z))
End If
Loop
Close 1
Document.Refresh
End Sub
