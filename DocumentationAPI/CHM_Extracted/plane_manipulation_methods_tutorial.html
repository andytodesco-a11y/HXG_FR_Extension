Introduction to Plane Manipulation Methods

This Tutorials shows how to modify location of an active workplane. The following code prompts the user to choose a point and then moves a workplane to that point using the Translate function.

private void TranslateWorkPlaneToPointDemo()
{
Esprit.Point point = null;
try
{
point = Document.GetPoint("Select Destination Point");
}
catch (System.Runtime.InteropServices.COMException)
{
// Element was not set
return;
}
if (point == null)
{
return;
}
var plane = Document.ActivePlane;
plane.Translate(point.X - plane.X, point.Y - plane.Y, point.Z - plane.Z);
plane.Activate();
}
Private Sub TranslateWorkPlaneToPointDemo()
Dim point As Esprit.Point = Nothing

Sub TranslateWorkPlaneToPointDemo()
Dim P As Esprit.Point
On Error Resume Next
Set P = Document.GetPoint("Select Destination Point")
If P Is Nothing Then Exit Sub ' in case user presses escape
Dim PL As Esprit.Plane
Set PL = TranslateWorkPlaneToPoint(P)
PL.Activate
Document.Refresh
End Sub
Public Function TranslateWorkPlaneToPoint(Destination As Esprit.Point) As Esprit.Plane
With Document.ActivePlane
Call .Translate(Destination.X - .X, Destination.Y - .Y, Destination.Z - .Z)
End With
Set TranslateWorkPlaneToPoint = Document.ActivePlane
End Function

private void TranslateWorkPlaneToPointCoordinatesDemo()
