The following code shows how to Load the Iges file with the Merge option and fit the graphics

C#
VB.NET
VBA
private void ImportIges()
{
_application.Configuration.IgesFile.LogFileMode = true;
var igesFilePath = string.Empty;
using (var openFileDialog = new System.Windows.Forms.OpenFileDialog())
{
var espritDir = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeEspritDrawings);
openFileDialog.InitialDirectory = espritDir;
openFileDialog.Filter = "Iges Files (*.igs)|*.igs|All Files (*.*)|*.*";
openFileDialog.FilterIndex = 1;
openFileDialog.Title = "Open Iges File";
openFileDialog.ShowDialog();
igesFilePath = openFileDialog.FileName;
}
if (igesFilePath == string.Empty)
{
return;
}
Document.MergeFile(igesFilePath);
Document.Windows.ActiveWindow.Fit();
Document.Refresh();
}
Private Sub ImportIges()
_application.Configuration.IgesFile.LogFileMode = True
Dim espritDir = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeEspritDrawings)
Dim openFileDialog = New System.Windows.Forms.OpenFileDialog() With {
.InitialDirectory = espritDir,
.Filter = "Iges Files (*.igs)|*.igs|All Files (*.*)|*.*",
.FilterIndex = 1,
.Title = "Open Iges File"
}
openFileDialog.ShowDialog()
Dim igesFilePath As String = openFileDialog.FileName
If igesFilePath = String.Empty Then
Return
End If
Document.MergeFile(igesFilePath)
Document.Windows.ActiveWindow.Fit()
Document.Refresh()
End Sub
Sub ImportIges()
Application.Configuration.IgesFile.LogFileMode = True
Dim IgesFileName As String, LogFileFolder As String, LogFileName As String
Load frmCommonDialog1
frmCommonDialog1.Hide
With frmCommonDialog1.CommonDialog1
.InitDir = Application.Configuration.GetFileDirectory(espFileTypeEspritDrawings)
.Filter = "Iges Files (*.igs)|*.igs|All Files (*.*)|*.*"
.FilterIndex = 1
.DialogTitle = "Open Iges File"
.ShowOpen
End With
IgesFileName = frmCommonDialog1.CommonDialog1.FileName
Unload frmCommonDialog1
If IgesFileName = "" Then Exit Sub ' In case Cancel button was clicked
' Load the file with the Merge option and fit the graphics
Call Document.MergeFile(IgesFileName)
Document.Windows.ActiveWindow.Fit
Document.Refresh
End Sub
