Feature Recognition: Face Profile Features

This tutorial shows how to utilize the CreateFaceProfileFeatures method of FeatureRecognition. It simply needs a SelectionSet object, giving the bodies on which to find the face profiles, and an array of Plane objects, which will define the planes in which those features will lie.

public void FindAllFaceProfilesOnXYZ()
{
var set = SelectionSetHelper.GetSelectionSet(Document, "Temp");
set.RemoveAll();
foreach (Esprit.Solid solid in Document.Solids)
{
set.Add(solid);
}
if (set.Count == 0)
{
MessageBox.Show("There are no solids in this file.", "FeatureRecognitionFaceProfileFeaturesTutorial", MessageBoxButton.OK);
return;
}
var plane = Document.Planes["XYZ"];
if (plane == null)
{
plane = PlaneHelper.GetPlane(Document, "Temp XYZ");
PlaneHelper.SetPlaneVectors(ref plane, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1);
}
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = (Document.SystemUnit == EspritConstants.espUnitType.espInch)
? 0.0001
: 0.0025;
try
{
Document.FeatureRecognition.CreateFaceProfileFeatures(set, new Esprit.Plane[1] { plane });
}
catch (Exception)
{
MessageBox.Show("Fail on creationg face profile features.", "FeatureRecognitionFaceProfileFeaturesTutorial", MessageBoxButton.OK);
return;
}
}
public static SelectionSet GetOrAddSelectionSet(ISelectionSets selectionSets, string name)
{
var set = selectionSets[name] ?? selectionSets.Add(name);
return set;
}
public static Esprit.Plane GetPlane(Esprit.Document document, string name)
{
Esprit.Plane plane = null;
foreach (Esprit.Plane p in document.Planes)
{
if (p.Name == name)
{
plane = p;
break;
}
}
if (plane == null)
{
plane = document.Planes.Add(name);
}
return plane;
}
public static void SetPlaneVectors(ref Esprit.Plane plane,
double x, double y, double z,
double Ux, double Uy, double Uz,
double Vx, double Vy, double Vz,
double Wx, double Wy, double Wz)
{
plane.X = x;
plane.Y = y;
plane.Z = z;
plane.Ux = Ux;
plane.Uy = Uy;
plane.Uz = Uz;
plane.Vx = Vx;
plane.Vy = Vy;
plane.Vz = Vz;
plane.Wx = Wx;
plane.Wy = Wy;
plane.Wz = Wz;
}
Public Sub FindAllFaceProfilesOnXYZ()
Dim selectionSet = SelectionSetHelper.GetSelectionSet(Document, "Temp")
selectionSet.RemoveAll()
For Each solid As Esprit.Solid In Document.Solids
selectionSet.Add(solid)
Next
If selectionSet.Count = 0 Then
System.Windows.MessageBox.Show("There are no solids in this file.", "FeatureRecognitionFaceProfileFeaturesTutorial", vbOK)
Return
End If
Dim plane(1) As Esprit.Plane
plane(0) = Document.Planes("XYZ")
If plane(0) Is Nothing Then
plane(0) = PlaneHelper.GetPlane(Document, "Temp XYZ")
PlaneHelper.SetPlaneVectors(plane(0), 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1)
End If
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = IIf(Document.SystemUnit = EspritConstants.espUnitType.espInch, 0.0001, 0.0025)
Try
Document.FeatureRecognition.CreateFaceProfileFeatures(selectionSet, plane)
Catch ex As Exception
System.Windows.MessageBox.Show("Fail on creating face profile features.", "FeatureRecognitionFaceProfileFeaturesTutorial", vbOK)
Return
End Try
End Sub
Public Function GetOrAddSelectionSet(document As Esprit.Document, name As String) As Esprit.SelectionSet
Dim selSet As Esprit.SelectionSet = document.SelectionSets.Item(name)
If selSet Is Nothing Then
selSet = document.SelectionSets.Add(name)
End If
Return selSet
End Function
Public Function GetPlane(document As Esprit.Document, name As String) As Esprit.Plane
Dim plane As Esprit.Plane = Nothing
For Each p As Esprit.Plane In document.Planes
If p.Name = name Then
plane = p
Exit For
End If
Next
If plane Is Nothing Then
plane = document.Planes.Add(name)
End If
Return plane
End Function

Sub FindAllFaceProfilesOnXYZ()
' add all of the solids in the file to a selection set
Dim SL As Esprit.Solid
Dim SS As Esprit.SelectionSet
Set SS = GetSelectionSet("Temp")
For Each SL In Document.Solids
Call SS.Add(SL)
Next
' check to make sure we have some solids
If SS.Count = 0 Then
Call MsgBox("There are no solids in this file. This macro is only for solids.", vbOKOnly, "Error")
Exit Sub
End If
' try to use the XYZ plane, note that we must declare it as an array of one to pass later
Dim PL(1 To 1) As Esprit.Plane
On Error Resume Next ' skip in case user deleted
Set PL(1) = Document.Planes.Item("XYZ")
On Error GoTo 0 ' disable further error trapping
If PL(1) Is Nothing Then
' make a plane parallel to XYZ if plane was not set
Set PL(1) = GetPlane("Temp XYZ")
SetPlaneVectors(PL(1), 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1).IsWork = False
End If
' perform face profile creation with an appropriate tolerance
If Document.SystemUnit = espInch Then
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.0001
Else
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.0025
End If
Call Document.FeatureRecognition.CreateFaceProfileFeatures(SS, PL)
End Sub
Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)
On Error GoTo 0
If GetSelectionSet Is Nothing Then
Set GetSelectionSet = Document.SelectionSets.Add(SSName)
End If
End Function
Function GetPlane(PlaneName As String) As Esprit.Plane
On Error Resume Next
Set GetPlane = Document.Planes.Item(PlaneName)
On Error GoTo 0
If GetPlane Is Nothing Then
Set GetPlane = Document.Planes.Add(PlaneName)
End If
End Function
Function SetPlaneVectors(PlaneObj As Esprit.Plane, X As Double, Y As Double, Z As Double, Ux As Double, Uy As Double, Uz As Double, Vx As Double, Vy As Double, Vz As Double, Wx As Double, Wy As Double, Wz As Double) As Esprit.Plane
With PlaneObj
.X = X
.Y = Y
.Z = Z
.Ux = Ux
.Uy = Uy
.Uz = Uz
.Vx = Vx
.Vy = Vy
.Vz = Vz
.Wx = Wx
.Wy = Wy
.Wz = Wz
End With
Set SetPlaneVectors = PlaneObj
End Function

Before running the code open a file that contains a solid, or create one in the Solid Modeler. Look at the features that are created. Did your solid include both pocket/contour profiles and holes? If so, then you should get both FeatureChain and FeaturePtop objects created.
