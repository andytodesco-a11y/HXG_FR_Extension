Scan the Graphics Collection

This GraphicsCollection is similar in concept to all of the other collections in the ESPRIT EDGE Document. This tutorial will loop through the entire GraphicsCollection and output a summary of certain information.

Notice that this code uses the Count property. Count is a value that is found on all collections. Almost always, that Count will tell you the exact number of elements in that collection. For example, Points.Count will tell you exactly how many points you have in the graphics of ESPRIT EDGE. Something to be aware of that you should notice in the results of this code is that the total number of elements that we can examine in the API is always less than the value given by the Count of the GraphicsCollection. The GraphicsCollection refers to the entire database of a ESPRIT EDGE document, and it contains many other items besides those that are exposed as graphic objects in the API. Those items are unknown to us. That is why the GraphicsCollection.Count does not reflect the total number of GraphicObject objects in the Document. We can bypass those unknown elements by testing their GraphicObjectType to see if it has a value of espUnknown.

This tutorial also introduces the use of the OutputWindow. The OutputWindow is a window in ESPRIT EDGE that can be used to display text information.

public void ScanGraphicsCollection()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", $"Database contains up to {Document.GraphicsCollection.Count} GraphicObject items");
var validObjectsCount = 0;
foreach (Esprit.GraphicObject graphicObject in Document.GraphicsCollection)
{
if (graphicObject.GraphicObjectType != espGraphicObjectType.espUnknown)
{
validObjectsCount++;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", $"Element {validObjectsCount} is {graphicObject.TypeName} {graphicObject.Key}");
}
}
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", $"Total Number of known GraphicObject items is {validObjectsCount}");
}
Public Sub ScanGraphicsCollection()
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
Dim validCount As Long = 0
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Database contains up to " & Document.GraphicsCollection.Count & " GraphicObject items")
For Each graphicObject As Esprit.GraphicObject In Document.GraphicsCollection
If graphicObject.GraphicObjectType <> espGraphicObjectType.espUnknown Then
validCount += 1
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Element " & validCount & " is " & graphicObject.TypeName & " " & graphicObject.Key)
End If
Next
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Total Number of known GraphicObject items is " & validCount)
End Sub
Sub ScanGraphicsCollection()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
Dim ValidCount As Long
ValidCount = 0
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Database contains up to " & Document.GraphicsCollection.Count & " GraphicObject items")
End With
Dim I As Long
For I = 1 To Document.GraphicsCollection.Count
Dim G As Esprit.GraphicObject
Set G = Document.GraphicsCollection.Item(I)
If G.GraphicObjectType <> espUnknown Then
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Item at position " & I & " is " & G.TypeName & " " & G.Key)
End With
ValidCount = ValidCount + 1
End If
Next
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanGraphicsCollectionTutorial", "Total Number of known GraphicObject items is " & ValidCount)
End With
End Sub
