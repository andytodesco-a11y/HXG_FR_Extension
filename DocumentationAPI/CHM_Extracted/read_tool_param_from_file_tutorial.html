Read Tool Parameters from a File

This tutorial shows how to load all of the tools and all of their parameters from file.

public void ReadToolParameters()
{
var folderPath = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage);
var tools = Document.Tools as EspritTools.Tools;
var dirInfo = new DirectoryInfo(folderPath);
var filePaths = dirInfo.GetFiles("*.test");
foreach (var filePath in filePaths)
{
tools.LoadTools(filePath.FullName);
}
}
Public Sub ReadToolParameters()
Dim toolTech As EspritTools.Tools = Document.Tools
Dim toolFolder As String = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage)
Dim dirInfo As New DirectoryInfo(toolFolder)
Dim filePath As IO.FileInfo() = dirInfo.GetFiles("*.test")
For Each fileInfo As FileInfo In filePath
toolTech.LoadTools(fileInfo.FullName)
Next
End Sub
Sub ReadToolParameters()
Dim ToolTech As EspritTools.Tools
Set ToolTech = Document.Tools
Dim ToolFolder As String, FileMask As String
ToolFolder = Application.Configuration.GetFileDirectory(espFileTypeToolPage)
FileMask = "\WriteToolParametersToFileTutorial_*.test"
Dim FilePath As String
FilePath = Dir(ToolFolder & FileMask)
Do While FilePath <> ""
ToolTech.LoadTools (ToolFolder & "\" & FilePath)
FilePath = Dir
Loop
End Sub
