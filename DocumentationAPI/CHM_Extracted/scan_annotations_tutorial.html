Scan Annotations

This tutorial introduces Annotations collection - an important group of GraphicObject objects related to dimensioning, e.g. Notes, Leader, Hatch and Dimension objects. The example of code simply loops through the Annotations collection and displays certain information about each item.

As with several other collections, the object returned from the Annotations collection is a generic Annotation object. The Annotation object has both an AnnotationType enumerated property and a TypeName string to describe the specific type of object that it is. We can use either AnnotationType or TypeName to cast the Annotation as a specific Notes, Leader, Hatch, or Dimension object.

public void ScanAnnotations()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
var index = 0;
foreach(Esprit.Annotation annotation in Document.Annotations)
{
index++;
switch (annotation.AnnotationType)
{
case EspritConstants.espAnnotationType.espAnnotationDimension:
var annotationDimension = annotation as Esprit.Dimension;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"Item {index} is Dimension {annotation.Key}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"Dimension {annotationDimension.Key} contains {annotationDimension.AnnotationSegments.Count} segment(s) and {annotationDimension.Notes.Count} note(s).");
break;
case EspritConstants.espAnnotationType.espAnnotationHatch:
var annotationHatch = annotation as Esprit.Hatch;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"Item {index} is Hatch {annotation.Key}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"{Environment.NewLine}Hatch {annotationHatch.Key} contains {annotationHatch.AnnotationSegments.Count} segment(s).");
break;
case EspritConstants.espAnnotationType.espAnnotationLeader:
var annotationLeader = annotation as Esprit.Leader;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"Item {index} is Leader {annotation.Key}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"{Environment.NewLine}Leader {annotationLeader.Key} contains {annotationLeader.LeaderPoints.Count} leader point(s).");
break;
case EspritConstants.espAnnotationType.espAnnotationNote:
var annotationNotes = annotation as Esprit.Notes;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"Item {index} is Note {annotation.Key}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", $"{Environment.NewLine}Note {annotationNotes.Key} contains {annotationNotes.Count} line(s) of text(s).");
break;
}
}
if (index == 0)
{
MessageBox.Show("There are no Annotations", "ScanAnnotationsTutorial");
}
}
Public Sub ScanAnnotations()
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
Dim index As Integer = 0
For Each annotation As Esprit.Annotation In Document.Annotations
index += 1
Select Case annotation.AnnotationType
Case EspritConstants.espAnnotationType.espAnnotationDimension
Dim annotationDimension As Esprit.Dimension = annotation
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & index & " is Dimension " & annotation.Key)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Dimension " & annotationDimension.Key & " contains " & annotationDimension.AnnotationSegments.Count & " segment(s) and " & annotationDimension.Notes.Count & " note(s).")
Case EspritConstants.espAnnotationType.espAnnotationHatch
Dim annotationHatch As Esprit.Hatch = annotation
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & index & " is Hatch " & annotation.Key)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Hatch " & annotationHatch.Key & " contains " & annotationHatch.AnnotationSegments.Count & " segments.")
Case EspritConstants.espAnnotationType.espAnnotationLeader
Dim annotationLeader As Esprit.Leader = annotation
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & index & " is Leader " & annotation.Key)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Leader " & annotationLeader.Key & " contains " & annotationLeader.LeaderPoints.Count & " leader point(s).")
Case EspritConstants.espAnnotationType.espAnnotationNote
Dim annotationNotes As Esprit.Notes = annotation
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & index & " is Note " & annotation.Key)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Note " & annotationNotes.Key & " contains " & annotationNotes.Count & " line(s) of text.")
End Select
Next
If index = 0 Then
Call MsgBox("There are no Annotations", vbOKOnly)
End If
End Sub
Sub ScanAnnotations()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
Dim An As Esprit.Annotation
Dim I As Long
If Document.Annotations.Count = 0 Then
Call MsgBox("There are no Annotations", vbOKOnly)
Exit Sub
End If
For I = 1 To Document.Annotations.Count
Set An = Document.Annotations.Item(I)
' type cast the annonation to a specific object based on the type
Select Case An.AnnotationType
Case espAnnotationDimension
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & I & " is Dimension " & An.Key)
Dim D As Esprit.Dimension
Set D = An
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Dimension " & D.Key & " contains " & D.AnnotationSegments.Count & " segment(s) and " & D.Notes.Count & " note(s).")
Case espAnnotationHatch
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & I & " is Hatch " & An.Key)
Dim H As Esprit.Hatch
Set H = An
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Hatch " & H.Key & " contains " & H.AnnotationSegments.Count & " segments.")
Case espAnnotationLeader
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & I & " is Leader " & An.Key)
Dim Ld As Esprit.Leader
Set Ld = An
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Leader " & Ld.Key & " contains " & Ld.LeaderPoints.Count & " leader point(s).")
Case espAnnotationNote
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Item " & I & " is Note " & An.Key)
Dim N As Esprit.Notes
Set N = An
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanAnnotationsTutorial", "Note " & N.Key & " contains " & N.Count & " line(s) of text.")
End Select
Next
End Sub
