Find Tool Parameters

The tutorial A Closer Look at the Parameter Object showed how to loop through Technology objects and extract Parameter information. One piece of information collected was the Property Name. That name is useful because it can be fed back into Technology.Item to find a particular parameter.

Instead of making a list of all parameters, suppose that we only want to find the property names of the ones related to tools. We can test the string returned for the TypeName of the Technology object to see if it contains the word Tool. If it does, then we know that Technology is for a tool. Furthermore, we could test whether it was a "LatheTool" or a "MillTool".

private void FindToolPropertyNames()
{
_toolLathePropertyNames.Clear();
_toolMillPropertyNames.Clear();
var technologyUtility = Document.TechnologyUtility as EspritTechnology.TechnologyUtility;
foreach (espTechnologyType technologyType in Enum.GetValues(typeof(espTechnologyType)))
{
try
{
var technology = technologyUtility.CreateTechnology(technologyType, Document.SystemUnit);
if (technology != null)
{
var techName = Microsoft.VisualBasic.Information.TypeName(technology);
techName = techName.Substring(1);
if (techName.StartsWith("ToolLathe"))
{
for (var j = 1; j <= technology.Count; j++)
{
if (!_toolLathePropertyNames.Contains(technology[j].Name))
{
_toolLathePropertyNames.Add(technology[j].Name);
}
}
}
else if (techName.StartsWith("ToolMill"))
{
for (var j = 1; j <= technology.Count; j++)
{
if (!_toolMillPropertyNames.Contains(technology[j].Name))
{
_toolMillPropertyNames.Add(technology[j].Name);
}
}
}
}
}
catch (Exception)
{ }
}
}
private void DisplayToolLathePropertyNames()
{
var stringBuilder = new StringBuilder();
stringBuilder.AppendLine("These are the Property Names for all Lathe Tools:");
foreach (var name in _toolLathePropertyNames)
{
stringBuilder.AppendLine($"\t{name}");
}
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FindToolParamsTutorial", stringBuilder.ToString());
}

Private Sub FindToolPropertyNames()
_toolLathePropertyNames.Clear()
_toolMillPropertyNames.Clear()
Dim technologyUtility As EspritTechnology.TechnologyUtility = Document.TechnologyUtility
For Each technologyType As espTechnologyType In [Enum].GetValues(GetType(espTechnologyType))
Try
Dim technology = technologyUtility.CreateTechnology(technologyType, Document.SystemUnit)
If Not technology Is Nothing Then
Dim techName As String = Microsoft.VisualBasic.Information.TypeName(technology)
techName = techName.Substring(1)
If techName.StartsWith("ToolLathe") Then
For j As Integer = 1 To technology.Count
If Not _toolLathePropertyNames.Contains(technology(j).Name) Then
_toolLathePropertyNames.Add(technology(j).Name)
End If
Next
ElseIf techName.StartsWith("ToolMill") Then
For j As Integer = 1 To technology.Count
If Not _toolMillPropertyNames.Contains(technology(j).Name) Then
_toolMillPropertyNames.Add(technology(j).Name)
End If
Next
End If
End If
Catch ex As Exception
End Try
Next
End Sub

Dim ToolLathePropertyNames As Collection
Dim ToolMillPropertyNames As Collection
Sub FindToolPropertyNames()
Set ToolLathePropertyNames = New Collection
Set ToolMillPropertyNames = New Collection
Dim TechUtil As EspritTechnology.TechnologyUtility
Set TechUtil = Document.TechnologyUtility
' loop through all possible technologies
' Can check EspritConstants.espTechnologyType for limits if desired
' or, just to be sure, try a large range like 0 to 999
Dim TType As Long
For TType = 0 To 999
Dim T As EspritTechnology.Technology
Set T = Nothing
On Error Resume Next
Set T = TechUtil.CreateTechnology(TType, Document.SystemUnit)
On Error GoTo 0
If Not (T Is Nothing) Then
Dim TechName As String
TechName = TypeName(T)
TechName = Mid$(TechName, 2) ' cut off the "I" at the start
' check to see if it is a tool type
If InStr(1, TechName, "Tool") = 1 Then
' loop through all of the parameters on this tech object
Dim PNum As Long
Dim P As EspritTechnology.Parameter
For PNum = 1 To T.Count
Set P = T.Item(PNum)
' add to the collection of the appropriate type if it is not already a member
Dim M As String
If InStr(1, TechName, "ToolLathe") = 1 Then
M = ""
On Error Resume Next
M = ToolLathePropertyNames.Item(P.Name)
On Error GoTo 0
If M = "" Then Call ToolLathePropertyNames.Add(P.Name, P.Name)
End If
If InStr(1, TechName, "ToolMill") = 1 Then
M = ""
On Error Resume Next
M = ToolMillPropertyNames.Item(P.Name)
On Error GoTo 0
If M = "" Then Call ToolMillPropertyNames.Add(P.Name, P.Name)
End If
Next
End If
End If
Next
End Sub
Sub DisplayToolLathePropertyNames()
If ToolLathePropertyNames Is Nothing Then FindToolPropertyNames
' Display the list on the output window
With Application.EventWindow
.Visible = True
Dim stringBuilder As String
stringBuilder = "These are the Property Names for all Lathe Tools:"
Dim I As Long
For I = 1 To ToolLathePropertyNames.Count
stringBuilder = stringBuilder & vbTab & ToolLathePropertyNames.Item(I)
Next
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FindToolParamsTutorial", stringBuilder)
End With
End Sub
