Application Document Events

An advantage of using classes is that they can also contain event procedures. Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

For example, for the Application object, some of the events are AfterDocumentClose, AfterDocumentOpen and BeforeQuit.

The Tutorial demostrates Application object events which deal with the application Document behavior.

Define a class and add the event procedures:

VB.NET
VBA
class ApplicationDocumentEventsHandler
{
private readonly Esprit.Application _application;
private readonly string _messageBoxTitle;
public ApplicationDocumentEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "ApplicationDocumentTutorial";
_application.AfterDocumentOpen += AfterDocumentOpen;
_application.AfterNewDocumentOpen += AfterNewDocumentOpen;
_application.AfterTemplateOpen += AfterTemplateOpen;
_application.AfterDocumentClose += AfterDocumentClose;
_application.BeforeDocumentOpen += BeforeDocumentOpen;
_application.OnLegacyFileOpen += OnLegacyFileOpen;
if (_application.Document != null)
{
var result = MessageBox.Show("Close current document?", _messageBoxTitle, MessageBoxButton.YesNo);
if (result == MessageBoxResult.Yes)
{
_application.CloseDocument();
}
}
}
private void OnLegacyFileOpen(string LegacyFilePath, ref string MachineFilePath, ref bool LoadOperations, ref bool OverrideParameters)
{
MessageBox.Show("OnLegacyFileOpen called on opening a *.esp file", _messageBoxTitle, MessageBoxButton.OK);
}
private void BeforeDocumentOpen(string filePath)
{
MessageBox.Show("BeforeDocumentOpen called on opening document", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterDocumentClose()
{
MessageBox.Show("AfterDocumentClose called on closing document", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterTemplateOpen(string filePath)
{
MessageBox.Show("AfterTemplateOpen called on opening a template or new document", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterNewDocumentOpen()
{
MessageBox.Show("AfterNewDocumentOpen called on opening new document", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterDocumentOpen(string filePath)
{
MessageBox.Show("AfterDocumentOpen called on opening a document file", _messageBoxTitle, MessageBoxButton.OK);
}
}
Public Class ApplicationDocumentEventsHandler
Private ReadOnly _application As ESPRIT.Application
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As ESPRIT.Application)
_application = app
_messageBoxTitle = "ApplicationDocumentTutorial"
With _application
AddHandler .AfterDocumentOpen, AddressOf Me.AfterDocumentOpen
AddHandler .AfterNewDocumentOpen, AddressOf Me.AfterNewDocumentOpen
AddHandler .AfterTemplateOpen, AddressOf Me.AfterTemplateOpen
AddHandler .AfterDocumentClose, AddressOf Me.AfterDocumentClose
AddHandler .OnLegacyFileOpen, AddressOf Me.OnLegacyFileOpen
AddHandler .BeforeDocumentOpen, AddressOf Me.BeforeDocumentOpen
End With
If Not _application.Document Is Nothing Then
Dim result = MessageBox.Show("Close current document?", _messageBoxTitle, MessageBoxButton.YesNo)
If result = MessageBoxResult.Yes Then
_application.CloseDocument()
End If
End If
End Sub
Private Sub AfterDocumentOpen(FilePath As String)
MessageBox.Show("AfterDocumentOpen called on opening a document file", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub AfterNewDocumentOpen()
MessageBox.Show("AfterNewDocumentOpen called on opening new document", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub AfterTemplateOpen(FilePath As String)
MessageBox.Show("AfterTemplateOpen called on opening a template or new document", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub AfterDocumentClose()
MessageBox.Show("AfterDocumentClose called on closing document", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub OnLegacyFileOpen(LegacyFilePath As String, ByRef MachineFilePath As String, ByRef LoadOperations As Boolean, ByRef OverrideParameters As Boolean)
MessageBox.Show("OnLegacyFileOpen called on opening a *.esp file", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub BeforeDocumentOpen(FilePath As String)
MessageBox.Show("BeforeDocumentOpen called on opening document", _messageBoxTitle, MessageBoxButton.OK)
End Sub
End Class
Dim WithEvents MyApp As Esprit.Application
Dim MessageBoxTitle As String
Private Sub Class_Initialize()
Set MyApp = Application
MessageBoxTitle = "ApplicationDocumentTutorial"
If Not MyApp.Document Is Nothing Then
If MsgBox("Close current document?", vbYesNo, MessageBoxTitle) = vbYes Then
Call MyApp.CloseDocument()
End If
End If
End Sub
Private Sub MyApp_BeforeDocumentOpen(ByVal FilePath As String)
Call MsgBox("BeforeDocumentOpen called on opening Application document", vbOKOnly, MessageBoxTitle)
End Sub
