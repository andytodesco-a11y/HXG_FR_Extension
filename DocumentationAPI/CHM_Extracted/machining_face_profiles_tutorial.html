Machining Face Profiles

The Tutorial: Feature Recognition: Face Profile Features introduced the CreateFaceProfileFeatures method of FeatureRecognition and discussed how it can produce both FeatureChain objects on profiles and FeaturePtop objects connecting holes. This tutorial will show how to take both types of features and automatically machine on them.

Suppose, for example, that your solid parts typically have circular bores and cutouts of other shapes. On bores that are above a certain minimum diameter, you first need to drill a start hole, but then you want pocket both the bores and the cutouts using the same tool and settings. The only difference will be that we will plunge into the bores at the center where the drill hole was, but not bother with entering at a drill hole for the cutouts (for now at least).

Notice in the code below how the various Configuration settings are used to define and limit the scope of feature recognition. The configuration settings that apply are found in three objects, ConfigurationFeatureRecognition, ConfigurationHoleRecognition, and ConfigurationProfileRecognition.

public void MachineHoleProfiles()
{
var plane = Document.ActivePlane;
if (plane.Name == string.Empty || plane.Name == "UVW")
{
MessageBox.Show("Must activate a named plane. This code cannot use UVW.", "MachiningFaceProfilesTutorial", MessageBoxButton.OK);
return;
}
var set = SelectionSetHelper.GetSelectionSet(Document, "Temp");
set.RemoveAll();
Esprit.Solid solid = null;
try
{
solid = Document.GetAnyElement("Select Reference Solid", EspritConstants.espGraphicObjectType.espSolidModel) as Esprit.Solid;
}
catch(System.Runtime.InteropServices.COMException)
{
// Element was not set
return;
}
if (solid == null)
{
return;
}
var minPocketDiameter = (Document.SystemUnit == EspritConstants.espUnitType.espInch)
? 2.0
: 50.0;
if (!RequestUserInput("Enter the Minimum Diameter for Pocketing", "Minimum Diameter", ref minPocketDiameter))
{
return;
}
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = (Document.SystemUnit == EspritConstants.espUnitType.espInch)
? 0.0001
: 0.0025;
_application.Configuration.ConfigurationHoleRecognition.ConnectAllHoles = true;
_application.Configuration.ConfigurationHoleRecognition.MaxDiameterDefined = false;
_application.Configuration.ConfigurationHoleRecognition.MinDiameterDefined = true;
_application.Configuration.ConfigurationHoleRecognition.MinDiameter = minPocketDiameter;
_application.Configuration.ConfigurationProfileRecognition.CreateHoles = true;
_application.Configuration.ConfigurationProfileRecognition.CreateProfilesOnHoles = true;
set.Add(solid);
Array faceFeatures = null;
try
{
faceFeatures = Document.FeatureRecognition.CreateFaceProfileFeatures(set, new Esprit.Plane[1] { plane });
}
catch(Exception)
{
MessageBox.Show("Fail on creationg face profile features.", "MachiningFaceProfilesTutorial", MessageBoxButton.OK);
return;
}
Esprit.FeatureChain outsideProfile = null;
double outsideArea = 0;
Esprit.FeaturePtop drillFeature = null;
var holeProfiles = new List<Esprit.FeatureChain>();
var cutoutProfiles = new List<Esprit.FeatureChain>();
for (var i = faceFeatures.GetLowerBound(0); i <= faceFeatures.GetUpperBound(0); i++)
{
var graphicObject = faceFeatures.GetValue(i) as Esprit.GraphicObject;
switch (graphicObject.GraphicObjectType)
{
case EspritConstants.espGraphicObjectType.espFeatureChain:
var featureChain = graphicObject as Esprit.FeatureChain;
if (featureChain.Area > outsideArea)
{
if (outsideProfile != null)
{
if (outsideProfile.IsClosed && outsideProfile.Count == 1)
{
holeProfiles.Add(outsideProfile);
}
else
{
cutoutProfiles.Add(outsideProfile);
}
}
outsideProfile = featureChain;
outsideArea = outsideProfile.Area;
}
else
{
if (featureChain.IsClosed && featureChain.Count == 1)
{
holeProfiles.Add(featureChain);
}
else
{
cutoutProfiles.Add(featureChain);
}
}
break;
case EspritConstants.espGraphicObjectType.espFeaturePtop:
var featurePtop = graphicObject as Esprit.FeaturePtop;
if (drillFeature == null)
{
drillFeature = featurePtop;
}
else if (featurePtop.Count > drillFeature.Count)
{
drillFeature = featurePtop;
}
break;
}
}
if (holeProfiles.Count == 0 && cutoutProfiles.Count == 0)
{
return;
}
if (drillFeature != null)
{
var drill = new EspritTechnology.TechMillDrill();
if (Document.GUI.EditTechnology(drill) == EspritConstants.espGuiReturnVal.espGuiComplete)
{
try
{
var obj = Document.PartOperations.Add(drill, drillFeature) as Esprit.GraphicObject;
GraphicObjectHelper.SetDefaultAttributes(obj, _application.Configuration);
}
catch(System.Runtime.InteropServices.COMException e)
{
MessageBox.Show(e.Message);
}
}
}
EspritTechnology.TechMillPocket2 millPocket1 = new EspritTechnology.TechMillPocket2();
EspritTechnology.TechMillPocket2 millPocket2 = new EspritTechnology.TechMillPocket2();
EspritTechnology.TechMillPocket2 millPocket3 = new EspritTechnology.TechMillPocket2();
if (Document.GUI.EditTechnology(millPocket1) == EspritConstants.espGuiReturnVal.espGuiCancel)
{
return;
}
var genericTech = millPocket1 as EspritTechnology.Technology;
genericTech.CopyTo(millPocket2 as EspritTechnology.Technology, true);
genericTech.CopyTo(millPocket3 as EspritTechnology.Technology, true);
for (var i = 0; i < holeProfiles.Count; i++)
{
millPocket2.EntryMode = EspritConstants.espMillPocketEntryMode.espMillPocketEntryModePlungeAtPoint;
millPocket2.TotalDepth = holeProfiles[i].Depth;
var arc = holeProfiles[i].Item[1] as Esprit.Arc;
holeProfiles[i].LeadInPoint = arc.CenterPoint;
try
{
var op = Document.PartOperations.Add(millPocket2, holeProfiles[i]);
GraphicObjectHelper.SetDefaultAttributes(op as Esprit.GraphicObject, _application.Configuration);
}
catch (Exception e)
{
MessageBox.Show(e.Message);
}
}
for (var i = 0; i < cutoutProfiles.Count; i++)
{
if (millPocket3.EntryMode == EspritConstants.espMillPocketEntryMode.espMillPocketEntryModeHelicalAtPoint)
{
millPocket3.EntryMode = EspritConstants.espMillPocketEntryMode.espMillPocketEntryModeHelixProtected;
}
else if (millPocket3.EntryMode == EspritConstants.espMillPocketEntryMode.espMillPocketEntryModePlungeAtPoint)
{
millPocket3.EntryMode = EspritConstants.espMillPocketEntryMode.espMillPocketEntryModePlunge;
}
millPocket3.TotalDepth = cutoutProfiles[i].Depth;
try
{
var op = Document.PartOperations.Add(millPocket3, cutoutProfiles[i]);
GraphicObjectHelper.SetDefaultAttributes(op as Esprit.GraphicObject, _application.Configuration);
}
catch (Exception e)
{
MessageBox.Show(e.Message);
}
}
Document.SelectionSets.Remove(set.Name);
Document.Refresh();
}

Sub MachineHoleProfiles()
' use the current active plane
Dim Pl(1 To 1) As Esprit.Plane
Set Pl(1) = Document.ActivePlane
' must be a saved plane, not UVW though, so check
If Pl(1).Name = "" Or Pl(1).Name = "UVW" Then
Call MsgBox("Must activate a named plane. This macro cannot use UVW.", vbOKOnly, "Error")
Exit Sub
End If
' prompt the user to pick a solid
Dim SL As Esprit.Solid
On Error Resume Next
Set SL = Document.GetAnyElement("Select Reference Solid", espSolidModel)
If SL Is Nothing Then Exit Sub ' in case user presses escape
On Error GoTo 0 ' to disable further error trapping
' set an appropriate tolerance and other configuration settings
Dim MinPocketDiameter As Double
With Application.Configuration
If Document.SystemUnit = espInch Then
.ConfigurationFeatureRecognition.Tolerance = 0.0001
MinPocketDiameter = 2
Else
.ConfigurationFeatureRecognition.Tolerance = 0.0025
MinPocketDiameter = 50
End If
MinPocketDiameter = Abs(Val(InputBox("Enter the Minimum Diameter for Pocketing", "Minimum Diameter", MinPocketDiameter)))
With .ConfigurationHoleRecognition
.ConnectAllHoles = True
.MaxDiameterDefined = False
.MinDiameterDefined = True
.MinDiameter = MinPocketDiameter
End With
With .ConfigurationProfileRecognition
.CreateHoles = True
.CreateProfilesOnHoles = True
End With
End With
' add the solid to the selection set
Dim SS As Esprit.SelectionSet
Set SS = GetSelectionSet("Temp")
SS.RemoveAll
Call SS.Add(SL)
' perform face profile creation
Dim FaceFeatures() As Esprit.GraphicObject
FaceFeatures = Document.FeatureRecognition.CreateFaceProfileFeatures(SS, Pl)
' setup parameters for finding the outside profile and categorizing features
Dim OutsideProfile As Esprit.FeatureChain, OutsideArea As Double
OutsideArea = 0
Dim DrillFeature As Esprit.FeaturePtop
Dim HoleProfiles() As Esprit.FeatureChain, CutoutProfiles() As Esprit.FeatureChain
Dim HoleCount As Long, CutoutCount As Long
HoleCount = 0
CutoutCount = 0
' loop through all of the features
Dim I As Long
For I = LBound(FaceFeatures) To UBound(FaceFeatures)
Select Case FaceFeatures(I).GraphicObjectType
Case espFeatureChain
Dim FC As Esprit.FeatureChain
Set FC = FaceFeatures(I)
' check the area and retain for later
If FC.Area > OutsideArea Then
' found a new outside profile
If Not (OutsideProfile Is Nothing) Then
' check previous outside and add it as a hole or cutout
If (OutsideProfile.IsClosed) And (OutsideProfile.Count = 1) Then
' must be circular hole
HoleCount = HoleCount + 1
ReDim Preserve HoleProfiles(1 To HoleCount) As Esprit.FeatureChain
Set HoleProfiles(HoleCount) = OutsideProfile
Else
' must be non-circular cutout
CutoutCount = CutoutCount + 1
ReDim Preserve CutoutProfiles(1 To CutoutCount) As Esprit.FeatureChain
Set CutoutProfiles(CutoutCount) = OutsideProfile
End If
End If
Set OutsideProfile = FC
OutsideArea = OutsideProfile.Area
Else
' found an inside hole profile or cutout
If (FC.IsClosed) And (FC.Count = 1) Then
' must be circular hole
HoleCount = HoleCount + 1
ReDim Preserve HoleProfiles(1 To HoleCount) As Esprit.FeatureChain
Set HoleProfiles(HoleCount) = FC
Else
' must be non-circular cutout
CutoutCount = CutoutCount + 1
ReDim Preserve CutoutProfiles(1 To CutoutCount) As Esprit.FeatureChain
Set CutoutProfiles(CutoutCount) = FC
End If
End If
Case espFeaturePtop
Dim FP As Esprit.FeaturePtop
Set FP = FaceFeatures(I)
' only use the feature with the most elements
' that will be the one connecting all holes
If DrillFeature Is Nothing Then
Set DrillFeature = FP
Else
If FP.Count > DrillFeature.Count Then
Set DrillFeature = FP
End If
End If
End Select
Next
' if features were created, then get the technology
' drilling
If Not (DrillFeature Is Nothing) Then
Dim DR As EspritTechnology.TechMillDrill
Set DR = New EspritTechnology.TechMillDrill
If Document.GUI.EditTechnology(DR) = espGuiComplete Then
On Error Resume Next
Call DefaultAttributes(Document.PartOperations.Add(DR, DrillFeature))
End If
End If
' pocketing
If (HoleCount > 0) Or (CutoutCount > 0) Then
Dim MP(0 To 2) As EspritTechnology.TechMillPocket2
For I = 0 To 2
Set MP(I) = New EspritTechnology.TechMillPocket2
Next
If Document.GUI.EditTechnology(MP(0)) = espGuiCancel Then Exit Sub
End If
' make copies of pocket technologies, so that entry mode can be set differently
Dim GenericTech As EspritTechnology.Technology
Set GenericTech = MP(0)
Call GenericTech.CopyTo(MP(1), True)
Call GenericTech.CopyTo(MP(2), True)
' loop through holes and pocket them, using center for entry
For I = 1 To HoleCount
MP(1).EntryMode = espMillPocketEntryModePlungeAtPoint
MP(1).TotalDepth = HoleProfiles(I).Depth
' find the center of the circular hole
Dim A As Arc
Set A = HoleProfiles(I).Item(1)
' note how the lead-in point of the feature is used as the entry point of the operation
HoleProfiles(I).LeadInPoint = A.CenterPoint
On Error Resume Next
Call DefaultAttributes(Document.PartOperations.Add(MP(1), HoleProfiles(I)))
Next
' loop through cutouts and pocket them
For I = 1 To CutoutCount
' use whatever was originally set for entry, unless at point
If MP(2).EntryMode = espMillPocketEntryModeHelicalAtPoint Then
MP(2).EntryMode = espMillPocketEntryModeHelixProtected
End If
If MP(2).EntryMode = espMillPocketEntryModePlungeAtPoint Then
MP(2).EntryMode = espMillPocketEntryModePlunge
End If
MP(2).TotalDepth = CutoutProfiles(I).Depth
On Error Resume Next
Call DefaultAttributes(Document.PartOperations.Add(MP(2), CutoutProfiles(I)))
Next
' cleanup
Call Document.SelectionSets.Remove(SS.Name)
Document.Refresh
End Sub
Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)

Dim I As Long
Select Case GraphicObjectVar.GraphicObjectType
Case espAnnotation
GraphicObjectVar.Color = Application.Configuration.ConfigurationNotes.Color

GraphicObjectVar.LineType = .LineType

End With
Case espCircle

GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espCurve

GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight

' note that Dimension, Hatch, and Leader all share a configuration
Case espDimension, espHatch, espLeader
With Application.Configuration.ConfigurationDimensions
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espFeatureChain
With Application.Configuration.ConfigurationFeatureChains
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espFeaturePtop
With Application.Configuration.ConfigurationFeaturePtops
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espLine
With Application.Configuration.ConfigurationLines
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espOperation
With Application.Configuration.ConfigurationOperations
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight
End With
Case espPoint
With Application.Configuration.ConfigurationPoints
GraphicObjectVar.Color = .Color
GraphicObjectVar.LineWeight = .LineWeight
End With

GraphicObjectVar.LineType = .LineType
GraphicObjectVar.LineWeight = .LineWeight

Before running the code open a file that contains a solid, or create one in the Solid Modeler. Hopefully the solid has both holes and pocket cavities in it.

Examine the features and operations created. Is only the FeaturePtop connecting all of the holes drilled? Do the pocket operations on the holes start in the center? Note how the LeadInPoint property of the FeatureChain was used for the entry point.

This tutorial, as well as the other tutorials on feature recognition, is only meant to demonstrate the power and capabilities of ESPRIT EDGE. They may not be practical examples. For instance, there is no contingency in this code for the pockets having islands. This particular code is only suitable for simple pockets with no islands. For complex pockets, we should use the CreatePocketFeatures method of FeatureRecognition.
