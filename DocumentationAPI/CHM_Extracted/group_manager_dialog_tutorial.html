Group Manager Dialog

In the Tutorial: Naming And Recalling Groups we were able to use the SelectionSet object to name and recall groups. Now we are going to improve the interface and build a complete dialog around those code to manage our groups.

The interface will consist of a ListBox control to list the named groups and several CommandButton controls to perform various tasks:

At the very least, we need one CommandButton for when we want to name a new group, and we need another CommandButton for when we want to recall or activate an existing group. For activating a group, the user will select a name in the list and click the Activate button. The user is allowed to select multiple group names from the list to be activated. The user also can activate a group by double clicking it in the ListBox. There is one Activate procedure that will be called both by clicking the Activate CommandButton and by double clicking the ListBox.

Another options available for the user are replace, rename and delete a group from the list. Delete will only remove the named selection set from the list and from the SelectionSets collection. It will not remove the GraphicObject elements that were contained in that SelectionSet from the graphics.

Remember that the (Name) property is the most critical property of each control, as well as of the form itself, because it is by this name that we refer to the control.

The properties of each control should have the following values:

private void OnActiveButtonClick(object sender, EventArgs e)
{
Document.Group.Clear();
var selectedItems = _listBox.SelectedItems;
for (var i = selectedItems.Count - 1; i >= 0; i--)
{
var set = Document.SelectionSets[selectedItems[i]];
for (var j = 1; j <= set.Count; j++)
{
var obj = set[j] as Esprit.GraphicObject;
obj.Grouped = true;
}
}
Document.Refresh();
}
Private Sub OnActiveButtonClick(sender As Object, e As System.EventArgs)
Document.Group.Clear()
' go through all of the names in the list to see if they are selected
Dim selectedItems = _listBox.SelectedItems
For i As Integer = 0 To selectedItems.Count - 1
For j As Integer = 1 To Document.SelectionSets.Item(selectedItems(i)).Count
' group the items in that selection set
Document.SelectionSets.Item(_listBox.SelectedItems(i)).Item(j).Grouped = True
Next
Next
Document.Refresh()
End Sub
Private Sub Activate()
Document.Group.Clear
' go through all of the names in the list to see if they are selected
Dim I As Long
For I = 0 To lstGroups.ListCount - 1
If lstGroups.Selected(I) Then
' use the selected name(s) from the list to index the selection set
Dim J As Long
For J = 1 To Document.SelectionSets.Item(lstGroups.List(I)).Count
' group the items in that selection set
Document.SelectionSets.Item(lstGroups.List(I)).Item(J).Grouped = True
Next
End If
Next
Document.Refresh
End Sub

private void OnNewButtonClick(object sender, EventArgs e)
{
var groupName = $"Group {Document.SelectionSets.Count + 1}";
Esprit.SelectionSet set = null;
do
{
if (!RequestUserInput("Enter Group Name", "Group Name", ref groupName) || groupName.Length == 0)
{
return;
}
else if (GroupNameExists(groupName))
{
var result = System.Windows.MessageBox.Show($"{groupName} already exists. Do you wish to replace it?", "GroupManagerDialogTutorial", MessageBoxButton.YesNoCancel);
switch (result)
{
case MessageBoxResult.Yes:
set = Document.SelectionSets[groupName];
set.RemoveAll();
break;
case MessageBoxResult.No:
break;
case MessageBoxResult.Cancel:
return;
}
}
else
{
set = Document.SelectionSets.Add(groupName);
set.OnAbort += Set_OnAbort;
set.OnAdd += Set_OnAdd;
set.OnRemove += Set_OnRemove;
_selectionSets.Add(set);
_listBox.Items.Add(groupName);
}
} while (set == null);
for (var i = 1; i <= Document.Group.Count; i++)
{
set.Add(Document.Group[i]);
}
}

private bool GroupNameExists(string groupName)
{
var result = false;
foreach (Esprit.SelectionSet set in Document.SelectionSets)
{
if (set.Name == groupName)
{
result = true;
break;
}
}
return result;
}
Private Sub OnNewButtonClick(sender As Object, e As System.EventArgs)
Dim selectionSet As Esprit.SelectionSet = Nothing
Dim groupName As String = "Group" & Document.SelectionSets.Count + 1
Do
If Not RequestUserInput("Enter Group Name", "Group Name", groupName) OrElse groupName.Length = 0 Then
Return
End If
If GroupNameExists(groupName) Then
Select Case MsgBox(groupName & " already exists. Do you wish to replace it?", vbYesNoCancel, "Replace Group")
Case vbYes
' remove everything from the existing selection set
selectionSet = Document.SelectionSets.Item(groupName)
Call selectionSet.RemoveAll()
Exit Do
Case vbNo
' by placing no code here, user will be re-prompted for name
Case vbCancel
Return
End Select
Else
' create the new selection set and add it to the list
selectionSet = Document.SelectionSets.Add(groupName)
Call _listBox.Items.Add(groupName)
AddHandler selectionSet.OnAbort, AddressOf Me.SelectionSets_OnAbort
AddHandler selectionSet.OnAdd, AddressOf Me.SelectionSets_OnAdd
AddHandler selectionSet.OnRemove, AddressOf Me.SelectionSets_OnRemove
Call _selectionSets.Add(selectionSet)
Exit Do
End If
Loop
' loop through the Group and add grouped items to the SelectionSet
For i As Integer = 1 To Document.Group.Count
selectionSet.Add(Document.Group.Item(i))
Next
End Sub

Private Sub cmdNew_Click()
Dim SS As Esprit.SelectionSet

Do

private void OnDeleteButtonClick(object sender, EventArgs e)
{
var selectedItems = _listBox.SelectedItems;

Private Sub OnDeleteButtonClick(sender As Object, e As System.EventArgs)
Dim selectedItems = _listBox.SelectedItems
' loop in reverse order to avoid re-indexing problems
For i As Integer = selectedItems.Count - 1 To 0 Step -1
Call Document.SelectionSets.Remove(selectedItems(i))
Call _listBox.Items.Remove(selectedItems(i))
Next
End Sub
Private Sub cmdDelete_Click()
Dim I As Long
' loop in reverse order to avoid re-indexing problems
For I = lstGroups.ListCount - 1 To 0 Step -1
If lstGroups.Selected(I) = True Then
' remove the selection set and remove it from the list also
Call Document.SelectionSets.Remove(lstGroups.List(I))
Call lstGroups.RemoveItem(I)
End If
Next
End Sub

private void OnRenameButtonClick(object sender, EventArgs e)
{
if (_listBox.SelectedItems.Count != 1)
{
System.Windows.MessageBox.Show("Please select one and only one Group for Rename.", "GroupManagerDialogTutorial");
}
else
{
var oldName = _listBox.SelectedItem as string;
var newName = oldName;
do
{
if (!RequestUserInput("Enter Group Name", "Group Name", ref newName))
{
return;
}
if (newName == string.Empty)
{
System.Windows.MessageBox.Show("Name cannot be empty. Please enter group name.", "GroupManagerDialogTutorial", MessageBoxButton.OKCancel);
}
else if (!GroupNameExists(newName))
{
break;
}
else if (System.Windows.MessageBox.Show($"{newName} already exists. Please choose another name.", "GroupManagerDialogTutorial", MessageBoxButton.OKCancel) == MessageBoxResult.Cancel)
{
return;
}
} while (true);
if (newName.Length > 0)
{
Document.SelectionSets[_listBox.SelectedItem].Name = newName;
_listBox.Items[_listBox.SelectedIndex] = newName;
}
}
}
Private Sub OnRenameButtonClick(sender As Object, e As System.EventArgs)
If _listBox.SelectedItems.Count <> 1 Then
Call MsgBox("Please select one and only one Group for Rename.", vbOKOnly, "Invalid Selection")
Else
Dim oldName As String = _listBox.SelectedItem
Dim newName As String = oldName
Do
If Not RequestUserInput("Enter Group Name", "Group Name", newName) Then
Return
End If
If newName = String.Empty Then
Call MsgBox("Name cannot be empty. Please enter group name.", "GroupManagerDialogTutorial", vbOKCancel)
End If
If GroupNameExists(newName) <> True Then
Exit Do
Else
Call MsgBox(newName & " already exists. Please choose another name.", vbOKOnly, "Error")
End If
Loop
' make sure the NewName is not null (user did not click Cancel)
If Len(newName) > 0 Then
Document.SelectionSets.Item(_listBox.SelectedItem).Name = newName
_listBox.Items(_listBox.SelectedIndex) = newName
End If
End If
End Sub
Private Sub cmdRename_Click()
If SelectedGroupCount <> 1 Then
Call MsgBox("Please select one and only one Group for Rename.", vbOKOnly, "Invalid Selection")
Else
Dim OldName As String, NewName As String
OldName = lstGroups.List(lstGroups.ListIndex)
Do
NewName = InputBox("Enter Group Name", "Group Name", OldName)
If GroupNameExists(NewName) <> True Then
Exit Do
Else
Call MsgBox(NewName & " already exists. Please choose another name.", vbOKOnly, "Error")
End If
Loop
' make sure the NewName is not null (user did not click Cancel)
If Len(NewName) > 0 Then
Document.SelectionSets.Item(lstGroups.List(lstGroups.ListIndex)).Name = NewName
lstGroups.List(lstGroups.ListIndex) = NewName
End If
End If
End Sub
Private Function SelectedGroupCount() As Long
' return the number of selections in the list
SelectedGroupCount = 0
Dim I As Long
For I = 0 To lstGroups.ListCount - 1
If lstGroups.Selected(I) = True Then
SelectedGroupCount = SelectedGroupCount + 1
End If
Next
End Function
