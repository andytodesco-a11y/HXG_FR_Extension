Examining Feature Sets

The Tutorial: Feature Recognition: Pocket Features used the CreatePocketFeatures method of FeatureRecognition. The resulting pocket features are FeatureSet objects. A FeatureSet is a feature object that is simply comprised of other features. The features that comprise a FeatureSet may be FeatureChain objects, or they can be entire FeatureSet objects themselves.

In this tutorial we will write code to loop through the contents of a FeatureSet and display information about its component members. Since there may be other times when we want to display FeatureSet information, let us make a separate procedure for individual feature sets, and another procedure that loops through the FeatureSets collection and calls the first procedure for each one.

public void ScanAllFeatureSets()
{
var output = _application.EventWindow;
output.Clear();
output.Visible = true;
foreach (Esprit.FeatureSet featureSet in Document.FeatureSets)
{
ScanFeatureSet(featureSet, output);
}
}
static private void ScanFeatureSet(Esprit.FeatureSet featureSet, Esprit.EventWindow output)
{
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) contains {featureSet.Count} items");
for (var i = 1; i <= featureSet.Count; i++)
{
switch (featureSet.Item[i].GraphicObjectType)
{
case EspritConstants.espGraphicObjectType.espFeatureChain:
var featureChainObject = featureSet.Item[i] as Esprit.FeatureChain;
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) item {i} is FeatureChain {featureChainObject.Key} ({featureChainObject.Name}) type = {TypeDescription(featureChainObject.Type)}");
break;
case EspritConstants.espGraphicObjectType.espFeatureSet:
var featureSetObject = featureSet.Item[i] as Esprit.FeatureSet;
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) item {i} is FeatureChain {featureSetObject.Key} ({featureSetObject.Name})");
break;
}
}
}
static private string TypeDescription(EspritConstants.espFeatureType type)
{
var result = string.Empty;
switch (type)
{
case EspritConstants.espFeatureType.espFeatureTypeBasic:
result = "Basic";
break;
case EspritConstants.espFeatureType.espFeatureTypeBoundary:
result = "Boundary";
break;
case EspritConstants.espFeatureType.espFeatureTypeHole:
result = "Hole";
break;
case EspritConstants.espFeatureType.espFeatureTypeIsland:
result = "Island";
break;
case EspritConstants.espFeatureType.espFeatureTypePocket:
result = "Pocket";
break;
case EspritConstants.espFeatureType.espFeatureTypeProfile:
result = "Profile";
break;
}
return result;
}

Private Sub ScanFeatureSet(featureSet As Esprit.FeatureSet, output As Esprit.EventWindow)
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) contains {featureSet.Count} items")
For i As Integer = 1 To featureSet.Count
Select Case featureSet.Item(i).GraphicObjectType
Case EspritConstants.espGraphicObjectType.espFeatureChain
Dim featureChainObject As Esprit.FeatureChain = featureSet.Item(i)
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) item {i} is FeatureChain {featureChainObject.Key} ({featureChainObject.Name}) type = {TypeDescription(featureChainObject.Type)}")
Case EspritConstants.espGraphicObjectType.espFeatureSet
Dim featureSetObject As Esprit.FeatureSet = featureSet.Item(i)
output.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", $"FeatureSet {featureSet.Key} ({featureSet.Name}) item {i} is FeatureChain {featureSetObject.Key} ({featureSetObject.Name})")
End Select
Next
End Sub

Sub ScanAllFeatureSets()
' prepare to display results
Application.EventWindow.Clear
Application.EventWindow.Visible = True
' loop through all of the feature sets in the document
Dim FS As Esprit.FeatureSet
For Each FS In Document.FeatureSets
Call ScanFeatureSet(FS)
Next
End Sub
Sub ScanFeatureSet(FeatureSetObj As Esprit.FeatureSet)
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", "FeatureSet " & FeatureSetObj.Key & " (" & FeatureSetObj.Name & ")")
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", " contains " & FeatureSetObj.Count & " items")
End With
Dim I As Long
For I = 1 To FeatureSetObj.Count
Select Case FeatureSetObj.Item(I).GraphicObjectType
Case espFeatureChain
Dim FC As Esprit.FeatureChain
Set FC = FeatureSetObj.Item(I)
' display feature chain information
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", "FeatureSet " & FeatureSetObj.Key & " (" & FeatureSetObj.Name & ")")
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", " item " & I & " is FeatureChain " & FC.Key & " (" & FC.Name & ")")
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", ", type = " & TypeDescription(FC.Type))
End With
Case espFeatureSet
Dim FS As Esprit.FeatureSet
Set FS = FeatureSetObj.Item(I)
' display feature set information
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", "FeatureSet " & FeatureSetObj.Key & " (" & FeatureSetObj.Name & ")")
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ExaminingFeatureSetsTutorial", " item " & I & " is FeatureSet " & FS.Key & " (" & FS.Name & ")")
End With
End Select
Next
End Sub
