Break Leader

In the Tutorial: Create Center Lines as Leaders we saw how to create Leader objects. The leaders we created only contained two LeaderPoints. A Leader may contain any number of LeaderPoints. What if you had previously created a leader, but now as you add to your drawing, you realize that leader is in the way of where you want to place other information. You do not want to remove the leader and recreate it; you just want to insert a point or two in the middle to move it out of the way. In this way you will break the leader or dogleg it around the other information that you wish to place.

Just like other objects that contain objects, the LeaderPoints has an Add method and both Count and Item properties.

public void BreakLeader()
{
Esprit.Leader leader = null;
try
{
leader = Document.GetAnyElement("Select Leader to Break", EspritConstants.espGraphicObjectType.espLeader) as Esprit.Leader;
}
catch (System.Runtime.InteropServices.COMException)
{
// Element was not set
MessageBox.Show("No leader was selected", "BreakLeaderTutorial");
return;
}
if (leader == null)
{
return;
}
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", $"Selected Leader contains {leader.LeaderPoints.Count} points.");
var leaderPoints = new List<Esprit.Point>();
for (var i = 1; i <= leader.LeaderPoints.Count; i++)
{
leaderPoints.Add(leader.LeaderPoints.Item[i]);
}
var index = 2;
var mask = new EspritConstants.espGraphicObjectType[2]
{
EspritConstants.espGraphicObjectType.espLeader,
EspritConstants.espGraphicObjectType.espPoint
};
do
{
Esprit.IGraphicObject graphicObject = null;
try
{
graphicObject = Document.GetAnyElement($"Select New Leader Vertex {index} Reselect Leader To Move to Next Vertex", mask);
}
catch (System.Runtime.InteropServices.COMException)
{
return;
}
if(graphicObject == null)
{
return;
}
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espPoint)
{
leaderPoints.Add(leaderPoints.Last());
for (var j = leader.LeaderPoints.Count; j >= index; j--)
{
leaderPoints[j] = leaderPoints[j - 1];
leader.LeaderPoints.Remove(j);
}
leaderPoints[index - 1] = graphicObject as Esprit.Point;
for (var j = index; j <= leaderPoints.Count; j++)
{
leader.LeaderPoints.Add(leaderPoints[j - 1]);
}
Document.Refresh();
if (MessageBox.Show("Continue to Break?", "Continue", MessageBoxButton.YesNo) == MessageBoxResult.No)
{
break;
}
}
index++;
} while (index <= leader.LeaderPoints.Count);
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", $"After operation the Leader contains {leader.LeaderPoints.Count} points.");
}
Public Sub BreakLeader()
Dim leader As Esprit.Leader = Nothing
Try
leader = Document.GetAnyElement("Select Leader to Break", EspritConstants.espGraphicObjectType.espLeader)
Catch ex As System.Runtime.InteropServices.COMException
' Element was Not set
MessageBox.Show("No leader was selected", "BreakLeaderTutorial")
Return
End Try
If leader Is Nothing Then
Return
End If
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", $"Selected Leader contains {leader.LeaderPoints.Count} points.")
Dim leaderPoints As List(Of Esprit.Point) = New List(Of Esprit.Point)
For i As Integer = 1 To leader.LeaderPoints.Count
leaderPoints.Add(leader.LeaderPoints.Item(i))
Next
Dim index As Integer = 2
Dim mask() As EspritConstants.espGraphicObjectType = {
EspritConstants.espGraphicObjectType.espLeader,
EspritConstants.espGraphicObjectType.espPoint
}
Do
Dim graphicObject As Esprit.IGraphicObject = Nothing
Try
graphicObject = Document.GetAnyElement($"Select New Leader Vertex {index} Reselect Leader To Move to Next Vertex", mask)
Catch ex As System.Runtime.InteropServices.COMException
Return
End Try
If graphicObject Is Nothing Then
Return
End If
If graphicObject.GraphicObjectType = EspritConstants.espGraphicObjectType.espPoint Then
leaderPoints.Add(leaderPoints.Last())
For j As Integer = leader.LeaderPoints.Count To index Step -1
leaderPoints(j) = leaderPoints(j - 1)
leader.LeaderPoints.Remove(j)
Next
leaderPoints(index - 1) = graphicObject
For j As Integer = index To leaderPoints.Count
leader.LeaderPoints.Add(leaderPoints(j - 1))
Next
Document.Refresh()
If MessageBox.Show("Continue to Break?", "Continue", MessageBoxButton.YesNo) = MessageBoxResult.No Then
Exit Do
End If
End If
index += 1
Loop While index <= leader.LeaderPoints.Count
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", $"After operation the Leader contains {leader.LeaderPoints.Count} points.")
End Sub
Sub BreakLeader()
' set up a dynamic array to help keep track of the points in the leader
Dim LeaderPoints() As Esprit.Point
Dim I As Long, J As Long, LPtsCount As Long
' pick the original leader
Dim Leader As Esprit.Leader
On Error Resume Next
Set Leader = Document.GetAnyElement("Select Leader to Break", espLeader)
If Leader Is Nothing Then
Exit Sub
End If
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", "Selected Leader contains " & Leader.LeaderPoints.Count & " points.")
' copy the original leader points into the array
ReDim LeaderPoints(1 To Leader.LeaderPoints.Count) As Esprit.Point
For I = 1 To Leader.LeaderPoints.Count
Set LeaderPoints(I) = Leader.LeaderPoints.Item(I)
Next
' loop through the points starting at the second
Index = 2
Dim TypeMask(1 To 2) As EspritConstants.espGraphicObjectType
TypeMask(1) = espLeader
TypeMask(2) = espPoint
Do
Dim GraphicObject As Esprit.GraphicObject
On Error Resume Next
' allow the user to insert a new point or continue to the next segment
Set GraphicObject = Document.GetAnyElement("Select New Leader Vertex " & Index & ", Reselect Leader To Move to Next Vertex", TypeMask)
If GraphicObject Is Nothing Then
Exit Sub
End If
If GraphicObject.GraphicObjectType = espPoint Then
' insert new point ahead of the current point
LPtsCount = Leader.LeaderPoints.Count + 1
ReDim Preserve LeaderPoints(1 To LPtsCount) As Esprit.Point
' copy the array elements after the current point to the next position in the array
For J = Leader.LeaderPoints.Count To Index Step -1
Set LeaderPoints(J + 1) = LeaderPoints(J)
Call Leader.LeaderPoints.Remove(J)
Next
' set the current array position to the newly chosen point
Set LeaderPoints(Index) = GraphicObject
' re-add the remaining elements of the array into the leader
For J = Index To LPtsCount
Call Leader.LeaderPoints.Add(LeaderPoints(J))
Next
Document.Refresh
If MsgBox("Continue to Break?", vbYesNo, "Continue") = vbNo Then
Exit Do
End If
End If
' go on to the next position, stop when we reach the end
Index = Index + 1
Loop Until Index = Leader.LeaderPoints.Count + 1
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "BreakLeader", "After operation the Leader contains " & Leader.LeaderPoints.Count & " points.")
End Sub

Make sure you watch the prompt and follow its instructions as you run the code. After you select a leader, you can select a new point to insert, or select on the leader again to move to the next vertex.
