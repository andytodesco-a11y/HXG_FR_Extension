Custom Cycles

This tutorial shows how to add an operations using a TechMillCustom technology object. The user is requested to fill in the information that is then passed into the properties of a TechMillCustom object.

Esprit.PartOperations _partOperations;
string _caption;
private void CreateCustomCycleForBoltHole()
{
_caption = "CustomCyclesTutorial";
double patternDiameter = 10;
int numberOfHoles = 6;
double startAngle = 10;
_partOperations = Document.PartOperations;
_partOperations.OnToolPathChanged += PartOperations_OnToolPathChanged;
_partOperations.Rebuild += PartOperations_Rebuild;
_partOperations.AfterRebuild += PartOperations_AfterRebuild;
_partOperations.BeforeRebuild += PartOperations_BeforeRebuild;
if (!RequestUserInput("Enter Pattern Diameter", "Pattern Diameter", ref patternDiameter) ||
!RequestUserInput("Enter Number of Holes", "Holes", ref numberOfHoles) ||
!RequestUserInput("Enter Start Angle", "Start Angle", ref startAngle))
{
return;
}
Esprit.Point centerPoint = null;
try
{
centerPoint = Document.GetPoint("Pick Pattern Center");
}
catch (System.Runtime.InteropServices.COMException)
{
// Point was not set
return;
}
var millCustom = new EspritTechnology.TechMillCustom
{
ChannelID = 1,
OperationName = "Mill Custom Technology",
CustomSetting1 = 9020,
CustomSetting2 = centerPoint.X,
CustomSetting3 = centerPoint.Y,
CustomSetting4 = patternDiameter,
CustomSetting5 = numberOfHoles,
CustomSetting6 = startAngle,
Comment = "BOLT HOLE CANNED CYCLE"
};
try
{
Document.PartOperations.Add(millCustom);
}
catch (System.Runtime.InteropServices.COMException e)
{
MessageBox.Show(e.Message);
}
catch (ArgumentException)
{
MessageBox.Show("Invalid technology");
}
}

Set CenterPoint = Document.GetPoint("Pick Pattern Center")

Exit Sub
End If
' transfer the settings to a custom technology and add the operation
Dim MC As EspritTechnology.TechMillCustom
Set MC = New EspritTechnology.TechMillCustom
MC.ChannelID = 1
MC.OperationName = "Mill Custom Technology"
