Custom Setup Sheet for Tooling

The tutorial Find Tool Parameters showed how we could find the property names of all of the Parameter objects that related to tools. Knowing those names, we could also get the values of those parameters. A use for this might be in making a custom setup sheet for tooling. We could present the user with a list of properties, they could pick the names for which they wanted values, and we could then output a setup sheet listing those values for each tool.

For this we would want to present the user with a dialog from which he could pick the properties. One list on the dialog will show all of the properties he could choose from, the other will list the ones that had been chosen. A pull down could let him pick whether just the lathe, just the mill, or both lathe and mill tool properties were displayed. We could then write the setup sheet out as a .csv file and then the user could later customize the fonts and other display formatting as part of an Excel macro

private void RemoveButtonClick(object sender, RoutedEventArgs e)
{
for (var i = 0; i < _dialog.OutputProperties.Items.Count; i++)
{
if (_dialog.OutputProperties.SelectedIndex == i)
{
_outputToolPropertyNames.Remove(_dialog.OutputProperties.Items.GetItemAt(i).ToString());
_dialog.OutputProperties.Items.RemoveAt(i);

}
}
private void AddButtonClick(object sender, RoutedEventArgs e)
{
for (var i = 0; i < _dialog.AvailableProperties.Items.Count; i++)
{
if (_dialog.AvailableProperties.SelectedIndex == i)
{
var value = _dialog.AvailableProperties.Items.GetItemAt(i).ToString();
if (!_outputToolPropertyNames.Contains(value))
{
_outputToolPropertyNames.Add(value);
_dialog.OutputProperties.Items.Add(value);
}
break;
}
}
}
private void SetAvailablePropertiesList(string type)
{
_dialog.AvailableProperties.Items.Clear();
switch (type)
{
case "All Tools":
foreach(var item in _allToolPropertyNames)
{
_dialog.AvailableProperties.Items.Add(item);
}
break;
case "Lathe Tools":
foreach (var item in _toolLathePropertyNames)
{
_dialog.AvailableProperties.Items.Add(item);
}
break;
case "Mill Tools":
foreach (var item in _toolMillPropertyNames)
{
_dialog.AvailableProperties.Items.Add(item);
}
break;
}
}
private void FindToolPropertyNames()
{
_toolLathePropertyNames.Clear();
_toolMillPropertyNames.Clear();
_allToolPropertyNames.Clear();
_outputToolPropertyNames.Clear();
var technologyUtility = Document.TechnologyUtility as EspritTechnology.TechnologyUtility;
foreach (espTechnologyType technologyType in Enum.GetValues(typeof(espTechnologyType)))
{
try
{
var technology = technologyUtility.CreateTechnology(technologyType, Document.SystemUnit);
if (technology != null)
{
var techName = Microsoft.VisualBasic.Information.TypeName(technology);
techName = techName.Substring(1);
if (techName.StartsWith("ToolLathe"))
{
for (var j = 1; j <= technology.Count; j++)
{
if (!_toolLathePropertyNames.Contains(technology[j].Name))
{
_toolLathePropertyNames.Add(technology[j].Name);
}
if (!_allToolPropertyNames.Contains(technology[j].Name))
{
_allToolPropertyNames.Add(technology[j].Name);
}
}
}
else if (techName.StartsWith("ToolMill"))
{
for (var j = 1; j <= technology.Count; j++)
{
if (!_toolMillPropertyNames.Contains(technology[j].Name))
{
_toolMillPropertyNames.Add(technology[j].Name);
}
if (!_allToolPropertyNames.Contains(technology[j].Name))
{
_allToolPropertyNames.Add(technology[j].Name);
}
}
}
}
}
catch (Exception)
{ }
}
}
private void OnApplyButtonClick(object sender, RoutedEventArgs e)
{
_dialog.Close();
var header = string.Empty;
for (var i = 0; i < _outputToolPropertyNames.Count; i++)
{
if (header.Length != 0)
{
header += ", ";
}
header += _outputToolPropertyNames[i];
}
var toolData = string.Empty;
var tools = Document.Tools as EspritTools.Tools;
foreach(EspritTechnology.Technology tech in tools)
{
for (var i = 0; i < _outputToolPropertyNames.Count; i++)
{
if (toolData.Length != 0)
{
toolData += ", ";
}
var param = tech.FindParameter(_outputToolPropertyNames[i]);
if (param != null)
{
toolData += param.Value.ToString();
}
}
}
var filePath = Path.Combine(_application.TempDir, Guid.NewGuid().ToString() + ".csv");
using (var outStream = new StreamWriter(filePath))
{
outStream.WriteLine(header);
outStream.WriteLine(toolData);
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ToolingCustomSetupSheetTutorial", $"File created: {filePath}");
}
}

Sub FindToolPropertyNames()
Dim TechUtil As EspritTechnology.TechnologyUtility
Set TechUtil = Document.TechnologyUtility
'
' loop through all possible technologies
' Can check EspritConstants.espTechnologyType for limits if desired
' or, just to be sure, try a large range like 0 to 999
'
Dim TType As Long
For TType = 0 To 999
Dim T As EspritTechnology.Technology
Set T = Nothing
On Error Resume Next
Set T = TechUtil.CreateTechnology(TType, Document.SystemUnit)
On Error GoTo 0
'
' make sure new technology was created
'
If Not (T Is Nothing) Then
Dim TechName As String
TechName = TypeName(T)
TechName = Mid$(TechName, 2) ' cut off the "I" at the start
'
' check to see if it is a tool type
'
If InStr(1, TechName, "Tool") = 1 Then
Dim P As EspritTechnology.Parameter
Dim PNum As Long
For PNum = 1 To T.Count
Set P = T.Item(PNum)
'
' add to the appropriate collection if it is not already a member
'
Dim M As String
M = ""
On Error Resume Next
M = AllToolPropertyNames.Item(P.Name)
On Error GoTo 0
If M = "" Then Call AllToolPropertyNames.Add(P.Name, P.Name)
If InStr(1, TechName, "ToolLathe") = 1 Then
M = ""
On Error Resume Next
M = ToolLathePropertyNames.Item(P.Name)
On Error GoTo 0
If M = "" Then Call ToolLathePropertyNames.Add(P.Name, P.Name)
End If
If InStr(1, TechName, "ToolMill") = 1 Then
M = ""
On Error Resume Next
M = ToolMillPropertyNames.Item(P.Name)
On Error GoTo 0
If M = "" Then Call ToolMillPropertyNames.Add(P.Name, P.Name)
End If
Next
End If
End If
Next
End Sub
Private Sub CycleType_Change()
Select Case CycleType.ListIndex
Case 0 ' All Tools
Call ListCollection(AllToolPropertyNames)
Case 1 ' Lathe Tools
Call ListCollection(ToolLathePropertyNames)
Case 2 ' Mill Tools
Call ListCollection(ToolMillPropertyNames)
End Select
End Sub
Private Sub ListCollection(PropertyCollection As Collection)
AvailableProperties.Clear
Dim I As Long
For I = 1 To PropertyCollection.Count
Call AvailableProperties.AddItem(PropertyCollection.Item(I))
Next
End Sub
Sub AddProperty()
Dim I As Long
For I = 0 To AvailableProperties.ListCount - 1
If AvailableProperties.Selected(I) Then
Dim M(1 To 2) As String
M(1) = AvailableProperties.List(I)
M(2) = ""
On Error Resume Next
M(2) = OutputPropertyNames.Item(AvailableProperties.List(I))
On Error GoTo 0
If M(2) = "" Then
Call OutputPropertyNames.Add(M(1), M(1))
Call OutputProperties.AddItem(M(1))
End If
End If
Next
End Sub
Sub RemoveProperty()
Dim I As Long
For I = 0 To OutputProperties.ListCount - 1
If OutputProperties.Selected(I) Then
Call OutputPropertyNames.Remove(OutputProperties.List(I))
Call OutputProperties.RemoveItem(I)
End If
Next
End Sub
Private Sub AddButton_Click()
Call AddProperty
End Sub
Private Sub OkButton_Click()
Me.Hide
Dim FileName As String
FileName = Application.TempDir + "Tooling Setup Sheet.csv"
Open FileName For Output As #1
'
' create the header
'
Dim I As Long, Header As String
For I = 0 To OutputProperties.ListCount - 1
If I > 0 Then Header = Header & ","
Header = Header & OutputProperties.List(I)
Next
'
' since we are formatting the data with commas ourself, use Print # instead of Write #
'
Print #1, Header
'
' Create a line of data for each tool in the document
'
Dim T As EspritTechnology.Technology, ToolData As String
For Each T In Document.Tools
ToolData = ""
For I = 0 To OutputProperties.ListCount - 1
If I > 0 Then ToolData = ToolData & ","
On Error Resume Next ' in case this technology does not have that property
ToolData = ToolData & Chr(34) & T.Item(OutputProperties.List(I)).value & Chr(34)
On Error GoTo 0
Next
Print #1, ToolData
Next
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ToolingCustomSetupSheetTutorial", "File created: " & FileName)
Close 1
Unload Me
End Sub
