Introduction to Feature Recognition

This tutorial introduces the ESPRIT EDGE feature recognition capability to produce multiple FeatureChain and FeaturePtop objects from solid and wireframe geometry. This can be done through the API using the FeatureRecognition object.

The following code uses the CreateAutoChains method of feature recognition. When creating auto chains through the ESPRIT EDGE interface, the user defines a group and then clicks the Auto Chain icon. Suppose that you import complex DXF or IGES files, however, and they have a lot of drill holes (circles) that you do not want to put feature chains on, but you want to put features on all of the other geometry. The code shown below will group all of the arcs and segments and use the FeatureRecognition.CreateAutoChains method to create feature chains on them.

public void ChainAll()
{
var set = SelectionSetHelper.GetOrAddSelectionSet(Document.SelectionSets, "Temp");
set.RemoveAll();
foreach (Esprit.IGraphicObject graphicObject in Document.GraphicsCollection)
{
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espArc || graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espSegment)
{
set.Add(graphicObject);
}
}
if (set.Count > 0)
{
try
{
Document.FeatureRecognition.CreateAutoChains(set);
}
catch(System.Runtime.InteropServices.COMException)
{ }
}
Document.SelectionSets.Remove(set.Name);
}
public static SelectionSet GetOrAddSelectionSet(ISelectionSets selectionSets, string name)
{
var set = selectionSets[name] ?? selectionSets.Add(name);
return set;
}

Public Function GetOrAddSelectionSet(document As Esprit.Document, name As String) As Esprit.SelectionSet
Dim selSet As Esprit.SelectionSet = document.SelectionSets.Item(name)
If selSet Is Nothing Then
selSet = document.SelectionSets.Add(name)
End If
Return selSet
End Function
Sub ChainAll()
Dim SS As Esprit.SelectionSet
Set SS = GetSelectionSet("Temp")
SS.RemoveAll
' loop through the graphics and include all bound segments and arcs in the selection set
Dim G As Esprit.GraphicObject
For Each G In Document.GraphicsCollection
If (G.GraphicObjectType = espArc) Or (G.GraphicObjectType = espSegment) Then
Call SS.Add(G)
End If
Next
' create the feature chains
If SS.Count > 0 Then
Call Document.FeatureRecognition.CreateAutoChains(SS)
End If
' cleanup
Call Document.SelectionSets.Remove(SS.Name)
End Sub
Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)
On Error GoTo 0
If GetSelectionSet Is Nothing Then
Set GetSelectionSet = Document.SelectionSets.Add(SSName)
End If
End Function
