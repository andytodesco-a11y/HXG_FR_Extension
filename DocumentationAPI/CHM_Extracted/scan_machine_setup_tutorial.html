This tutorial scans Machine setup tree and displays common information about machine items in Output window. The following code scans the items:

private void ScanMachineItem(Esprit.MachineItem item)
{
var eventWindow = _application.EventWindow;
var stringBuilder = new StringBuilder();
var basicInfo = "Name: " + item.Name + Environment.NewLine;
basicInfo += "Type: " + DisplayMachineItemType(item.MachineItemType);
stringBuilder.AppendLine(basicInfo);
if (item.ChildAdaptors.Count > 0)
{
var childAdaptorsInfo = "Child Adaprtors:";
foreach (Esprit.Adaptor adaptor in item.ChildAdaptors)
{
childAdaptorsInfo += Environment.NewLine + $"\tChild Adaptor {adaptor.Name} Type {DisplayAdaptorType(adaptor.AdaptorType)}";
}
stringBuilder.AppendLine(childAdaptorsInfo);
}
if (item.MountingAdaptors.Count > 0)
{
var mountingAdaptorsInfo = "Mounting Adaptors:";
foreach (Esprit.Adaptor adaptor in item.MountingAdaptors)
{
mountingAdaptorsInfo += Environment.NewLine + $"\tMounting Adaptor {adaptor.Name} Type {DisplayAdaptorType(adaptor.AdaptorType)}";
}
if (item.Mounting != null)
{
mountingAdaptorsInfo += Environment.NewLine + "\tMounted Adaptor: " + item.Mounting.Name;
}
stringBuilder.AppendLine(mountingAdaptorsInfo);
}
stringBuilder.AppendLine(DisplayMatrixData(item.RelativeTransformation, "RelativeTransformation:"));
stringBuilder.AppendLine(DisplayMatrixData(item.CumulativeTransformation, "CumulativeTransformation:"));
switch(item.MachineItemType)
{
case EspritConstants.espMachineItemType.espMachineItemTypeFixture:
{
var fixture = item as Esprit.Fixture;
var fixtureInfo = "Fixture Data:";
fixtureInfo += Environment.NewLine + "\tClearance: " + fixture.SafetyDistance;
fixtureInfo += Environment.NewLine + "\tFileName: " + fixture.FileName;
stringBuilder.AppendLine(fixtureInfo);
}
break;
case EspritConstants.espMachineItemType.espMachineItemTypeWorkpiece:
{
var workpieceItem = item as Esprit.WorkpieceInstance;
var workpieceInfo = "Workpiece Data:";
if (workpieceItem.PartWorkpiece != null)
{
workpieceInfo += Environment.NewLine + "\tWorkpiece Name: " + workpieceItem.PartWorkpiece.Name;
workpieceInfo += Environment.NewLine + "\tParts Count: " + workpieceItem.PartWorkpiece.Parts.Count;
}
if (workpieceItem.WorkOffsets.Count > 0)
{
var offsetsInfo = "\tOffsets:";
foreach (Esprit.WorkOffset offset in workpieceItem.WorkOffsets)
{
offsetsInfo += Environment.NewLine + "\t\t" + offset.Name;
}
workpieceInfo += Environment.NewLine + offsetsInfo;
}
stringBuilder.AppendLine(workpieceInfo);
}
break;
}
eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanMachineSetupTutorial", stringBuilder.ToString());
}
private void ScanMachineItemsReq(Esprit.MachineItem item)
{
ScanMachineItem(item);
foreach(Esprit.MachineItem childItem in item.MachineItems)
{
ScanMachineItemsReq(childItem);
}
}
private void ScanMachineItems()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
if (Document.InitialMachineSetup.MachineItems.Count == 0)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanMachineSetupTutorial", "There is empty machine");
}
else
{
foreach (Esprit.MachineItem item in Document.InitialMachineSetup.MachineItems)
{
ScanMachineItemsReq(item);
}
}
}
Private Sub ScanMachineItem(item As Esprit.MachineItem)
Dim eventWindow = _application.EventWindow
Dim stringBuilder = New StringBuilder()
Dim basicInfo As String = "Name: " + item.Name + Environment.NewLine
basicInfo += "Type: " + DisplayMachineItemType(item.MachineItemType)
stringBuilder.AppendLine(basicInfo)
If item.ChildAdaptors.Count > 0 Then
Dim childAdaptorsInfo As String = "Child Adaprtors:"
For Each adaptor As Esprit.Adaptor In item.ChildAdaptors
childAdaptorsInfo += Environment.NewLine + vbTab + $"Child Adaptor {adaptor.Name} Type {DisplayAdaptorType(adaptor.AdaptorType)}"
Next
stringBuilder.AppendLine(childAdaptorsInfo)
End If
If item.MountingAdaptors.Count > 0 Then
Dim mountingAdaptorsInfo As String = "Mounting Adaptors:"
For Each adaptor As Esprit.Adaptor In item.MountingAdaptors
mountingAdaptorsInfo += Environment.NewLine + vbTab + $"Mounting Adaptor {adaptor.Name} Type {DisplayAdaptorType(adaptor.AdaptorType)}"
Next
If Not item.Mounting Is Nothing Then
mountingAdaptorsInfo += Environment.NewLine + vbTab + "Mounted Adaptor: " + item.Mounting.Name
End If
stringBuilder.AppendLine(mountingAdaptorsInfo)
End If
stringBuilder.AppendLine(DisplayMatrixData(item.RelativeTransformation, "RelativeTransformation:"))
stringBuilder.AppendLine(DisplayMatrixData(item.CumulativeTransformation, "CumulativeTransformation:"))
Select Case item.MachineItemType
Case EspritConstants.espMachineItemType.espMachineItemTypeFixture
Dim fixture As Esprit.Fixture = item
Dim fixtureInfo As String = "Fixture Data:"
fixtureInfo += Environment.NewLine + vbTab + "Clearance: " + fixture.SafetyDistance.ToString()
fixtureInfo += Environment.NewLine + vbTab + "FileName: " + fixture.FileName
stringBuilder.AppendLine(fixtureInfo)
Case EspritConstants.espMachineItemType.espMachineItemTypeWorkpiece
Dim workpieceItem As Esprit.WorkpieceInstance = item
Dim workpieceInfo As String = "Workpiece Data:"
If Not workpieceItem.PartWorkpiece Is Nothing Then
workpieceInfo += Environment.NewLine + vbTab + "Workpiece Name: " + workpieceItem.PartWorkpiece.Name
workpieceInfo += Environment.NewLine + vbTab + "Parts Count: " + workpieceItem.PartWorkpiece.Parts.Count.ToString()
End If
If workpieceItem.WorkOffsets.Count > 0 Then
Dim offsetsInfo As String = vbTab + "Offsets:"
For Each offset As Esprit.WorkOffset In workpieceItem.WorkOffsets
offsetsInfo += Environment.NewLine + vbTab + offset.Name
Next
workpieceInfo += Environment.NewLine + offsetsInfo
End If
stringBuilder.AppendLine(workpieceInfo)
End Select
eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanMachineSetupTutorial", stringBuilder.ToString())
End Sub
Private Sub ScanMachineItemsReq(item As Esprit.MachineItem)
ScanMachineItem(item)
For Each childItem As Esprit.MachineItem In item.MachineItems
ScanMachineItemsReq(childItem)
Next
End Sub
Private Sub ScanMachineItems()
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
If (Document.InitialMachineSetup.MachineItems.Count = 0) Then
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeWarning, "ScanMachineSetupTutorial", "There is empty machine")
Else
For Each item As Esprit.MachineItem In Document.InitialMachineSetup.MachineItems
ScanMachineItemsReq(item)
Next
End If
End Sub
Sub ScanMachineItem(MItem As Esprit.MachineItem)
Dim stringBuilder As String
Dim basicInfo As String
basicInfo = "Name: " & MItem.Name & vbNewLine
basicInfo = basicInfo & "Type: " & DisplayMachineItemType(MItem.MachineItemType)
stringBuilder = stringBuilder & vbNewLine & basicInfo
If MItem.ChildAdaptors.Count > 0 Then
Dim childAdaptorsInfo As String
childAdaptorsInfo = "Child Adaprtors:"
For Each Adaptor In MItem.ChildAdaptors
childAdaptorsInfo = childAdaptorsInfo & vbNewLine & vbTab & "Child Adaptor " & Adaptor.Name & " Type " & DisplayAdaptorType(Adaptor.AdaptorType)
Next
stringBuilder = stringBuilder & vbNewLine & childAdaptorsInfo
End If
If MItem.MountingAdaptors.Count > 0 Then
Dim mountingAdaptorsInfo As String
mountingAdaptorsInfo = "Mounting Adaptors:"
For Each Adaptor In MItem.MountingAdaptors
mountingAdaptorsInfo = mountingAdaptorsInfo & vbNewLine & vbTab & "Mounting Adaptor " & Adaptor.Name & " Type " & DisplayAdaptorType(Adaptor.AdaptorType)
Next
If Not (MItem.Mounting Is Nothing) Then
mountingAdaptorsInfo = mountingAdaptorsInfo & vbNewLine & vbTab & "Mounted Adaptor: " & MItem.Mounting.Name
End If
stringBuilder = stringBuilder & vbNewLine & mountingAdaptorsInfo
End If
stringBuilder = stringBuilder & vbNewLine & DisplayMatrixData(MItem.RelativeTransformation, "RelativeTransformation:")
stringBuilder = stringBuilder & vbNewLine & DisplayMatrixData(MItem.CumulativeTransformation, "CumulativeTransformation:")
Select Case MItem.MachineItemType
Case espMachineItemTypeFixture:
Dim mFixture As Esprit.Fixture
Set mFixture = MItem
Dim fixtureInfo As String
fixtureInfo = "Fixture Data:"
fixtureInfo = fixtureInfo & vbNewLine & vbTab & "Clearance: " & mFixture.SafetyDistance
fixtureInfo = fixtureInfo & vbNewLine & vbTab & "FileName: " & mFixture.FileName
stringBuilder = stringBuilder & vbNewLine & fixtureInfo
Case espMachineItemTypeWorkpiece:
Dim mWorkpieceItem As Esprit.WorkpieceInstance
Set mWorkpieceItem = MItem
Dim mWorkpiece As Esprit.Workpiece
Set mWorkpiece = mWorkpieceItem.PartWorkpiece
Dim workpieceInfo As String
workpieceInfo = "Workpiece Data:"
If Not (mWorkpiece Is Nothing) Then
workpieceInfo = workpieceInfo & vbNewLine & vbTab & "Workpiece Name: " & mWorkpiece.Name
workpieceInfo = workpieceInfo & vbNewLine & vbTab & "Parts Count: " & mWorkpiece.Parts.Count
End If
If mWorkpieceItem.WorkOffsets.Count > 0 Then
Dim offsetsInfo
offsetsInfo = vbTab & "Offsets:"
For Each Offset In mWorkpieceItem.WorkOffsets
offsetsInfo = offsetsInfo & vbNewLine & vbTab & Offset.Name
Next
workpieceInfo = workpieceInfo & Cb & offsetsInfo
End If
stringBuilder = stringBuilder & vbNewLine & workpieceInfo
End Select
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanMachineSetupTutorial", stringBuilder)
End Sub
Sub ScanMachineItemsReq(MItem As Esprit.MachineItem)
Call ScanMachineItem(MItem)
Dim MChildItem As Esprit.MachineItem
For Each MChildItem In MItem.MachineItems
Call ScanMachineItemsReq(MChildItem)
Next
End Sub
Sub ScanMachineItems()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
If Document.InitialMachineSetup.MachineItems.Count = 0 Then
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeWarning, "ScanMachineSetupTutorial", "There is empty machine")
Else
Dim MItem As Esprit.MachineItem
For Each MItem In Document.InitialMachineSetup.MachineItems
Call ScanMachineItemsReq(MItem)
Next
End If
End Sub

private string DisplayMachineItemType(EspritConstants.espMachineItemType value)
{
var result = "MachineItem";
switch (value)
{
case EspritConstants.espMachineItemType.espMachineItemTypeFixture:
result = "Fixture";
break;
case EspritConstants.espMachineItemType.espMachineItemTypeWorkpiece:
result = "Workpiece";
break;
}
return result;
}
private string DisplayAdaptorType(EspritConstants.espAdaptorType value)
{
var result = string.Empty;
switch(value)
{
case EspritConstants.espAdaptorType.espAdaptorTypeFixture:
result = "Fixture adaptor";
break;
case EspritConstants.espAdaptorType.espAdaptorTypeWorkpiece:
result = "Workpiece adaptor";
break;
case EspritConstants.espAdaptorType.espAdaptorTypeJaw:
result = "Jaw adaptor";
break;
}
return result;
}
private string DisplayDoubleValue(double value)
{
return (Math.Abs(value) > 0.000001)
? value.ToString("F5")
: "0";
}
private string DisplayMatrixData(EspritGeometryBase.IComMatrix matrix, string name)
{
string resultString = name;
resultString += Environment.NewLine + "\tP.X: " + DisplayDoubleValue(matrix.P.X);
resultString += Environment.NewLine + "\tP.Y: " + DisplayDoubleValue(matrix.P.Y);
resultString += Environment.NewLine + "\tP.Z: " + DisplayDoubleValue(matrix.P.Z);
resultString += Environment.NewLine + "\tU.X: " + DisplayDoubleValue(matrix.U.X);
resultString += Environment.NewLine + "\tU.Y: " + DisplayDoubleValue(matrix.U.Y);
resultString += Environment.NewLine + "\tU.Z: " + DisplayDoubleValue(matrix.U.Z);
resultString += Environment.NewLine + "\tV.X: " + DisplayDoubleValue(matrix.V.X);
resultString += Environment.NewLine + "\tV.Y: " + DisplayDoubleValue(matrix.V.Y);
resultString += Environment.NewLine + "\tV.Z: " + DisplayDoubleValue(matrix.V.Z);
resultString += Environment.NewLine + "\tW.X: " + DisplayDoubleValue(matrix.W.X);
resultString += Environment.NewLine + "\tW.Y: " + DisplayDoubleValue(matrix.W.Y);
resultString += Environment.NewLine + "\tW.Z: " + DisplayDoubleValue(matrix.W.Z);
return resultString;
}
