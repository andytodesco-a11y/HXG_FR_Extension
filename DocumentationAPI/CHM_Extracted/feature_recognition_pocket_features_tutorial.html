Feature Recognition: Pocket Features

This tutorial introduces the CreatePocketFeatures method of FeatureRecognition. One thing that distinguishes CreatePocketFeatures from the other feature recognition methods is that the SelectionSet that is passed to it does not contain the objects on which to recognize features. Instead the SelectionSet is a boundary, inside which all pockets will be recognized, and the object on which to recognize those pockets is passed as a separate Body argument.

Suppose we want to find all of the pockets on a certain solid with respect to the current active plane. The solid is the Body argument. We need to come up with its outside boundary in order to add that to the SelectionSet argument. Well, we already know how to get the outside boundary of a solid. We did that in Tutorial: Feature Recognition: Part Profiles by using the CreatePartProfileShadow method of feature recognition.

public void FindAllXYZPockets()
{
Esprit.Solid solid = null;
try
{
solid = Document.GetAnyElement("Select Reference Solid", EspritConstants.espGraphicObjectType.espSolidModel) as Esprit.Solid;
}
catch(System.Runtime.InteropServices.COMException)
{
// Element was not set
return;
}
if(solid == null)
{
return;
}
var set = SelectionSetHelper.GetSelectionSet(Document, "Temp");
set.RemoveAll();
set.Add(solid);
var plane = Document.Planes["XYZ"];
if (plane == null)
{
plane = PlaneHelper.GetPlane(Document, "Temp XYZ");
PlaneHelper.SetPlaneVectors(ref plane, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1);
}
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = (Document.SystemUnit == EspritConstants.espUnitType.espInch)
? 0.0001
: 0.0025;
Array boundaries = null;
try
{
boundaries = Document.FeatureRecognition.CreatePartProfileShadow(set, plane, EspritConstants.espGraphicObjectReturnType.espFeatureChains);
}
catch (Exception)
{
MessageBox.Show("Fail on creating part profile shadow.", "FeatureRecognitionPocketFeaturesTutorial", MessageBoxButton.OK);
return;
}
Esprit.FeatureChain boundary = null;
for (var i = boundaries.GetLowerBound(0); i <= boundaries.GetUpperBound(0); i++)
{
var graphicObject = boundaries.GetValue(i) as Esprit.GraphicObject;
if (graphicObject.GraphicObjectType != EspritConstants.espGraphicObjectType.espFeatureChain)
{
continue;
}
var featureChain = graphicObject as Esprit.FeatureChain;
if (boundary == null)
{
boundary = featureChain;
}
else if (featureChain.Area > boundary.Area)
{
boundary = featureChain;
}
}
if (boundary == null)
{
MessageBox.Show("Unable to determine boundary. Aborting.", "FeatureRecognitionPocketFeaturesTutorial", MessageBoxButton.OK);
return;
}
set.RemoveAll();
set.Add(boundary);
try
{
Document.FeatureRecognition.CreatePocketFeatures(set, EspritConstants.espPocketType.espPocketTypeMultiple, solid, new Esprit.Plane[1] { plane });
}
catch (Exception)
{
MessageBox.Show("Fail on creating pocket features.", "FeatureRecognitionPocketFeaturesTutorial", MessageBoxButton.OK);
return;
}
}

public static Esprit.Plane GetPlane(Esprit.Document document, string name)
{
Esprit.Plane plane = null;
foreach (Esprit.Plane p in document.Planes)
{
if (p.Name == name)
{
plane = p;
break;
}
}
if (plane == null)
{
plane = document.Planes.Add(name);
}
return plane;
}

Dim boundary As Esprit.FeatureChain = Nothing
For i As Integer = boundaries.GetLowerBound(0) To boundaries.GetUpperBound(0)
Dim graphicObject As Esprit.GraphicObject = boundaries.GetValue(i)
If graphicObject.GraphicObjectType <> EspritConstants.espGraphicObjectType.espFeatureChain Then
Continue For
End If
Dim featureChain As Esprit.FeatureChain = graphicObject
If boundary Is Nothing Then
boundary = featureChain
ElseIf featureChain.Area > boundary.Area Then
boundary = featureChain
End If
Next
If boundary Is Nothing Then

Return
End If
selectionSet.RemoveAll()
selectionSet.Add(boundary)
Try
Document.FeatureRecognition.CreatePocketFeatures(selectionSet, EspritConstants.espPocketType.espPocketTypeMultiple, solid, plane)
Catch ex As Exception
System.Windows.MessageBox.Show("Fail on creating pocket features.", "FeatureRecognitionPocketFeaturesTutorial", vbOK)

Sub FindAllXYZPockets()
' prompt the user to pick a solid
Dim SL As Esprit.Solid
On Error Resume Next
Set SL = Document.GetAnyElement("Select Reference Solid", espSolidModel)
If SL Is Nothing Then Exit Sub ' in case user presses escape
On Error GoTo 0 ' to disable further error trapping
' try to use the XYZ plane, note that we must declare it as an array of one to pass later
Dim PL(1 To 1) As Esprit.Plane
On Error Resume Next ' skip in case user deleted
Set PL(1) = Document.Planes.Item("XYZ")
On Error GoTo 0 ' disable further error trapping
If PL(1) Is Nothing Then
' make a plane parallel to XYZ if plane was not set
Set PL(1) = GetPlane("Temp XYZ")
SetPlaneVectors(PL(1), 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1).IsWork = False
End If
' get a selection set for use in feature recognition
Dim SS As Esprit.SelectionSet
Set SS = GetSelectionSet("Temp")
SS.RemoveAll
' first used the selection set for the solid
Call SS.Add(SL)
' find the pocket boundary by using part profile shadow with an appropriate tolerance
If Document.SystemUnit = espInch Then
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.0001
Else
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.0025
End If
Dim Boundaries() As Esprit.GraphicObject, Boundary As Esprit.FeatureChain
Boundaries = Document.FeatureRecognition.CreatePartProfileShadow(SS, PL(1), espFeatureChains)
' only one boundary should be created, but just in case, loop through and find largest
Dim I As Long
For I = LBound(Boundaries) To UBound(Boundaries)
If Boundaries(I).GraphicObjectType = espFeatureChain Then
Dim TempFC As Esprit.FeatureChain
Set TempFC = Boundaries(I)
If Boundary Is Nothing Then
Set Boundary = TempFC
Else
If TempFC.Area > Boundary.Area Then
' we have found a larger boundary
Set Boundary = TempFC
End If
End If
End If
Next
' now use the selection set for the boundary
If Boundary Is Nothing Then
Call MsgBox("Unable to determine boundary. Aborting.", vbOKOnly, "Error")
Exit Sub
End If
SS.RemoveAll
Call SS.Add(Boundary)
' perform pocket feature creation, not that PL is passed as an array (without an index #)
Call Document.FeatureRecognition.CreatePocketFeatures(SS, espPocketTypeMultiple, SL, PL)
End Sub
Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)
On Error GoTo 0
If GetSelectionSet Is Nothing Then
Set GetSelectionSet = Document.SelectionSets.Add(SSName)
End If
End Function
Function GetPlane(PlaneName As String) As Esprit.Plane
On Error Resume Next
Set GetPlane = Document.Planes.Item(PlaneName)
On Error GoTo 0
If GetPlane Is Nothing Then
Set GetPlane = Document.Planes.Add(PlaneName)
End If
End Function
Function SetPlaneVectors(PlaneObj As Esprit.Plane, X As Double, Y As Double, Z As Double, Ux As Double, Uy As Double, Uz As Double, Vx As Double, Vy As Double, Vz As Double, Wx As Double, Wy As Double, Wz As Double) As Esprit.Plane
With PlaneObj
.X = X
.Y = Y
.Z = Z
.Ux = Ux
.Uy = Uy
.Uz = Uz
.Vx = Vx
.Vy = Vy
.Vz = Vz
.Wx = Wx
.Wy = Wy
.Wz = Wz
End With
Set SetPlaneVectors = PlaneObj
End Function
