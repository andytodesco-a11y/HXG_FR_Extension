Stock Automation Events

The Tutorial demostrates the Stock Automation object events.

Define a class and add the event procedures:

C#
VB.NET
VBA
class StockAutomationEventsHandler
{
private readonly Esprit.Application _application;
private readonly Esprit.StockAutomation _stockAutomationObject;
private readonly string _messageBoxTitle;
public StockAutomationEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "StockAutomationEventsTutorial";
_stockAutomationObject = _application.Document.StockAutomation;
_application.EventWindow.Visible = true;
_application.EventWindow.Clear();
_stockAutomationObject.OnCalculationStart += OnCalculationStart;
_stockAutomationObject.OnCalculationEnd += OnCalculationEnd;
_stockAutomationObject.OnCalculationAbort += OnCalculationAbort;
_stockAutomationObject.OnCalculationError += OnCalculationError;
}
private void OnCalculationStart(object operation)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationStart event");
}
private void OnCalculationAbort(object operation)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationAbort event");
}
private void OnCalculationEnd(object operation)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationEnd event");
}
private void OnCalculationError(object operation)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationError event");
}
}
Public Class StockAutomationEventsHandler
Private ReadOnly _application As Esprit.Application
Private ReadOnly _stockAutomationObject As Esprit.StockAutomation
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As Esprit.Application)

_application.EventWindow.Clear()

AddHandler _stockAutomationObject.OnCalculationEnd, AddressOf Me.OnCalculationEnd
AddHandler _stockAutomationObject.OnCalculationAbort, AddressOf Me.OnCalculationAbort
AddHandler _stockAutomationObject.OnCalculationError, AddressOf Me.OnCalculationError
End Sub

_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationStart event")
End Sub
Private Sub OnCalculationEnd(Operation As Object)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationEnd event")
End Sub
Private Sub OnCalculationAbort(Operation As Object)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, "OnCalculationAbort event")
End Sub

VB.NET
VBA
private StockAutomationEventsHandler _stockAutomationEventHandler;
public void StockAutomationEventExample()
{
if (_stockAutomationEventHandler == null)
{
_stockAutomationEventHandler = new StockAutomationEventsHandler(_application);
}
}
Private _stockAutomationEventHandler As StockAutomationEventsHandler
Public Sub StockAutomationEventExample()
If _stockAutomationEventHandler Is Nothing Then
_stockAutomationEventHandler = New StockAutomationEventsHandler(_application)
End If
End Sub

Before running the tutorial, create new or open a file that has several operations on it. Than press Rebuild All button to start the rebuild process, all stock automation events are logged in EventWindow.
