Technology Events

Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

The Tutorial demostrates the Tools object events.

Define a class and add the event procedures:

class TechnologyEventsHandler
{
private readonly Esprit.Application _application;
private readonly string _messageBoxTitle;
EspritTechnology.Technology _tech;
EspritTechnology.TechnologyUtility _techUtility;
EspritTechnology.Parameter _techParameter;
public TechnologyEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "TechnologyEventsTutorial";
_application.AfterDocumentClose += OnAfterDocumentClose;
if (_techUtility == null)
{
_techUtility = _application.Document.TechnologyUtility;
_techUtility.AfterOpenTechnologyFile += _techUtility_AfterOpenTechnologyFile;
_techUtility.AfterSetEdmMachineData += _techUtility_AfterSetEdmMachineData;
_techUtility.OnBeforeRebuild += _techUtility_OnBeforeRebuild;
_techUtility.OnCustomizeGui += _techUtility_OnCustomizeGui;
_techUtility.OnEdit += _techUtility_OnEdit;
_techUtility.OnFeedSpeed += _techUtility_OnFeedSpeed;
_techUtility.OnGetDefaultCaptions += _techUtility_OnGetDefaultCaptions;
_techUtility.OnShutdown += _techUtility_OnShutdown;
_techUtility.OnTechPageUiLoaded += _techUtility_OnTechPageUiLoaded;
}
if (_tech == null && _application.Document.PartOperations.Count > 0)
{
_tech = _application.Document.PartOperations[1].Technology;
_tech.OnFeedSpeed += _tech_OnFeedSpeed;
_tech.OnModify += _tech_OnModify;
}
if (_techParameter == null && _tech != null)
{
int clCode = 193;
_techParameter = _tech.FindParameter(clCode);
if (_techParameter != null)
{
_techParameter.OnModify += _techParameter_OnModify;
_techParameter.OnUpdate += _techParameter_OnUpdate;
}
}
}
private void _techParameter_OnUpdate(int Flag)
{
/* Flag value mask:
#define LOCKED_STATUS (1<<0)
#define HIDDEN_STATUS (1<<1)
#define BACK_COLOR_STATUS (1<<2)
#define FORE_COLOR_STATUS (1<<3)
#define SOURCE_STATUS (1<<4)
#define VALUE_STATUS (1<<5)
#define CAPTION_STATUS (1<<6)
#define EXPR_STATUS (1<<7)
*/
MessageBox.Show($"TechParameter OnModify called on changing value via API. Prefer to use OnModify event. Flag Value {Flag}", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techParameter_OnModify()
{
MessageBox.Show($"TechParameter OnModify called on changing value", _messageBoxTitle, MessageBoxButton.OK);
}
private void _tech_OnModify(int ClCode, bool ByUser, espTechEditType TechEditType)
{
MessageBox.Show($"Technology OnModify called on changing techology ClCode {ClCode} value", _messageBoxTitle, MessageBoxButton.OK);
}
private void _tech_OnFeedSpeed(int ClCode, bool ByUser, espTechEditType TechEditType)
{
MessageBox.Show($"ClCode {ClCode} OnFeedSpeed called to demostrate that FeedSpeed should be recalculated", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnTechPageUiLoaded(object Technology)
{
var tech = Technology as EspritTechnology.Technology;
MessageBox.Show($"TechUtility-OnTechPageUiLoaded called on start editing technology { tech.Caption} when UI ready for editing", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnShutdown(object Technology, int ResultCode)
{
var tech = Technology as EspritTechnology.Technology;
MessageBox.Show($"TechUtility-OnShutdown called on finishing editing technology {tech.Caption}", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnGetDefaultCaptions(object Technology)
{
MessageBox.Show("TechUtility-OnGetDefaultCaptions calls on loading a tool from tool library or a file", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnFeedSpeed(object Technology)
{
var tech = Technology as EspritTechnology.Technology;
MessageBox.Show($"TechUtility-OnFeedSpeed called for example on rebuild command for {tech.Caption} when it requries FeedSpeed recalculation", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnEdit(object Technology, espTechEditType TechEditType, ref bool Override, ref int ResultCode)
{
var tech = Technology as EspritTechnology.Technology;
MessageBox.Show($"TechUtility-OnEdit called on start editing technology {tech.Caption}", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnCustomizeGui(object Technology, object TechPageGuiInfo)
{
var tech = Technology as EspritTechnology.Technology;
MessageBox.Show($"TechUtility-OnCustomizeGui called on start editing technology {tech.Caption}", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_OnBeforeRebuild(object Technology)
{
MessageBox.Show("TechUtility-OnBeforeRebuild called before starting rebuild command", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_AfterSetEdmMachineData(object Technology, int ClCode)
{
MessageBox.Show("TechUtility-AfterSetEdmMachineData called after API call TechnologyUtility.AfterSetEdmMachineData", _messageBoxTitle, MessageBoxButton.OK);
}
private void _techUtility_AfterOpenTechnologyFile(object Technology)
{
MessageBox.Show("TechUtility-AfterOpenTechnologyFile calls on loading a tool from tool library or a file", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnAfterDocumentClose()
{
_tech = null;
_techUtility = null;
_techParameter = null;
}
}
Imports System.Windows
Imports EspritConstants
Public Class TechnologyEventsHandler
Private ReadOnly _application As Esprit.Application
Private ReadOnly _tools As EspritTools.Tools
Private ReadOnly _messageBoxTitle As String
Private ReadOnly _tech As EspritTechnology.Technology
Private ReadOnly _techUtility As EspritTechnology.TechnologyUtility
Private ReadOnly _techParameter As EspritTechnology.Parameter
Public Sub New(app As Esprit.Application)
_application = app
_messageBoxTitle = "TechnologyEventsTutorial"
If _techUtility Is Nothing Then
_techUtility = _application.Document.TechnologyUtility
AddHandler _techUtility.AfterOpenTechnologyFile, AddressOf _techUtility_AfterOpenTechnologyFile
AddHandler _techUtility.AfterSetEdmMachineData, AddressOf _techUtility_AfterSetEdmMachineData
AddHandler _techUtility.OnBeforeRebuild, AddressOf _techUtility_OnBeforeRebuild
AddHandler _techUtility.OnCustomizeGui, AddressOf _techUtility_OnCustomizeGui
AddHandler _techUtility.OnEdit, AddressOf _techUtility_OnEdit
AddHandler _techUtility.OnFeedSpeed, AddressOf _techUtility_OnFeedSpeed
AddHandler _techUtility.OnGetDefaultCaptions, AddressOf _techUtility_OnGetDefaultCaptions
AddHandler _techUtility.OnShutdown, AddressOf _techUtility_OnShutdown
AddHandler _techUtility.OnTechPageUiLoaded, AddressOf _techUtility_OnTechPageUiLoaded
End If
If _tech Is Nothing AndAlso _application.Document.PartOperations.Count > 0 Then
_tech = _application.Document.PartOperations.Item(1).Technology
AddHandler _tech.OnFeedSpeed, AddressOf _tech_OnFeedSpeed
AddHandler _tech.OnModify, AddressOf _tech_OnModify
End If
If _techParameter Is Nothing AndAlso _tech IsNot Nothing Then
Dim clCode As Integer = 193
_techParameter = _tech.FindParameter(clCode)
If _techParameter IsNot Nothing Then
AddHandler _techParameter.OnModify, AddressOf _techParameter_OnModify
AddHandler _techParameter.OnUpdate, AddressOf _techParameter_OnUpdate
End If
End If
End Sub
