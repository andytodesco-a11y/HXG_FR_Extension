Expanding the Bolt Hole Pattern Class

The Tutorial: Using Class Modules: Bolt Hole Pattern created a class for a bolt hole pattern. This tutorial will expand upon that class by adding several Methods for manipulating the pattern after its creation. A class method is really just any procedure or function.

Define a class for creating and manipulating a bolt hole pattern:

class BoltHolePatternExpandHandler : IDisposable
{
private readonly Esprit.Application _application;
public double HoleDiameter;
public double PatternDiameter;
public double StartAngleRad;
public int NumberOfHoles;
public bool IsBoltHoleCreated;
public Esprit.Point LocalPatternCenter;
public List<Esprit.Circle> Holes;
public BoltHolePatternExpandHandler(Esprit.Application app)
{
_application = app;
if (_application.Document.SystemUnit == EspritConstants.espUnitType.espInch)
{
HoleDiameter = 1;
PatternDiameter = 10;
}
else
{
HoleDiameter = 25;
PatternDiameter = 250;
}
NumberOfHoles = 10;
StartAngleRad = 0;
LocalPatternCenter = null;
Holes = new List<Esprit.Circle>();
}
public void Dispose()
{
RemoveHoles();
}
public void RemoveHoles()
{
foreach(var hole in Holes)
{
_application.Document.Circles.Remove(hole.Key);
}
Holes.Clear();
_application.Document.Refresh();
}
public void Move(Esprit.Point point)
{
var xMove = point.X - LocalPatternCenter.X;
var yMove = point.Y - LocalPatternCenter.Y;
foreach(var hole in Holes)
{
hole.X += xMove;
hole.Y += yMove;
}
LocalPatternCenter = point;
_application.Document.Refresh();
}
public List<Esprit.Circle> DrawPattern(Esprit.Point point)
{
RemoveHoles();
if (point != null)
{
LocalPatternCenter = point;
}
if (LocalPatternCenter != null)
{
for (var i = 0; i < NumberOfHoles; i++)
{
var angle = StartAngleRad + i * 2 * Math.PI / NumberOfHoles;
var x = (PatternDiameter / 2) * Math.Cos(angle) + LocalPatternCenter.X;
var y = (PatternDiameter / 2) * Math.Sin(angle) + LocalPatternCenter.Y;
Holes.Add(_application.Document.Circles.Add(_application.Document.GetPoint(x, y, 0), HoleDiameter / 2.0));
}
}
_application.Document.Refresh();
return Holes;
}
}

Private Holes() As Esprit.Circle
Private LocalPatternCenter As Esprit.Point
Public HoleDiameter As Double
Public NumberOfHoles As Long
Public PatternDiameter As Double
Public StartAngle As Double ' assumed to be in radians
Private Sub Class_Initialize()
' not absolutely necessary, but can set some default values here
If Document.SystemUnit = espInch Then
HoleDiameter = 1
PatternDiameter = 10
Else
HoleDiameter = 25
PatternDiameter = 250
End If
StartAngle = 0
NumberOfHoles = 10
End Sub
Function DrawPattern(PatternCenter As Esprit.Point) As Esprit.Circle()
Set LocalPatternCenter = PatternCenter
ReDim Holes(0 To NumberOfHoles - 1) As Esprit.Circle
Dim HoleNumber As Long
' loop through all of the holes in the pattern
For HoleNumber = 0 To NumberOfHoles - 1
' calculate the angle of the center of this hole
' with respect to the center of the pattern
Angle = StartAngle + HoleNumber * 2 * (4 * Atn(1)) / NumberOfHoles
' calculate the coordinates of the center of the hole
' relative the center of the pattern
X = (PatternDiameter / 2) * Cos(Angle) + PatternCenter.X
Y = (PatternDiameter / 2) * Sin(Angle) + PatternCenter.Y
' add a circle on those coordinates (without adding a point at the center)
Dim HoleCenter As Esprit.Point
Set HoleCenter = Document.GetPoint(X, Y, 0)
Set Holes(HoleNumber) = Document.Circles.Add(HoleCenter, HoleDiameter / 2)
Next
' redraw the graphics
Document.Refresh
' return the array
DrawPattern = Holes
End Function
Function UpdatePattern() As Esprit.Circle()
Dim HoleNumber As Long
Dim PreviousNumberOfHoles As Long
PreviousNumberOfHoles = UBound(Holes) + 1
' see if number of holes has changed
If NumberOfHoles > PreviousNumberOfHoles Then
' must redimension if increased
ReDim Preserve Holes(0 To NumberOfHoles - 1) As Esprit.Circle
End If
If NumberOfHoles < PreviousNumberOfHoles Then
' must remove excess holes if decreased
For HoleNumber = (PreviousNumberOfHoles - 1) To NumberOfHoles Step -1
Call Document.Circles.Remove(Holes(HoleNumber).Key)
Set Holes(HoleNumber) = Nothing
Next
End If
' loop through all of the holes in the pattern
For HoleNumber = 0 To NumberOfHoles - 1
' calculate the angle of the center of this hole
' with respect to the center of the pattern
Angle = StartAngle + HoleNumber * 2 * (4 * Atn(1)) / NumberOfHoles
' calculate the coordinates of the center of the hole
' relative the center of the pattern
X = (PatternDiameter / 2) * Cos(Angle) + LocalPatternCenter.X
Y = (PatternDiameter / 2) * Sin(Angle) + LocalPatternCenter.Y
' check to see if hole already exists
If Holes(HoleNumber) Is Nothing Then
' add a circle on those coordinates (without adding a point at the center)
Dim HoleCenter As Esprit.Point
Set HoleCenter = Document.GetPoint(X, Y, 0)
Set Holes(HoleNumber) = Document.Circles.Add(HoleCenter, HoleDiameter / 2)
Else
' update the circle on those coordinates
Holes(HoleNumber).X = X
Holes(HoleNumber).Y = Y
Holes(HoleNumber).Radius = HoleDiameter / 2
End If
Next
' redraw the graphics
Document.Refresh
' return the array
UpdatePattern = Holes
End Function
Sub Move(NewCenterPoint As Esprit.Point)
Dim XMove As Double, YMove As Double
XMove = NewCenterPoint.X - LocalPatternCenter.X
YMove = NewCenterPoint.Y - LocalPatternCenter.Y
' loop through and move each hole
For HoleNumber = 0 To UBound(Holes)
Holes(HoleNumber).X = Holes(HoleNumber).X + XMove
Holes(HoleNumber).Y = Holes(HoleNumber).Y + YMove
Next
Set LocalPatternCenter = NewCenterPoint
' redraw the graphics
Document.Refresh
End Sub
Sub Remove()
For HoleNumber = UBound(Holes) To LBound(Holes) Step -1
Call Document.Circles.Remove(Holes(HoleNumber).Key)
Set Holes(HoleNumber) = Nothing
Next
' redraw the graphics
Document.Refresh
End Sub

private void OnCreateButtonClick(object sender, EventArgs e)

MakeHole();

private void MakeHole()
{
if (_boltHolePattern == null)
{
_boltHolePattern = new BoltHolePatternExpandHandler(_application);
_boltHolePattern.IsBoltHoleCreated = false;
}
var holeDiameter = _boltHolePattern.HoleDiameter;
var patternDiameter = _boltHolePattern.PatternDiameter;
var numberOfHoles = _boltHolePattern.NumberOfHoles;
var startAngle = (_boltHolePattern.StartAngleRad * 180/ Math.PI);
if (RequestUserInput("Enter Hole Diameter", "Hole Diameter", ref holeDiameter) &&
RequestUserInput("Enter Pattern Diameter", "Pattern Diameter", ref patternDiameter) &&
RequestUserInput("Enter Number of Holes", "Holes", ref numberOfHoles) &&
RequestUserInput("Enter New Start Angle in Degrees", "Start Angle", ref startAngle) )
{
if (holeDiameter > 0 && patternDiameter > 0 && numberOfHoles > 0)
{
_boltHolePattern.HoleDiameter = holeDiameter;
_boltHolePattern.NumberOfHoles = numberOfHoles;
_boltHolePattern.StartAngleRad = startAngle * Math.PI / 180;
_boltHolePattern.HoleDiameter = holeDiameter;
Esprit.Point point = null;
if (!_boltHolePattern.IsBoltHoleCreated)
{
try
{
point = Document.GetPoint("Pick The Pattern Center");
}
catch (System.Runtime.InteropServices.COMException)
{
// Point was not set
_boltHolePattern.IsBoltHoleCreated = false;
return;
}
_boltHolePattern.IsBoltHoleCreated = true;
}
_boltHolePattern.DrawPattern(point);
}
}
}

Dim BoltHolePattern As BoltHolePatternExpHandler
Sub MakeBoltHolePattern()
If BoltHolePattern Is Nothing Then
Set BoltHolePattern = New BoltHolePatternExpHandler
End If
' prompt the user for parameters and draw the bolt hole pattern
With BoltHolePattern
.HoleDiameter = Val(InputBox("Enter Hole Diameter", "Hole Diameter", .HoleDiameter))
.NumberOfHoles = Val(InputBox("Enter Number of Holes", "Holes", .NumberOfHoles))
.PatternDiameter = Val(InputBox("Enter Pattern Diameter", "Pattern Diameter", .PatternDiameter))
.StartAngle = Val(InputBox("Enter Start Angle in Degrees", "Start Angle", .StartAngle * 180 / (4 * Atn(1))) * 4 * Atn(1) / 180)
On Error GoTo ErrorHandler
Call .DrawPattern(Document.GetPoint("Pick The Pattern Center"))
End With
ErrorHandler:
Exit Sub
End Sub
Sub UpdateBoltHolePattern()
If BoltHolePattern Is Nothing Then
Call MakeBoltHolePattern
Else
' prompt the user for new parameters and update the bolt hole pattern
With BoltHolePattern
.HoleDiameter = Val(InputBox("Enter New Hole Diameter", "Hole Diameter", .HoleDiameter))
.NumberOfHoles = Val(InputBox("Enter New Number of Holes", "Holes", .NumberOfHoles))
.PatternDiameter = Val(InputBox("Enter New Pattern Diameter", "Pattern Diameter", .PatternDiameter))
.StartAngle = Val(InputBox("Enter New Start Angle in Degrees", "Start Angle", .StartAngle * 180 / (4 * Atn(1))) * 4 * Atn(1) / 180)
Call .UpdatePattern
End With
End If
End Sub
Sub MoveBoltHolePattern()
If BoltHolePattern Is Nothing Then Exit Sub
On Error GoTo ErrorHandler
Call BoltHolePattern.Move(Document.GetPoint("Pick The New Pattern Center"))
ErrorHandler:
Exit Sub
End Sub
Sub RemoveBoltHolePattern()
If Not BoltHolePattern Is Nothing Then
BoltHolePattern.Remove
End If
Set BoltHolePattern = Nothing
End Sub
