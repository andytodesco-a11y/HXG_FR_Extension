Graphics Events

In Tutorial: Introduction to Custom Properties we added a custom Diameter property to all of our circles. However, new circles that were created after we ran this code would not have the Diameter property, and existing circles that had their Radius value changed did not have the Diameter already updated. How can we ensure that all of our circles always have a correct Diameter value?

We need to know when circles are created and modified. If we search the ESPRIT EDGE library, we will discover that the GraphicsCollection has OnCreate and OnModify events. Those events are triggered whenever graphics are created or modified.

First define a class that responds to those events and loops through GraphicsArray containing all of the GraphicObject objects that were created or modified to pick out the circles:

class GraphicsEventsHandler
{
Esprit.EventWindow _eventWindow;
//Collect objects and properties to catch events
List<Esprit.IGraphicObject> _graphicObjects = new List<Esprit.IGraphicObject>();
List<EspritProperties.CustomProperty> _customProperties = new List<EspritProperties.CustomProperty>();
public GraphicsEventsHandler(Esprit.Application app)
{
app.Document.GraphicsCollection.OnCreate += GraphicsCollectionOnCreate;
app.Document.GraphicsCollection.OnModify += GraphicsCollectionOnModify;
app.Document.GraphicsCollection.OnFiredProgrammableEvent += GraphicsCollection_OnFiredProgrammableEvent;
app.Document.GraphicsCollection.OnDelete += GraphicsCollection_OnDelete;
_eventWindow = app.EventWindow;
foreach (Esprit.GraphicObject circle in app.Document.Circles)
{
_graphicObjects.Add(circle);
var property = GraphicObjectHelper.GetOrAddCustomProperty(circle, "Diameter", EspritConstants.espPropertyType.espPropertyTypeDouble);
property.Caption = "Diameter";
property.ReadOnly = true;
property.Value = 2 * (circle as Esprit.Circle).Radius;
_customProperties.Add(property);
var comCircle = circle as EspritGeometry.ComCircle;
property.OnValueChanged += Property_OnValueChanged;
comCircle.OnBeforeModify += ComCircle_OnBeforeModify;
comCircle.OnModify += ComCircle_OnModify;
circle.FireProgrammableEvent();
}
}
private void GraphicsCollection_OnDelete(object GraphicsArray)
{
var goArray = GraphicsArray as Esprit.IGraphicsArray;
for (var i = 1; i <= goArray.Count; i++)
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "GraphicsEventsTutorial", $"a {goArray[i].GuiTypeName} deleted");
}
}
private void GraphicsCollection_OnFiredProgrammableEvent(object GraphicObject)
{
var go = GraphicObject as Esprit.IGraphicObject;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "GraphicsEventsTutorial", $"a {go.GuiTypeName} fired programmable event");
}
private void ComCircle_OnModify()
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CustomPropertiesTutorial", $"ComCircle_OnModify called after object modifying");
}
private void ComCircle_OnBeforeModify()
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CustomPropertiesTutorial", $"OnBeforeModify called before object modifying");
}
private void GraphicsCollectionOnCreate(object graphicsArray)
{
var data = graphicsArray as System.Collections.IEnumerable;
foreach (Esprit.GraphicObject graphicObject in data)
{
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espCircle)
{
_graphicObjects.Add(graphicObject);
var property = GraphicObjectHelper.GetOrAddCustomProperty(graphicObject, "Diameter", EspritConstants.espPropertyType.espPropertyTypeDouble);
property.Caption = "Diameter";
property.ReadOnly = true;
property.Value = 2 * (graphicObject as Esprit.Circle).Radius;
_customProperties.Add(property);
var comCircle = graphicObject as EspritGeometry.ComCircle;
property.OnValueChanged += Property_OnValueChanged;
comCircle.OnBeforeModify += ComCircle_OnBeforeModify;
comCircle.OnModify += ComCircle_OnModify;
}
}
}
private void GraphicsCollectionOnModify(object graphicsArray)
{
var data = graphicsArray as System.Collections.IEnumerable;
foreach(Esprit.GraphicObject graphicObject in data)
{
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espCircle)
{
var property = GraphicObjectHelper.GetOrAddCustomProperty(graphicObject, "Diameter", EspritConstants.espPropertyType.espPropertyTypeDouble);
property.Value = 2 * (graphicObject as Esprit.Circle).Radius;
}
}
}
private void Property_OnValueChanged(EspritProperties.CustomProperty Property)
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CustomPropertiesTutorial", $"Custom property \"{Property.Name}\" changed to {Property.Value}");
}
}
static public EspritProperties.CustomProperty GetOrAddCustomProperty(Esprit.IGraphicObject graphicObject, string name, EspritConstants.espPropertyType type)
{
EspritProperties.CustomProperty customProperty = graphicObject.CustomProperties.Item(name);
if (customProperty != null && customProperty.VariableType != type)
{
graphicObject.CustomProperties.Remove(name);
customProperty = null;
}
if (customProperty == null)
{
var properties = graphicObject.CustomProperties as EspritProperties.CustomProperties;
customProperty = (type == EspritConstants.espPropertyType.espPropertyTypeString)
? properties.Add(name, name, type, "")
: properties.Add(name, name, type, 0);
}
return customProperty;
}

Public Sub New(app As ESPRIT.Application)
_application = app
AddHandler app.Document.GraphicsCollection.OnCreate, AddressOf Me.GraphicsCollectionOnCreate
AddHandler app.Document.GraphicsCollection.OnModify, AddressOf Me.GraphicsCollectionOnModify
AddHandler app.Document.GraphicsCollection.OnFiredProgrammableEvent, AddressOf Me.GraphicsCollection_OnFiredProgrammableEvent
AddHandler app.Document.GraphicsCollection.OnDelete, AddressOf Me.GraphicsCollection_OnDelete
For Each circle As ESPRIT.GraphicObject In app.Document.Circles
_graphicObjects.Add(circle)
Dim prop = GraphicObjectHelper.GetOrAddCustomProperty(circle, "Diameter", EspritConstants.espPropertyType.espPropertyTypeDouble)
prop.Caption = "Diameter"
prop.ReadOnly = True
prop.Value = 2 * (CType(circle, ESPRIT.Circle)).Radius
_customProperties.Add(prop)
Dim comCircle As EspritGeometry.ComCircle = circle
AddHandler prop.OnValueChanged, AddressOf Property_OnValueChanged
AddHandler comCircle.OnBeforeModify, AddressOf ComCircle_OnBeforeModify
AddHandler comCircle.OnModify, AddressOf ComCircle_OnModify

Next
End Sub

Public Function GetOrAddCustomProperty(graphicObject As Esprit.IGraphicObject, name As String, type As EspritConstants.espPropertyType) As EspritProperties.CustomProperty
Dim customProperty As EspritProperties.CustomProperty = graphicObject.CustomProperties.Item(name)
If Not customProperty Is Nothing AndAlso customProperty.VariableType <> type Then
graphicObject.CustomProperties.Remove(name)
customProperty = Nothing
End If
If customProperty Is Nothing Then
Dim properties As EspritProperties.CustomProperties = graphicObject.CustomProperties
If type = EspritConstants.espPropertyType.espPropertyTypeString Then
customProperty = properties.Add(name, name, type, "")
Else
customProperty = properties.Add(name, name, type, 0)
End If
End If
Return customProperty
End Function
Dim WithEvents MyGraphics As Esprit.GraphicsCollection
Attribute MyGraphics.VB_VarHelpID = -1
Dim PropCount
'_IOnModifyEvent interface is not available for VBA script
Dim Prop() As PropertyEvents
Private Sub Class_Initialize()
Set MyGraphics = Document.GraphicsCollection
PropCount = 0
' loop through all of the existing circles to make sure they have a diameter
Dim C As Esprit.Circle
For Each C In Document.Circles
' get a diameter custom property and set its value accordingly
Dim CP As EspritProperties.CustomProperty
Set CP = GetCustomProperty(C, "Diameter", espPropertyTypeDouble)
PropCount = PropCount + 1
ReDim Preserve Prop(1 To PropCount) As PropertyEvents
Dim PropertyEvent As PropertyEvents
Set PropertyEvent = New PropertyEvents
Call PropertyEvent.Init(CP)
Set Prop(PropCount) = PropertyEvent
CP.Caption = "Diameter"
CP.ReadOnly = True
CP.Value = 2 * C.Radius
Next
End Sub
Private Sub MyGraphics_OnCreate(ByVal GraphicsArray As Object)
Dim G As Esprit.GraphicObject
Dim I As Long
For I = 1 To GraphicsArray.Count
Set G = GraphicsArray(I)
If G.GraphicObjectType = espCircle Then
Dim C As Esprit.Circle
Set C = G
' add the diameter propety to the newly created circle
Dim CP As EspritProperties.CustomProperty
Set CP = GetCustomProperty(C, "Diameter", espPropertyTypeDouble)
PropCount = PropCount + 1
ReDim Preserve Prop(1 To PropCount) As PropertyEvents
Dim PropertyEvent As PropertyEvents
Set PropertyEvent = New PropertyEvents
Call PropertyEvent.Init(CP)
Set Prop(PropCount) = PropertyEvent
CP.Caption = "Diameter"
CP.ReadOnly = True
CP.Value = 2 * C.Radius
C.FireProgrammableEvent
End If
Next
End Sub
Private Sub MyGraphics_OnModify(ByVal GraphicsArray As Object)
Dim G As Esprit.GraphicObject
Dim I As Long
For I = 1 To GraphicsArray.Count
Set G = GraphicsArray(I)
If G.GraphicObjectType = espCircle Then
Dim C As Esprit.Circle
Set C = G
' update the diameter on the modified circle
Dim CP As EspritProperties.CustomProperty
Set CP = GetCustomProperty(C, "Diameter", espPropertyTypeDouble)
' since it should already have been created/added
' we should only have to set the value, not the caption or readonly
' but we could set those also just to be safe
CP.Value = 2 * C.Radius
End If
Next
End Sub
Private Sub MyGraphics_OnFiredProgrammableEvent(ByVal GraphicsObject As Object)
Dim G As Esprit.GraphicObject
Set G = GraphicsObject
Call MsgBox("A " & G.GuiTypeName & " fired programmable event")
End Sub
Private Sub MyGraphics_OnDelete(ByVal GraphicsArray As Object)
Dim goArray As Esprit.IGraphicsArray
Set goArray = GraphicsArray
Dim i As Long
For i = 1 To goArray.Count
Call MsgBox("A " & goArray.Item(i).GuiTypeName & " deleted")
Next
End Sub
Function GetCustomProperty(GraphicObj As Esprit.GraphicObject, PropertyName As String, PropertyType As EspritConstants.espPropertyType) As EspritProperties.CustomProperty
On Error Resume Next
Set GetCustomProperty = GraphicObj.CustomProperties.Item(PropertyName)
On Error GoTo 0
If Not (GetCustomProperty Is Nothing) Then
' check the type as well
If GetCustomProperty.VariableType <> PropertyType Then
' must remove if not the same type
Call GraphicObj.CustomProperties.Remove(PropertyName)
Set GetCustomProperty = Nothing
End If
End If
' add new property if necessary, with an appropriate default value
If GetCustomProperty Is Nothing Then
If PropertyType = espPropertyTypeString Then
Set GetCustomProperty = GraphicObj.CustomProperties.Add(PropertyName, PropertyName, PropertyType, "")
Else
Set GetCustomProperty = GraphicObj.CustomProperties.Add(PropertyName, PropertyName, PropertyType, 0)
End If
End If
End Function
