Create Tools

In this tutorial you will create a couple of milling tools. Similar to creating other objects, tool objects are created using the Add method of the Tools collection. What is different with tools is that the technology that defines a tool is itself an object and it comes from the separate EspritTechnology library, there are a lot of specific types of tool technology objects, and in order to create a tool, you must first create the corresponding tool technology object and then add that technology object to the tools collection. It can be done using the CreateTechnology method of the TechnologyUtility object.

After we create a new tool technology object, you will notice that they have many more properties than other objects. Working with them is no different than working with the properties of other objects, there are just more properties to deal with.

public void CreateTools()
{
var technologyUtility = Document.TechnologyUtility as EspritTechnology.TechnologyUtility;
var toolEndMill = technologyUtility.CreateTechnology(EspritConstants.espTechnologyType.espToolMillEndMill, Document.SystemUnit);
var toolDrill = technologyUtility.CreateTechnology(EspritConstants.espTechnologyType.espToolMillDrill, Document.SystemUnit);
var parameter = toolEndMill.FindParameter("ToolID");
if (parameter != null)
{
parameter.Value = "EM 3";
}
parameter = toolEndMill.FindParameter("Comment");
if (parameter != null)
{
parameter.Value = "This endmill was added by C# tutorial extension";
}
parameter = toolEndMill.FindParameter("ToolNumber");
if (parameter != null)
{
parameter.Value = 3;
}
parameter = toolEndMill.FindParameter("Coolant");
if (parameter != null)
{
parameter.Value = EspritConstants.espCoolantType.espCoolantOn;
}
parameter = toolDrill.FindParameter("ToolID");
if (parameter != null)
{
parameter.Value = "DR 4";
}
parameter = toolDrill.FindParameter("Comment");
if (parameter != null)
{
var stringBuilder = new StringBuilder();
stringBuilder.AppendLine("This drill was added by C# tutorial extension");
stringBuilder.AppendLine("Notice how multiple line comments are generated by");
stringBuilder.AppendLine("adding carriage return and line feed characters.");
parameter.Value = stringBuilder.ToString();
}
parameter = toolDrill.FindParameter("ToolNumber");
if (parameter != null)
{
parameter.Value = 4;
}
parameter = toolDrill.FindParameter("Coolant");
if (parameter != null)
{
parameter.Value = EspritConstants.espCoolantType.espCoolantFlood;
}
var tools = Document.Tools as EspritTools.Tools;
try
{
tools.Add(toolEndMill);
tools.Add(toolDrill);
}
catch (Exception)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeWarning, "CreateToolsTutorial", "Exception occured - failed to add tool");
}
}
Public Sub CreateTools()
Dim techUtil As EspritTechnology.TechnologyUtility = Document.TechnologyUtility
Dim documentTools As EspritTools.Tools = Document.Tools
' declare and set the tool technologies from the EspritTechnology library
Dim endMill As EspritTechnology.ToolMillEndMill = techUtil.CreateTechnology(EspritConstants.espTechnologyType.espToolMillEndMill, Document.SystemUnit)
Dim drill As EspritTechnology.ToolMillDrill = techUtil.CreateTechnology(EspritConstants.espTechnologyType.espToolMillDrill, Document.SystemUnit)
With endMill
.ToolID = "EM 3" ' always be sure to set the ToolID
.Comment = "This endmill was added by a Vb.Net tutorial extension"
.Coolant = EspritConstants.espCoolantType.espCoolantOn
.ToolNumber = 3
' All other endmill properties that you wish to set can go here
End With
With drill
.ToolID = "DR 4" ' always be sure to set the ToolID
.Comment = "This drill was added by a Vb.Net tutorial extension." & Chr(13) & Chr(10)
.Comment = .Comment & " Notice how multiple line comments are generated by" & Chr(13) & Chr(10)
.Comment = .Comment & " adding carriage return and line feed characters."
.Coolant = EspritConstants.espCoolantType.espCoolantFlood
.ToolNumber = 4
' All other drill properties that you wish to set can go here
End With
' now add those tool technologies to the Tools collection
Try
Call documentTools.Add(endMill)
Call documentTools.Add(drill)
Catch ex As Exception
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeWarning, "CreateToolsTutorial", "Exception occured - failed to add tool")
End Try
End Sub
Sub CreateTools()
Dim TechUtil As EspritTechnology.TechnologyUtility
Set TechUtil = Document.TechnologyUtility
Dim MyTools As EspritTools.Tools
Set MyTools = Document.Tools
' declare and set the tool technologies from the EspritTechnology library
Dim EM As EspritTechnology.ToolMillEndMill
Dim DR As EspritTechnology.ToolMillDrill
Set EM = TechUtil.CreateTechnology(espToolMillEndMill, Document.SystemUnit)
Set DR = TechUtil.CreateTechnology(espToolMillDrill, Document.SystemUnit)
With EM
.ToolID = "EM 3" ' always be sure to set the ToolID
.Comment = "This endmill was added by VBA tutorial extension"
.Coolant = espCoolantOn
.ToolNumber = 3
' All other endmill properties that you wish to set can go here
End With
With DR
.ToolID = "DR 4" ' always be sure to set the ToolID
.Comment = "This drill was added by VBA tutorial extension." & Chr(13) & Chr(10)
.Comment = .Comment & " Notice how multiple line comments are generated by" & Chr(13) & Chr(10)
.Comment = .Comment & " adding carriage return and line feed characters."
.Coolant = espCoolantFlood
.ToolNumber = 4
' All other drill properties that you wish to set can go here
End With
' now add those tool technologies to the Tools collection
On Error Resume Next
Call MyTools.Add(EM)
Call MyTools.Add(DR)
End Sub

Go to the tool tab in the Project Manager after running the code and you will see the newly created tools in the list. Edit the tools and check the properties that you set. Modify your code and set more properties of your choice and rerun it.
