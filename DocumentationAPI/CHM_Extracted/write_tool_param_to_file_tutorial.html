Write Tool Parameters to a File

This tutorial shows how to outputs all of the tools and all of their parameters into a file.

public void WriteToolParameters()
{
var folderPath = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage);
var tools = Document.Tools as EspritTools.Tools;
var index = 0;
foreach (EspritTechnology.ITechnology technology in tools)
{
index++;
var filePath = System.IO.Path.Combine(folderPath, $"WriteToolParametersToFileTutorial_{index}.test");
technology.Save(filePath);
}
if (index == 0)
MessageBox.Show("No tools to save", "WriteToolParametersToFileTutorial");
else
MessageBox.Show($"{index} files saved to {System.IO.Path.Combine(folderPath)}", "WriteToolParametersToFileTutorial");
}
Public Sub WriteToolParameters()
Dim index As Integer = 0
Dim toolFolder As String = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage)
For Each toolTech As EspritTechnology.Technology In Document.Tools
index += 1
Dim fileName As String = toolFolder & "\WriteToolParametersToFileTutorial_" & index & ".test"
toolTech.Save(fileName)
Next
If index = 0 Then
MsgBox("No tools to save ", vbOKOnly, "WriteToolParametersToFileTutorial")
Else
' tell the user where the file is located
Call MsgBox("Created " & index & "Files at location: " & toolFolder, vbOKOnly, "WriteToolParametersToFileTutorial")
End If
End Sub
