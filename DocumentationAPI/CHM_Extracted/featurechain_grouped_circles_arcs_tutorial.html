FeatureChain Grouped Circles and Arcs

In Tutorial: FeatureChain Visible Circles and Arcs we looped through all of the Circle and Arc objects and put a FeatureChain on each of the visible ones. This tutorial shows how to feature chain the circles and arcs that are in the Group.

public void FeatureChainGroupedCirclesAndArcs()
{
var angle = 90.0;
if (!RequestUserInput("Enter Starting Angle", "Starting Angle", ref angle))
{
return;
}
angle = angle * Math.PI / 180;
var previousActiveColor = Document.ActiveColor;
Document.ActiveColor = ColorHelper.ColorToUInt(System.Drawing.Color.Blue);
foreach (Esprit.Circle circul in Document.Circles)
{
if (circul.Grouped)
{
var featureChain = Document.FeatureChains.Add(Document.GetPoint(circul.X + circul.Radius * Math.Cos(angle), circul.Y + circul.Radius * Math.Sin(angle), circul.Z));
featureChain.Layer = circul.Layer;
featureChain.Add(Document.GetArc(circul.CenterPoint, circul.Radius, angle, 2 * Math.PI + angle));
}
}
foreach (Esprit.Arc arc in Document.Arcs)
{
if (arc.Grouped && (Math.Abs(arc.EndAngle - arc.StartAngle) == 2 * Math.PI))
{
var featureChain = Document.FeatureChains.Add(arc.Extremity(EspritConstants.espExtremityType.espExtremityStart));
featureChain.Layer = arc.Layer;
featureChain.Add(arc);
}
}
Document.ActiveColor = previousActiveColor;
WindowHelper.FitAllWindows(Document.Windows);
}

Public Sub FeatureChainGroupedCirclesAndArcs()
Dim angle As Double = 90.0
If Not RequestUserInput("Enter Starting Angle", "Starting Angle", angle) Then
Return
End If
angle = angle * Math.PI / 180 'convert to Radians
' remember current settings
Dim activeClr As Long = Document.ActiveColor
Document.ActiveColor = ColorToUInt(System.Drawing.Color.Blue)
' loop through all the circles
For Each circle As Esprit.Circle In Document.Circles
If circle.Grouped Then
' create the feature chain
Dim featureChain As Esprit.FeatureChain = Document.FeatureChains.Add(Document.GetPoint(circle.X + circle.Radius * Math.Cos(angle), circle.Y + circle.Radius * Math.Sin(angle), circle.Z))
featureChain.Layer = circle.Layer
' because you cannot add a circle to a FeatureChain, make temporary arc instead
Call featureChain.Add(Document.GetArc(circle.CenterPoint, circle.Radius, angle, 2 * Math.PI + angle))
End If
Next
' loop through all the arcs
For Each arc As Esprit.Arc In Document.Arcs
If arc.Grouped Then
If (Math.Abs(arc.EndAngle - arc.StartAngle) = 2 * Math.PI) Then
' create the feature chain
Dim featureChain As Esprit.FeatureChain = Document.FeatureChains.Add(arc.Extremity(EspritConstants.espExtremityType.espExtremityStart))
Call featureChain.Add(arc)
featureChain.Layer = arc.Layer
End If
End If
Next
' reset to original value
Document.ActiveColor = activeClr
Call WindowHelper.FitAllWindows(Document.Windows)
Document.Refresh()
End Sub

Public Function ColorToUInt(color As System.Drawing.Color) As UInteger
Dim a As UInteger = color.A
Dim r As UInteger = color.R
Dim g As UInteger = color.G
Dim b As UInteger = color.B
Dim resuls As UInteger = ((a << 24) Or (b << 16) Or (g << 8) Or (r << 0))
Return resuls
End Function

Sub FeatureChainGroupedCirclesAndArcs()
Dim PI As Double
PI = 4 * Atn(1)
Dim A As Esprit.Arc
Dim C As Esprit.Circle
Dim FC As Esprit.FeatureChain
Dim Angle As Double
Angle = InputBox("Enter Starting Angle", "Starting Angle", 90)
Angle = Angle * PI / 180 'convert to Radians
' remember current settings
Dim ActiveClr As Long
ActiveClr = Document.ActiveColor
Document.ActiveColor = vbBlue
' loop through all the circles
For Each C In Document.Circles
If C.Grouped Then
' create the feature chain
Set FC = Document.FeatureChains.Add(Document.GetPoint(C.X + C.Radius * Cos(Angle), C.Y + C.Radius * Sin(Angle), C.Z))
FC.Layer = C.Layer
' because you cannot add a circle to a FeatureChain, make temporary arc instead
Call FC.Add(Document.GetArc(C.CenterPoint, C.Radius, Angle, 2 * PI + Angle))
End If
Next
' loop through all the arcs
For Each A In Document.Arcs
If A.Grouped Then
If (Abs(A.EndAngle - A.StartAngle) = 2 * PI) Then
' create the feature chain
Set FC = Document.FeatureChains.Add(A.Extremity(espExtremityStart))
Call FC.Add(A)
FC.Layer = A.Layer
End If
End If
Next
' reset to original value
Document.ActiveColor = ActiveClr
Call FitAllWindows
Document.Refresh
End Sub

W.Fit

' it does not redraw the graphics, so we still need a Refresh
W.Refresh
Next
End Sub
