Group Events

Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

The Tutorial demostrates the Group object events.

Define a class and add the event procedures:

VB.NET
VBA
class GroupEventsHandler
{
Esprit.EventWindow _eventWindow;
private readonly string _messageBoxTitle;
private readonly Esprit.Group _group;
public GroupEventsHandler(Esprit.Application app)
{
_eventWindow = app.EventWindow;
_group = app.Document.Group as Esprit.Group;
_messageBoxTitle = "GroupEventsTutorial";
_group.AfterDelete += _group_AfterDelete;
_group.BeforeDelete += _group_BeforeDelete;
_group.OnAdd += _group_OnAdd;
_group.OnChanged += _group_OnChanged;
_group.OnRemove += _group_OnRemove;
}
private void _group_OnRemove(ref System.Array objects)
{
foreach (var obj in objects)
{
var go = obj as Esprit.IGraphicObject;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"A {go.GuiTypeName} removed from group");
}
}
private void _group_OnChanged()
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Group collection changed");
}
private void _group_OnAdd(ref System.Array objects, object DigitizedInfo)
{
foreach (var obj in objects)
{
var go = obj as Esprit.IGraphicObject;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"A {go.GuiTypeName} added to group");
}
}
private void _group_BeforeDelete(ref bool Override)
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called before selected graphic objects deleted");
}
private void _group_AfterDelete()
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called after selected graphic objects deleted");
}
}
Public Class GroupEventsHandler
Private ReadOnly _application As Esprit.Application
Private ReadOnly _group As Esprit.Group
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As Esprit.Application)
_application = app
_group = app.Document.Group
_messageBoxTitle = "GroupEventsTutorial"
AddHandler _group.AfterDelete, AddressOf Me.Group_AfterDelete
AddHandler _group.BeforeDelete, AddressOf Me.Group_BeforeDelete
AddHandler _group.OnAdd, AddressOf Me.Group_OnAdd
AddHandler _group.OnChanged, AddressOf Me.Group_OnChanged
AddHandler _group.OnRemove, AddressOf Me.Group_OnRemove
End Sub
Private Sub Group_OnRemove(ByRef objects As System.Array)
For Each obj As Object In objects
Dim go As Esprit.IGraphicObject = obj
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"A {go.GuiTypeName} removed from group")
Next
End Sub
Private Sub Group_OnChanged()
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Group collection changed")
End Sub
Private Sub Group_OnAdd(ByRef objects As System.Array, DigitizedInfo As Object)
For Each obj As Object In objects
Dim go As Esprit.IGraphicObject = obj
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"A {go.GuiTypeName} added to group")
Next
End Sub
Private Sub Group_BeforeDelete(ByRef Override As Boolean)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called before selected graphic objects deleted")
End Sub
Private Sub Group_AfterDelete()
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called after selected graphic objects deleted")
End Sub
End Class
Dim WithEvents GroupObject As Esprit.Group
Attribute GroupObject.VB_VarHelpID = -1
Dim MessageBoxTitle As String
Private Sub Class_Initialize()
Set GroupObject = Application.Document.Group
End Sub
Private Sub GroupObject_OnRemove(ByRef objects() As Object)
For Each obj In objects
Dim go As Esprit.IGraphicObject
Set go = obj
Call MsgBox("A " & go.GuiTypeName & " removed from group")
Next
End Sub
Private Sub GroupObject_OnChanged()
Call MsgBox("Group collection changed")
End Sub
Private Sub GroupObject_OnAdd(ByRef objects() As Object, ByVal DigitizedInfo As Object)
For Each obj In objects
Dim go As Esprit.IGraphicObject
Set go = obj
Call MsgBox("A " & go.GuiTypeName & " added to group")
Next
End Sub
Private Sub GroupObject_BeforeDelete(ByRef Override As Boolean)
Call MsgBox("Called before selected graphic objects deleted")
End Sub
Private Sub GroupObject_AfterDelete()
Call MsgBox("Called after selected graphic objects deleted")
End Sub

Before running the tutorial, create new or open a document that has several graphic objects on it. Than play with selection, information about occurence of each GroupEvent is logged in EventWindow.
