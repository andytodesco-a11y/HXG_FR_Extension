NcCode Events

Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

The Tutorial demostrates the NcCode object events.

Define a class and add the event procedures:

VB.NET
VBA
class NcCodeEventsHandler
{
private readonly Esprit.Application _application;
private readonly Esprit.NCCode _ncCodeObject;
private readonly string _messageBoxTitle;
public NcCodeEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "NcCodeEventsTutorial";
_ncCodeObject = _application.Document.NCCode;
_ncCodeObject.OnCreateNCOutput += NcCodeObject_OnCreateNCOutput;
_ncCodeObject.OnExecute += NcCodeObject_OnExecute;
}
private void NcCodeObject_OnExecute(string MachineFormat, string NCCodeFileName, EspritConstants.espNCCodeType PostOption)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called execute NcCode {NCCodeFileName}");
}
private void NcCodeObject_OnCreateNCOutput(ref System.Array Posts, string NCCodeFileName, EspritConstants.espNCCodeType PostOption)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on creating NcCode output file {NCCodeFileName}");
}
}
Imports EspritConstants
Public Class NcCodeEventsHandler
Private ReadOnly _application As Esprit.Application
Private ReadOnly _ncCodeObject As Esprit.NCCode
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As Esprit.Application)
_application = app
_messageBoxTitle = "NcCodeEventsTutorial"
_ncCodeObject = _application.Document.NCCode
AddHandler _ncCodeObject.OnCreateNCOutput, AddressOf Me.NcCodeObject_OnCreateNCOutput
AddHandler _ncCodeObject.OnExecute, AddressOf Me.NcCodeObject_OnExecute
End Sub
Private Sub NcCodeObject_OnCreateNCOutput(ByRef Posts As Array, NCCodeFileName As String, PostOption As espNCCodeType)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on creating NcCode output file {NCCodeFileName}")
End Sub
Private Sub NcCodeObject_OnExecute(MachineFormat As String, NCCodeFileName As String, PostOption As espNCCodeType)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called execute NcCode {NCCodeFileName}")
End Sub
End Class
Dim WithEvents NcCodeObject As Esprit.NcCode
Attribute NcCodeObject.VB_VarHelpID = -1
Dim MessageBoxTitle As String
Private Sub Class_Initialize()
Set NcCodeObject = Application.Document.NcCode
MessageBoxTitle = "NcCodeEventsTutorial"
End Sub
Private Sub NcCodeObject_OnCreateNCOutput(ByRef Posts() As String, ByVal NCCodeFileName As String, ByVal PostOption As espNCCodeType)
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, "Called on creating NcCode output file " & NCCodeFileName)
End Sub
Private Sub NcCodeObject_OnExecute(ByVal MachineFormat As String, ByVal NCCodeFileName As String, ByVal PostOption As espNCCodeType)
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, "Called execute NcCode " & NCCodeFileName)
End Sub
