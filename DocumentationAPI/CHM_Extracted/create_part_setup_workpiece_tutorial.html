Create Part Setup Workpiece

This tutorial demonstrates how to add a workpiece into Part setup. Workpiece contains at least one Part and any Stocks. The following code creates test workpiece with 2 parts (first is default) and 3 different types of stocks:

private void CreateStocks(Esprit.Workpiece wp)
{
Esprit.IStock stock;
stock = AddSolidStock();
if (stock == null)
{
MessageBox.Show("We do not create a solid stock because your document does not have any solids", "Create Part Setup Workpiece Tutorial");
}
else
{
MessageBox.Show($"Stock: {stock.Name}; StockType: {stock.StockType}", "Create Part Setup Workpiece Tutorial");
}
stock = AddBarStock();
MessageBox.Show($"Stock: {stock.Name}; StockType: {stock.StockType}", "Create Part Setup Workpiece Tutorial");
stock = AddBlockStock();
MessageBox.Show($"Stock: {stock.Name}; StockType: {stock.StockType}", "Create Part Setup Workpiece Tutorial");
}
private void CreateWorkpiece()
{
Document.PartSetup.BeginEdit();
var wp = GetTestWorkpiece(true);
CreatePart(wp);
CreateStocks(wp);
Document.PartSetup.EndEdit();
}
private Esprit.IStock AddBarStock()
{
//Document.PartSetup.BeginEdit
var wp = GetTestWorkpiece(false);
var bs = wp.Stocks.AddBarStock();
bs.Tolerance = 0.1;
bs.OutsideDiameter = 50;
var alignment = new EspritConstants.STOCKALIGNMENT();
alignment.AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered;
alignment.Size = 200;
bs.set_LengthAlignment(alignment);
//Document.PartSetup.EndEdit
return bs;
}
private Esprit.IStock AddBlockStock()
{
//Document.PartSetup.BeginEdit
var wp = GetTestWorkpiece(false);
var bs = wp.Stocks.AddBlockStock();
var alignment = new EspritConstants.STOCKALIGNMENT();
var alignment1 = new EspritConstants.STOCKALIGNMENT();
var alignment2 = new EspritConstants.STOCKALIGNMENT();
alignment.AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered;
alignment.Size = 200;
alignment1.AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered;
alignment1.Size = 100;
alignment2.AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered;
alignment2.Size = 300;
bs.set_LengthAlignment(alignment);
bs.set_WidthAlignment(alignment1);
bs.set_HeightAlignment(alignment2);
//Document.PartSetup.EndEdit
return bs;
}
private Esprit.IStock AddSolidStock()
{
if (Document.Solids.Count == 0)
{
return null;
}
Esprit.Solid solid = Document.Solids[1];
//Document.PartSetup.BeginEdit
var wp = GetTestWorkpiece(false);
var ss = wp.Stocks.AddSolidStock();
ss.Name = "Test Solid";
ss.Tolerance = 0.1;
ss.Solid = solid.SolidBody;
//Document.PartSetup.EndEdit
return ss;
}
private void CreatePart(Esprit.Workpiece wp)
{
//Document.PartSetup.BeginEdit
var part = wp.Parts.Add();
part.Name = "Test part";
MessageBox.Show($"Part Name: {part.Name}", "Create Part Setup Workpiece Tutorial");
foreach(Esprit.Solid solid in Document.Solids)
{
part.Geometry.Add(solid);
MessageBox.Show($"{part.Name} add geometry: {solid.TypeName} {solid.Key}", "Create Part Setup Workpiece Tutorial");
}
//Document.PartSetup.EndEdit
}

Sub CreateStocks(wp As Workpiece)
Dim wpStock As IStock
Set wpStock = AddSolidStock
If wpStock Is Nothing Then
Call MsgBox("We do not create a solid stock because your document does not have any solids", vbOKOnly, "Create Part Setup Workpiece Tutorial")
Else
Call MsgBox("Stock: " & wpStock.Name & "; StockType: " & wpStock.StockType, vbOKOnly, "Create Part Setup Workpiece Tutorial")
End If
Set wpStock = AddBarStock
Call MsgBox("Stock: " & wpStock.Name & "; StockType: " & wpStock.StockType, vbOKOnly, "Create Part Setup Workpiece Tutorial")
Set wpStock = AddBlockStock
Call MsgBox("Stock: " & wpStock.Name & "; StockType: " & wpStock.StockType, vbOKOnly, "Create Part Setup Workpiece Tutorial")
End Sub

Function AddSolidStock() As IStock
Dim solidObj As solid
If Document.Solids.Count > 0 Then
Set solidObj = Document.Solids.Item(1)
Else
Set solidObj = Nothing
End If
If Not solidObj Is Nothing Then
'Document.PartSetup.BeginEdit
Dim wp As Workpiece
Set wp = GetTestWorkpiece(False)
Dim ss As SolidStock
Set ss = wp.Stocks.AddSolidStock
ss.Name = "Test Solid"
ss.Tolerance = 0.1
ss.solid = solidObj.SolidBody
'Document.PartSetup.EndEdit
Set AddSolidStock = ss
Else
Set AddSolidStock = Nothing
End If
'Document.PartSetup.EndEdit
End Function
Sub CreatePart(wp As Workpiece)
'Document.PartSetup.BeginEdit
Dim wpPart As part
Set wpPart = wp.Parts.Add
wpPart.Name = "Test part"
Call MsgBox("Part Name: " & wpPart.Name, vbOKOnly, "Create Part Setup Workpiece Tutorial")
Dim solidObj As solid
For Each solidObj In Document.Solids
Call wpPart.Geometry.Add(solidObj)
Call MsgBox(wpPart.Name & " add geometry: " & solidObj.TypeName & " " & solidObj.Key, vbOKOnly, "Create Part Setup Workpiece Tutorial")
Next
'Document.PartSetup.EndEdit
End Sub

private void ClearWorkpiece(Esprit.Workpiece wp)
{
//Document.PartSetup.BeginEdit
for (var i = wp.Stocks.Count; i > 0; i--)
{
wp.Stocks.Remove(i);
}
for (var i = wp.Parts.Count; i > 1; i--)
{
//Workpiece must contain 1 part as minimum
wp.Parts.Remove(i);
}
//Document.PartSetup.EndEdit
}
private Esprit.Workpiece GetTestWorkpiece(bool isEmpty)
{
Esprit.Workpiece wp = null;
foreach (Esprit.Workpiece item in Document.Workpieces)
{
if (item.Name == "Test workpiece")
{
if (isEmpty)
{
ClearWorkpiece(item);
}
wp = item;
break;
}
}
//Document.PartSetup.BeginEdit
if (wp == null)
{
wp = Document.Workpieces.Add();
wp.Name = "Test workpiece";
}
//Document.PartSetup.EndEdit
return wp;
}
Private Sub ClearWorkpiece(wp As Esprit.Workpiece)
' Document.PartSetup.BeginEdit
For i As Integer = wp.Stocks.Count To 1 Step -1
wp.Stocks.Remove(i)
Next
For i As Integer = wp.Parts.Count To 2 Step -1
' Workpiece must contain 1 part as minimum
wp.Parts.Remove(i)
Next
' Document.PartSetup.EndEdit
End Sub
Private Function GetTestWorkpiece(isEmpty As Boolean) As Esprit.Workpiece
Dim wp As Esprit.Workpiece = Nothing
For Each item As Esprit.Workpiece In Document.Workpieces
If item.Name = "Test workpiece" Then
If isEmpty Then
ClearWorkpiece(item)
End If
wp = item
Exit For
End If
Next
' Document.PartSetup.BeginEdit
If wp Is Nothing Then
wp = Document.Workpieces.Add()
wp.Name = "Test workpiece"
End If
' Document.PartSetup.EndEdit
Return wp
End Function
Sub ClearWorkpiece(wp As Workpiece)
'Document.PartSetup.BeginEdit
If wp.Stocks.Count > 0 Then
Dim i As Integer
For i = wp.Stocks.Count To 1 Step -1
wp.Stocks.Remove (i)
Next
End If
If wp.Parts.Count > 0 Then
'Workpiece must contain 1 part as minimum
For i = wp.Parts.Count To 2 Step -1
wp.Parts.Remove (i)
Next
End If
'Document.PartSetup.EndEdit
End Sub
Function GetTestWorkpiece(isEmpty As Boolean) As Workpiece
Set GetTestWorkpiece = Nothing
Dim wp As Workpiece
For Each wp In Document.Workpieces
If wp.Name = "Test workpiece" Then
If isEmpty = True Then
Call ClearWorkpiece(wp)
End If
Set GetTestWorkpiece = wp
Exit For
End If
Next
'Document.PartSetup.BeginEdit
If GetTestWorkpiece Is Nothing Then
Set wp = Document.Workpieces.Add
wp.Name = "Test workpiece"
Set GetTestWorkpiece = wp
End If
'Document.PartSetup.EndEdit
End Function
