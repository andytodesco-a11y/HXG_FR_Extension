Application Commands

An advantage of using classes is that they can also contain event procedures. Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

For example, for the Application object, some of the events are AfterDocumentClose, AfterDocumentOpen and BeforeQuit.

The Tutorial demostrates Application object events which deal with the application commands behavior.

Define a class and add the event procedures:

class CommandsEventsHandler
{
private readonly Esprit.Application _application;
private readonly string _messageBoxTitle;
public CommandsEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "ApplicationCommandsTutorial";
_application.AfterCommandRun += AfterCommandRun;
_application.AfterCommandAbort += AfterCommandAbort;
_application.BeforeCommandRun += BeforeCommandRun;
_application.OnSubCommandSet += OnSubCommandSet;
_application.RunCommand(EspritConstants.espCommands.espCmdSnappoint);
}
private void OnSubCommandSet(int commandId, int subCommandId, int promptId)
{
if (commandId == (int)EspritConstants.espCommands.espCmdSnappoint)
{
MessageBox.Show("OnSubCommandSet called on passing a command state", _messageBoxTitle, MessageBoxButton.OK);
}
}
private void BeforeCommandRun(int commandId, int auxId, ref bool overrided)
{
if (commandId == (int)EspritConstants.espCommands.espCmdSnappoint)
{
MessageBox.Show("BeforeCommandRun called before running the command", _messageBoxTitle, MessageBoxButton.OK);
}
}
private void AfterCommandAbort(int commandId, int auxId)
{
if (commandId == (int)EspritConstants.espCommands.espCmdSnappoint)
{
MessageBox.Show("AfterCommandAbort called aborting the command", _messageBoxTitle, MessageBoxButton.OK);
}
}
private void AfterCommandRun(int commandId, int nAuxId)
{
if (commandId == (int)EspritConstants.espCommands.espCmdSnappoint)
{
MessageBox.Show("AfterCommandRun called after finishing the command", _messageBoxTitle, MessageBoxButton.OK);
}
}
}

Public Class CommandsEventsHandler
Private ReadOnly _application As ESPRIT.Application
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As ESPRIT.Application)
_application = app
_messageBoxTitle = "ApplicationCommandsTutorial"
AddHandler app.AfterCommandRun, AddressOf Me.AfterCommandRun
AddHandler app.AfterCommandAbort, AddressOf Me.AfterCommandAbort
AddHandler app.BeforeCommandRun, AddressOf Me.BeforeCommandRun
AddHandler app.OnSubCommandSet, AddressOf Me.OnSubCommandSet
_application.RunCommand(EspritConstants.espCommands.espCmdSnappoint)
End Sub
