Modify Note Text

This tutorial shows how to modify the text of existing Notes and to modify the Notes within a Dimension object. We will make use of the Count and Item properties.

VB.NET

public void ModifyNoteOrDimensionText()
{
var mask = new EspritConstants.espGraphicObjectType[2]
{
EspritConstants.espGraphicObjectType.espAnnotation,
EspritConstants.espGraphicObjectType.espDimension
};
Esprit.Annotation annotation = null;
try
{
annotation = Document.GetAnyElement("Pick an Annotation", mask) as Esprit.Annotation;
}
catch (System.Runtime.InteropServices.COMException)
{
// Element was not set
return;
}
if (annotation == null)
{
return;
}
if (annotation.AnnotationType == EspritConstants.espAnnotationType.espAnnotationNote)
{
var notes = annotation as Esprit.Notes;
for (var i = 1; i <= notes.Count; i++)
{
var tempString = notes.Item[i].String;
if (RequestUserInput($"Enter New Note (Line {i})", "Modify Note", ref tempString))
{
notes.Item[i].String = tempString;
}
}
}
else if (annotation.AnnotationType == EspritConstants.espAnnotationType.espAnnotationDimension)
{
var dimension = annotation as Esprit.Dimension;
var notes = dimension.Notes;
for (var i = 1; i <= notes.Count; i++)
{
var tempString = notes.Item[i].String;
if (RequestUserInput($"Enter New Dimension (Part {i} of {notes.Count})", "Modify Note", ref tempString))
{
notes.Item[i].String = tempString;
}
}
}
Document.Refresh();
}
Public Sub ModifyNoteOrDimensionText()
Dim mask() As EspritConstants.espGraphicObjectType = {
EspritConstants.espGraphicObjectType.espAnnotation,
EspritConstants.espGraphicObjectType.espDimension
}
Dim annotation As Esprit.Annotation = Nothing
Try
annotation = Document.GetAnyElement("Pick an Annotation", mask)
Catch ex As System.Runtime.InteropServices.COMException
' Element was Not set
Return
End Try
If annotation Is Nothing Then
Return
End If
If annotation.AnnotationType = EspritConstants.espAnnotationType.espAnnotationNote Then
Dim notes As Esprit.Notes = annotation
For i As Integer = 1 To notes.Count
Dim tempString As String = notes.Item(i).String
If RequestUserInput($"Enter New Note (Line {i})", "Modify Note", tempString) Then
notes.Item(i).String = tempString
End If
Next
ElseIf annotation.AnnotationType = EspritConstants.espAnnotationType.espAnnotationDimension Then
Dim dimension As Esprit.Dimension = annotation
Dim notes As Esprit.Notes = dimension.Notes
For i As Integer = 1 To notes.Count
Dim tempString As String = notes.Item(i).String
If RequestUserInput($"Enter New Dimension (Part {i} of {notes.Count})", "Modify Note", tempString) Then
notes.Item(i).String = tempString
End If
Next
End If
Document.Refresh()
End Sub
Sub ModifyNoteOrDimensionText()
Dim An As Esprit.Annotation
' allow the user to pick either a text note or dimension
Dim TypeMask(1 To 2) As EspritConstants.espGraphicObjectType
TypeMask(1) = espAnnotation
TypeMask(2) = espDimension
On Error Resume Next
Set An = Document.GetAnyElement("Pick an Annotation", TypeMask)
If An Is Nothing Then
Return
End If
Dim I As Long
If An.AnnotationType = espAnnotationNote Then
Dim Notes1 As Esprit.Notes
Set Notes1 = An
' loop through each Note line within the Notes
For I = 1 To Notes1.Count
Notes1.Item(I).String = InputBox("Enter New Note" & " (" & "Line" & " " & I & ")", "Modify Note", Notes1.Item(I).String)
Next
ElseIf An.AnnotationType = espAnnotationDimension Then
Dim Dimension As Esprit.Dimension
Dim Notes2 As Esprit.Notes
Set Dimension = An
Set Notes2 = D.Notes
' loop through each Note part within the Notes of the dimension
For I = 1 To Notes2.Count
Notes2.Item(I).String = InputBox("Enter New Dimension" & " (" & "Part" & " " & I & " " & "of" & " " & Notes2.Count & ")", "Modify Note", Notes2.Item(I).String)
Next
End If
Document.Refresh
End Sub

Before running the code, open a file that contains annotations or create some new ones of your own. Select a note in response to the prompt. If you select Notes with multiple lines, notice how you loop through each line one at a time.
