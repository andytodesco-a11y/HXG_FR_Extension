Simple Square

In this tutorial you will create four segments to form a square. The important concept introduced in this tutorial is the use of the Add method to create and add objects to a collection - we are adding Segment objects to the Segments collection. Also this tutorial introduces the use of the "Get" methods - GetPoint is used to create temporary points. These temporary points are not shown in the graphics and are not added to the Points collection. Note the use of an array variable to hold the temporary points.

public void SimpleSquare()
{
var length = 10.0;
if (RequestUserInput("Enter Length", "Square Length", ref length))
{
var points = new List<Esprit.Point>();
points.Add(Document.GetPoint(0, 0, 0)); //Bottom left
points.Add(Document.GetPoint(0, length, 0)); //Top left
points.Add(Document.GetPoint(length, length, 0)); //Top right
points.Add(Document.GetPoint(length, 0, 0)); //Bottom right
Document.Segments.Add(points[0], points[1]);
Document.Segments.Add(points[1], points[2]);
Document.Segments.Add(points[2], points[3]);
Document.Segments.Add(points[3], points[0]);
Document.Refresh();
}
}
Public Sub SimpleSquare()
Dim length As Double = 10.0
If RequestUserInput("Enter Length", "Square Length", length) Then
Dim points As New List(Of Esprit.Point)
points.Add(Document.Points.Add(0, 0, 0)) ' Bottom left
points.Add(Document.Points.Add(0, length, 0)) ' Top left
points.Add(Document.Points.Add(length, length, 0)) ' Top Right
points.Add(Document.Points.Add(length, 0, 0)) ' Bottom right
Document.Segments.Add(points.Item(0), points.Item(1))
Document.Segments.Add(points.Item(1), points.Item(2))
Document.Segments.Add(points.Item(2), points.Item(3))
Document.Segments.Add(points.Item(3), points.Item(0))
Document.Refresh()
End If
End Sub
Sub SimpleSquare()
Dim Length As Double
Length = InputBox("Enter Length", "Square Length", 10)
' declare P as an arrary to keep track of all four points
Dim P(1 To 4) As Esprit.Point
Set P(1) = Document.GetPoint(0, 0, 0)
Set P(2) = Document.GetPoint(0, Length, 0)
Set P(3) = Document.GetPoint(Length, Length, 0)
Set P(4) = Document.GetPoint(Length, 0, 0)
Dim S As Esprit.Segment
Set S = Document.Segments.Add(P(1), P(2))
Set S = Document.Segments.Add(P(2), P(3))
Set S = Document.Segments.Add(P(3), P(4))
Set S = Document.Segments.Add(P(4), P(1))
Document.Refresh
End Sub

Do a Fit in ESPRIT EDGE after running the code and you will see the newly created segments in the graphics. Note, however, that no points are actually created at the corners. If necessary turn off your XYZ Axis under the View menu to see the segments that are on top of the X and Y axes.
