More SelectionSet Manipulations: Grid Translation

Suppose that you often translate a group in one direction, and then translate the original and all of the copies to end up with a grid pattern. Instead of doing this in two steps, you want to do it in one. What we will need to do is transfer the group into a selection set, then use the Translate method. In addition, let us suppose that the user often wants to edit some properties of only the copies, not the original.

In the Tutorial: Introduction to SelectionSet Manipulations: XY Symmetry, the copies were included in the selection set, and hence in the group, with the originals. To just end up with only the copies in the group, we simply need to remove the originals from the group. We can easily do this by noting the Count of the selection set before and after the manipulation.

public void GridTranslation()
{
var groupName = "Temp";
var set = SelectionSetHelper.GetSelectionSet(Document, groupName);
set.RemoveAll();
var originalCount = Document.Group.Count;
for (var i = 1; i <= originalCount; i++)
{
set.Add(Document.Group[i]);
}
set.AddCopiesToSelectionSet = true;
var firstGridNumber = 10;
var secondGridNumber = 10;
var firstDistance = 1.0;
var secondDistance = 1.0;
var firstAngle = 0.0;
var secondAngle = 90.0;
if (RequestUserInput("Enter First Number of Grid Items", "First Number", ref firstGridNumber) &&
RequestUserInput("Enter Second Number of Grid Items", "Second Number", ref secondGridNumber) &&
RequestUserInput("Enter First Distance", "First Distance", ref firstDistance) &&
RequestUserInput("Enter Second Distance", "Second Distance", ref secondDistance) &&
RequestUserInput("Enter First Angle", "First Angle", ref firstAngle) &&
RequestUserInput("Enter Second Angle", "Second Angle", ref secondAngle))
{
firstAngle *= Math.PI / 180;
secondAngle *= Math.PI / 180;
set.Translate(firstDistance * Math.Cos(firstAngle), firstDistance * Math.Sin(firstAngle), 0, firstGridNumber - 1);
set.Translate(secondDistance * Math.Cos(secondAngle), secondDistance * Math.Sin(secondAngle), 0, secondGridNumber - 1);
}
Document.Refresh();
for (var i = 1; i <= originalCount; i++)
{
var graphicObject = set[i] as Esprit.IGraphicObject;
if (graphicObject != null)
graphicObject.Grouped = false;
}
Document.SelectionSets.Remove(groupName);
}
Public Sub GridTranslation()
Dim selectionSetName As String = "Temp"
Dim selectionSet As Esprit.SelectionSet = Document.SelectionSets.Item(selectionSetName)
If selectionSet Is Nothing Then
selectionSet = Document.SelectionSets.Add(selectionSetName)
End If
' transfer group to SelectionSet
selectionSet.RemoveAll()
Dim originalCount As Long = Document.Group.Count
For i As Integer = 1 To Document.Group.Count
Call selectionSet.Add(Document.Group.Item(i))
Next
Dim firstGridNumber As Integer = 10
Dim secondGridNumber As Integer = 10
Dim firstDistance As Double = 1.0
Dim secondDistance As Double = 1.0
Dim firstAngle As Double = 0.0
Dim secondAngle As Double = 90.0
' prompt for translation information and perform the manipulations
selectionSet.AddCopiesToSelectionSet = True
If RequestUserInput("Enter First Number of Grid Items", "First Number", firstGridNumber) AndAlso
RequestUserInput("Enter Second Number of Grid Item", "Second Number", secondGridNumber) AndAlso
RequestUserInput("Enter First Distance", "First Distance", firstDistance) AndAlso
RequestUserInput("Enter Second Distance", "Second Distance", secondDistance) AndAlso
RequestUserInput("Enter First Angle", "First Angle", firstAngle) AndAlso
RequestUserInput("Enter Second Angle", "Second Angle", secondAngle) Then
firstAngle *= Math.PI / 180
secondAngle *= Math.PI / 180
selectionSet.Translate(firstDistance * Math.Cos(firstAngle), firstDistance * Math.Sin(firstAngle), 0, firstGridNumber - 1)
selectionSet.Translate(secondDistance * Math.Cos(secondAngle), secondDistance * Math.Sin(secondAngle), 0, secondGridNumber - 1)
' now ungroup the originals
For i As Integer = 1 To originalCount
If Not selectionSet.Item(i) Is Nothing Then
selectionSet.Item(i).Grouped = False
End If
Next
Document.Refresh()
Call Document.SelectionSets.Remove(selectionSetName)
End If
End Sub
Sub GridTranslation()
Dim GridNumber(1 To 2) As Long
Dim Distance(1 To 2) As Double
Dim Angle(1 To 2) As Double
Dim SS As Esprit.SelectionSet
' check for existence of Temp SelectionSet, create it if necessary
On Error Resume Next
Dim SelectionSetName As String
SelectionSetName = "Temp"
Set SS = Document.SelectionSets.Item(SelectionSetName)
If SS Is Nothing Then
Set SS = Document.SelectionSets.Add(SelectionSetName)
End If
' transfer group to SelectionSet
SS.RemoveAll
Dim I As Long, OriginalCount As Long
OriginalCount = Document.Group.Count
For I = 1 To Document.Group.Count
Call SS.Add(Document.Group.Item(I))
Next
' prompt for translation information and perform the manipulations
SS.AddCopiesToSelectionSet = True
GridNumber(1) = Val(InputBox("Enter First Number of Grid Items", "First Number", 10))
GridNumber(2) = Val(InputBox("Enter Second Number of Grid Item", "Second Number", 10))
Distance(1) = Val(InputBox("Enter First Distance", "First Distance", 1))
Distance(2) = Val(InputBox("Enter Second Distance", "Second Distance", 1))
' immediately convert angle to radians
Angle(1) = Val(InputBox("Enter First Angle", "First Angle", 0)) * 4 * Atn(1) / 180
Angle(2) = Val(InputBox("Enter Second Angle", "Second Angle", 90)) * 4 * Atn(1) / 180
Call SS.Translate(Distance(1) * Cos(Angle(1)), Distance(1) * Sin(Angle(1)), 0, GridNumber(1) - 1)
Call SS.Translate(Distance(2) * Cos(Angle(2)), Distance(2) * Sin(Angle(2)), 0, GridNumber(2) - 1)
' now ungroup the originals
For I = 1 To OriginalCount
SS.Item(I).Grouped = False
Next
Document.Refresh
' clean up by removing Temp SelectionSet
Call Document.SelectionSets.Remove(SelectionSetName)
End Sub
