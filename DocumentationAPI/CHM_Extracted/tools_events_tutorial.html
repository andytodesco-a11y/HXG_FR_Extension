Tools Events

Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

The Tutorial demostrates the Tools object events.

Define a class and add the event procedures:

C#
VB.NET
VBA
class ToolsEventsHandler
{
Esprit.EventWindow _eventWindow;
private readonly string _messageBoxTitle;
private readonly EspritTools.Tools _tools;
public ToolsEventsHandler(Esprit.Application app)
{
_eventWindow = app.EventWindow;
_tools = app.Document.Tools;
_messageBoxTitle = "ToolsEventsTutorial";
_tools.OnCreate += _tools_OnCreate;
_tools.OnDelete += _tools_OnDelete;
_tools.OnModify += _tools_OnModify;
_tools.OnReorder += _tools_OnReorder;
//Same order just for demo
var reorderList = new System.Collections.Generic.List<EspritTechnology.ITechnology>();
for (var i = 1; i <= _tools.Count; i++)
{
reorderList.Add(_tools[i]);
}
_tools.Reorder(reorderList.ToArray());
}
private void _tools_OnReorder()
{
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Tools_OnReorder called reordering via API (Tools.Reorder)");
}
private void _tools_OnModify(string OldName, int CLCode, object Tool)
{
var tool = Tool as EspritTechnology.ITechnology;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"ClCode {CLCode} of {tool.Caption} tool modified");
}
private void _tools_OnDelete(object Tool)
{
var tool = Tool as EspritTechnology.ITechnology;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"{tool.Caption} tool deleted");
}
private void _tools_OnCreate(object Tool)
{
var tool = Tool as EspritTechnology.ITechnology;
_eventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"{tool.Caption} tool created");
}
}
Public Class ToolsEventsHandler
Private ReadOnly _application As ESPRIT.Application
Private ReadOnly _tools As EspritTools.Tools
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As ESPRIT.Application)
_application = app
_messageBoxTitle = "ToolsEventsTutorial"
_tools = app.Document.Tools
AddHandler _tools.OnCreate, AddressOf OnCreate
AddHandler _tools.OnDelete, AddressOf OnDelete
AddHandler _tools.OnModify, AddressOf OnModify
AddHandler _tools.OnReorder, AddressOf OnReorder
'Same order just for demo
Dim reorderList = New System.Collections.Generic.List(Of EspritTechnology.ITechnology)
For i As Integer = 1 To _tools.Count
reorderList.Add(_tools.Item(i))
Next
_tools.Reorder(reorderList.ToArray())
End Sub
Private Sub OnCreate(ToolObj As Object)
Dim tool As EspritTechnology.ITechnology = ToolObj
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"{tool.Caption} tool created")
End Sub
Private Sub OnDelete(ToolObj As Object)
Dim tool As EspritTechnology.ITechnology = ToolObj
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"{tool.Caption} tool deleted")
End Sub
Private Sub OnModify(OldName As String, CLCode As Integer, ToolObj As Object)
Dim tool As EspritTechnology.ITechnology = ToolObj
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"ClCode {CLCode} of {tool.Caption} tool modified")
End Sub
Private Sub OnReorder()
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Tools_OnReorder called reordering via API (Tools.Reorder)")
End Sub
End Class
Dim WithEvents MyTools As EspritTools.Tools
Attribute MyTools.VB_VarHelpID = -1
Dim MessageBoxTitle As String
Private Sub Class_Initialize()
Set MyTools = Document.Tools
MessageBoxTitle = "ToolsEventsTutorial"
'Same order just for demo
Dim reorderList() as Object 'EspritTechnology.ITechnology
Dim i As Integer
For i = 1 To MyTools.Count
ReDim Preserve reorderList(1 To i) as Object 'EspritTechnology.ITechnology
Set reorderList(i) = MyTools.Item(i)
Next
Call MyTools.Reorder(reorderList)
End Sub
Private Sub MyTools_OnCreate(ByVal ToolObj As Object)
Dim tool As EspritTechnology.ITechnology
Set tool = ToolObj
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, tool.Caption & " tool created")
End Sub
Private Sub MyTools_OnDelete(ByVal ToolObj As Object)
Dim tool As EspritTechnology.ITechnology
Set tool = ToolObj
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, tool.Caption & " tool deleted")
End Sub
Private Sub MyTools_OnModify(ByVal OldName As String, ByVal CLCode As Long, ByVal ToolObj As Object)
Dim tool As EspritTechnology.ITechnology
Set tool = ToolObj
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, "ClCode " & CLCode & " of " & tool.Caption & " tool modified")
End Sub
Private Sub MyTools_OnReorder()
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, MessageBoxTitle, "Tools_OnReorder called reordering via API (Tools.Reorder)")
End Sub

Before running the tutorial, create new or open a document that has several tools on it. Than play Tool Assemblies pane, information about occurence of each ToolsEvent is logged in EventWindow.
