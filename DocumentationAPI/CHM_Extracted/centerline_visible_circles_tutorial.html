Center Line Visible Circles

This tutorial will loop through all of the circles in the Document and create segments representing a centerline dimensioning mark mark, similar to what was done in Tutorial: Center Line Circles. Only in this case it will apply to only the visible circles.

In Tutorial: Swap Layers we checked the Visible property of a Layer using the LayerObject.Visible syntax. The Layer itself is a property of all of the graphical objects, such as Circle objects. You can access the Layer that a Circle is on with syntax such as CircleObject.Layer. Whether or not a graphical object is on a visible Layer can be directly determined by using the Object.Layer.Visible property. This is demonstrated below for a Circle.

public void CenterLineVisibleCircles()
{
var segments = Document.Segments;
Document.ActiveColor = ColorHelper.ColorToUInt(System.Drawing.Color.Green);
var clRatio = new double[3]
{
1.15, 0.2, 0.4
};
foreach (Esprit.Circle circul in Document.Circles)
{
if (circul.Layer.Visible)
{
segments.Add(Document.GetPoint(circul.X - clRatio[0] * circul.Radius, circul.Y, circul.Z), Document.GetPoint(circul.X - clRatio[2] * circul.Radius, circul.Y, circul.Z));
segments.Add(Document.GetPoint(circul.X - clRatio[1] * circul.Radius, circul.Y, circul.Z), Document.GetPoint(circul.X + clRatio[1] * circul.Radius, circul.Y, circul.Z));
segments.Add(Document.GetPoint(circul.X + clRatio[2] * circul.Radius, circul.Y, circul.Z), Document.GetPoint(circul.X + clRatio[0] * circul.Radius, circul.Y, circul.Z));
segments.Add(Document.GetPoint(circul.X, circul.Y - clRatio[0] * circul.Radius, circul.Z), Document.GetPoint(circul.X, circul.Y - clRatio[2] * circul.Radius, circul.Z));
segments.Add(Document.GetPoint(circul.X, circul.Y - clRatio[1] * circul.Radius, circul.Z), Document.GetPoint(circul.X, circul.Y + clRatio[1] * circul.Radius, circul.Z));
segments.Add(Document.GetPoint(circul.X, circul.Y + clRatio[2] * circul.Radius, circul.Z), Document.GetPoint(circul.X, circul.Y + clRatio[0] * circul.Radius, circul.Z));
}
}
WindowHelper.FitAllWindows(Document.Windows);
}

public static uint ColorToUInt(System.Drawing.Color color)
{

(color.G << 8) | color.R);
return value;
}

The following function is used to fit all windows:
public static void FitAllWindows(Esprit.Windows windows)
{
foreach (Esprit.Window window in windows)
{
window.Fit();
window.Refresh();
}
}

Public Sub CenterLineVisibleCircles()
Document.ActiveColor = ColorToUInt(System.Drawing.Color.Green)
' set up the center line proportions
Dim cLRatio(0 To 2) As Double
cLRatio(0) = 1.15
cLRatio(1) = 0.2
cLRatio(2) = 0.4
' loop through all the circles, notice how the For Each syntax is used
For Each circle As Esprit.Circle In Document.Circles
If circle.Layer.Visible Then
' create segments to represent the center lines
With Document.Segments
.Add(Document.GetPoint(circle.X - cLRatio(0) * circle.Radius, circle.Y, circle.Z), Document.GetPoint(circle.X - cLRatio(2) * circle.Radius, circle.Y, circle.Z))
.Add(Document.GetPoint(circle.X - cLRatio(1) * circle.Radius, circle.Y, circle.Z), Document.GetPoint(circle.X + cLRatio(1) * circle.Radius, circle.Y, circle.Z))
.Add(Document.GetPoint(circle.X + cLRatio(2) * circle.Radius, circle.Y, circle.Z), Document.GetPoint(circle.X + cLRatio(0) * circle.Radius, circle.Y, circle.Z))
.Add(Document.GetPoint(circle.X, circle.Y - cLRatio(0) * circle.Radius, circle.Z), Document.GetPoint(circle.X, circle.Y - cLRatio(2) * circle.Radius, circle.Z))
.Add(Document.GetPoint(circle.X, circle.Y - cLRatio(1) * circle.Radius, circle.Z), Document.GetPoint(circle.X, circle.Y + cLRatio(1) * circle.Radius, circle.Z))
.Add(Document.GetPoint(circle.X, circle.Y + cLRatio(2) * circle.Radius, circle.Z), Document.GetPoint(circle.X, circle.Y + cLRatio(0) * circle.Radius, circle.Z))
End With
End If
Next
Call FitAllWindows(Document.Windows)
End Sub
The following function is used to convert the color value:
Public Function ColorToUInt(color As System.Drawing.Color) As UInteger
Dim a As UInteger = color.A
Dim r As UInteger = color.R
Dim g As UInteger = color.G
Dim b As UInteger = color.B
Dim resuls As UInteger = ((a << 24) Or (b << 16) Or (g << 8) Or (r << 0))
Return resuls
End Function

Public Sub FitAllWindows(Windows As Esprit.Windows)

Sub CenterLineVisibleCircles()
Document.ActiveColor = vbGreen
' set up the center line proportions
Dim CLRatio(1 To 3) As Double
CLRatio(1) = 1.15
CLRatio(2) = 0.2
CLRatio(3) = 0.4
' loop through all the circles, notice how the For Each syntax is used
Dim C As Esprit.Circle
For Each C In Document.Circles
If C.Layer.Visible Then
' create segments to represent the center lines
Dim S As Esprit.Segment
With Document.Segments
Set S = .Add(Document.GetPoint(C.X - CLRatio(1) * C.Radius, C.Y, C.Z), Document.GetPoint(C.X - CLRatio(3) * C.Radius, C.Y, C.Z))
Set S = .Add(Document.GetPoint(C.X - CLRatio(2) * C.Radius, C.Y, C.Z), Document.GetPoint(C.X + CLRatio(2) * C.Radius, C.Y, C.Z))
Set S = .Add(Document.GetPoint(C.X + CLRatio(3) * C.Radius, C.Y, C.Z), Document.GetPoint(C.X + CLRatio(1) * C.Radius, C.Y, C.Z))
Set S = .Add(Document.GetPoint(C.X, C.Y - CLRatio(1) * C.Radius, C.Z), Document.GetPoint(C.X, C.Y - CLRatio(3) * C.Radius, C.Z))
Set S = .Add(Document.GetPoint(C.X, C.Y - CLRatio(2) * C.Radius, C.Z), Document.GetPoint(C.X, C.Y + CLRatio(2) * C.Radius, C.Z))
Set S = .Add(Document.GetPoint(C.X, C.Y + CLRatio(3) * C.Radius, C.Z), Document.GetPoint(C.X, C.Y + CLRatio(1) * C.Radius, C.Z))
End With
End If
Next
Call FitAllWindows
End Sub
The following function is used to fit all windows:
Sub FitAllWindows()
Dim W As Esprit.Window
For Each W In Document.Windows
W.Fit
' note that Fit just sizes the window to fit the graphics
' it does not redraw the graphics, so we still need a Refresh
W.Refresh
Next
End Sub

After running the code and you will see the newly created centerline dimension marks in the graphics for every visible circle in the document. Turn layers on and off to confirm this.

If you like, try modifying the CLRatio values to change the proportions of the segments with respect to their circles.
