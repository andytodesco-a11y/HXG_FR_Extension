Parts Events

Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

The Tutorial demostrates the Parts object events.

Define a class and add the event procedures:

VB.NET
VBA
class PartsEventsHandler
{
private readonly Esprit.Application _application;
private readonly Esprit.Parts _partsObject;
private readonly string _messageBoxTitle;
public PartsEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "PartsEventsTutorial";
_partsObject = _application.Document.Parts;
_partsObject.OnPartActive += PartsObject_OnPartActive;
_partsObject.OnPartCreate += PartsObject_OnPartCreate;
_partsObject.OnPartDelete += PartsObject_OnPartDelete;
_partsObject.OnPartModify += PartsObject_OnPartModify;
}
private void PartsObject_OnPartModify(object Part)
{
var part = Part as Esprit.Part;
//Sends on part name changed
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on part \"{part.Name}\" modified");
}
private void PartsObject_OnPartDelete(object Part)
{
//Part object mostly invalid at this moment. You can only get part's key.
var part = Part as Esprit.Part;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on part \"{part.Key}\" deleted");
}
private void PartsObject_OnPartCreate(object Part)
{
var part = Part as Esprit.Part;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on part \"{part.Name}\" created");
}
private void PartsObject_OnPartActive(object Part)
{
var part = Part as Esprit.Part;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, _messageBoxTitle, $"Called on part \"{part.Name}\" active");
}
}
Public Class PartsEventsHandler
Private ReadOnly _application As Esprit.Application
Private ReadOnly _partsObject As Esprit.Parts
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As Esprit.Application)
_application = app
_messageBoxTitle = "PartsEventsTutorial"
_partsObject = _application.Document.Parts
AddHandler _partsObject.OnPartActive, AddressOf Me.Parts_OnPartActive
AddHandler _partsObject.OnPartCreate, AddressOf Me.Parts_OnPartCreate
AddHandler _partsObject.OnPartDelete, AddressOf Me.Parts_OnPartDelete
AddHandler _partsObject.OnPartModify, AddressOf Me.Parts_OnPartModify
End Sub

End Sub

End Sub

Dim WithEvents PartsObject As Esprit.Parts
