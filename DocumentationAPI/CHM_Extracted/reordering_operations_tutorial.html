Reordering Operations

This tutorial shows how to add operations to a specific locations using the Reorder method from the PartOperations collection. This method requires an array of operations, a numerical index indicating the position in the list (the location in the F2 Project Manager, starting with 1 at the top), and a flag InsertBefore specifies whether or not to insert the array before (True) or after (False) the indicated Position.

For example, to insert an array of operations at the top, the position would be 1 and InsertBefore would be True. To move an array of operations to the end, the position would equal the number of operations on the list (the Count) and InsertBefore would be False.

private void MoveGroupedOperationsToTop()
{
var operations = new List<Esprit.PartOperation>();
for (var i = 1; i <= Document.Group.Count; i++)
{
var obj = Document.Group[i] as Esprit.IGraphicObject;
if (obj.GraphicObjectType == EspritConstants.espGraphicObjectType.espOperation)
{
var operation = obj as Esprit.PartOperation;
operations.Add(operation);
}
}
if (operations.Count > 0)
{
Document.PartOperations.Reorder(operations.ToArray(), 1, true);
}
}

{
var operations = new List<Esprit.PartOperation>();

Sub MoveGroupedOperationsToTop()
Dim operations As List(Of Esprit.PartOperation) = New List(Of Esprit.PartOperation)
' loop through the group
For i As Integer = 1 To Document.Group.Count
' see if it is an operation, if so add it to the array
If Document.Group.Item(i).GraphicObjectType = EspritConstants.espGraphicObjectType.espOperation Then
Dim operation As Esprit.PartOperation = Document.Group.Item(i)
operations.Add(operation)
End If
Next
' move the operations to the top (insert before position 1)
If operations.Count > 0 Then Call Document.PartOperations.Reorder(operations.ToArray, 1, True)
End Sub

Sub MoveGroupedOperationsToTop()
' set up an array to hold the operations to move
Dim Ops() As Esprit.PartOperation
Dim GroupedOpCount As Long
' loop through the group
Dim I As Long
For I = 1 To Document.Group.Count
' see if it is an operation, if so add it to the array
If Document.Group.Item(I).GraphicObjectType = espOperation Then
GroupedOpCount = GroupedOpCount + 1
ReDim Preserve Ops(1 To GroupedOpCount) As Esprit.PartOperation
Set Ops(GroupedOpCount) = Document.Group.Item(I)
End If
Next
' move the operations to the top (insert before position 1)
If GroupedOpCount > 0 Then Call Document.PartOperations.ReOrder(Ops, 1, True)
DoEvents
End Sub
