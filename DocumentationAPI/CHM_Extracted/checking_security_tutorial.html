Checking Security

LicenseType property and OptionIsValid method of the Application. These give a developer a way to check security before running portions of a macro. We need to give it a Long for the OptionNumber, and it returns a Boolean. What numbers do we give it for each option? Well, the license type values started at zero, so let us start with 0 for the option numbers as well. Available security options you can find at SecurityManager application.

private string LicenseTypeName(int type)
{
switch(type)
{
case 0:
return "Normal";
case 1:
return "Demonstration";
case 2:
return "Dealer";
case 3:
return "Education";
case 4:
return "Student";
}
return "Error";
}
public void DisplayLicenseOptions()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CheckingSecurityTutorial", $"License type: {LicenseTypeName(_application.LicenseType)}");
var message = _application.OptionIsValid(715)
? "STL Import/Export option is included in this license."
: "STL Import/Export option is not included in this license.";
MessageBox.Show(message, "CheckingSecurityTutorial", MessageBoxButton.OK);
for (var i = 0; i < 50; i++)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CheckingSecurityTutorial", $"Option {i} is {_application.OptionIsValid(i)}");
}
}
Private Function LicenseTypeName(type As Integer) As String

Case 1
Return "Demonstration"

Return "Dealer"

Return "Education"

End Function

Function LicenseTypeName(LType As Long) As String
Select Case LType
Case 0
LicenseTypeName = "Normal"
Case 1
LicenseTypeName = "Demonstration"
Case 2
LicenseTypeName = "Dealer"
Case 3
LicenseTypeName = "Education"
Case 4

Sub DisplayLicenseOptions()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CheckingSecurityTutorial", "License type: " & LicenseTypeName(Application.LicenseType))
If Application.OptionIsValid(715) = True Then
Call MsgBox("STL Import/Export option is included in this license.", vbOKOnly, "STL License")
Else
Call MsgBox("STL Import/Export option is not included in this license.", vbOKOnly, "STL License")
End If
Dim L As Long
For L = 0 To 50
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "CheckingSecurityTutorial", "Option " & L & " is " & Application.OptionIsValid(L))
Next
End Sub
