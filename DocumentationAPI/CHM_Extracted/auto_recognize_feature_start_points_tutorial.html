Automatically Recognizing Feature Start Points

In the Tutorial: Accessing the Features Resulting from FeatureRecognition the feature chains that resulted from FeatureRecognition were returned as an array by the CreateAutoChains method. In addition to looping throught the elements of the array, the elements of the features in the array themselves can be looped through if necessary, just like as was done in Tutorial: Making a FeatureChain into a Spiral.

Suppose, for example, that the file that we are doing feature recognition on includes start points along some of the elements. We can try to Split the feature at that point, and compare the Count before and after. If the count increases, then we know there was a successful split, and we can loop through the elements of the items in the FeatureChain, comparing their endpoints to the point where the split took place.

private static Esprit.FeatureChain RestartFeatureChain(Esprit.Document document, Esprit.FeatureChain chainToRestart, Esprit.Point splitPoint)
{
var originalCount = chainToRestart.Count;
try
{
chainToRestart.Split(splitPoint);
}
catch (System.Runtime.InteropServices.COMException)
{
}
var featureChain = chainToRestart;
if (chainToRestart.Count > originalCount)
{
Esprit.Point endPoint = null;
for (var i = 1; i <= chainToRestart.Count; i++)
{
switch (chainToRestart.Item[i].GraphicObjectType)
{
case EspritConstants.espGraphicObjectType.espArc:
var arc = chainToRestart.Item[i] as Esprit.Arc;
endPoint = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
break;
case EspritConstants.espGraphicObjectType.espSegment:
var segment = chainToRestart.Item[i] as Esprit.Segment;
endPoint = segment.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
break;
}
if (endPoint != null && SamePoint(endPoint, splitPoint))
{
featureChain = document.FeatureChains.Add(splitPoint);
for (var j = i+1; j<= chainToRestart.Count; j++)
{
try
{
featureChain.Add(chainToRestart.Item[j]);
}
catch (Exception)
{ }
}
for (var j = 1; j <= i; j++)
{
try
{
featureChain.Add(chainToRestart.Item[j]);
}
catch (Exception)
{ }
}
break;
}
}
}
return featureChain;
}
private static bool SamePoint(Esprit.Point point1, Esprit.Point point2)
{
return ((point1.X == point2.X) && (point1.Y == point2.Y) && (point1.Z == point2.Z));
}
Private Function RestartFeatureChain(document As Esprit.Document, chainToRestart As Esprit.FeatureChain, splitPoint As Esprit.Point) As Esprit.FeatureChain
Dim originalCount As Integer = chainToRestart.Count
Try
chainToRestart.Split(splitPoint)
Catch ex As System.Runtime.InteropServices.COMException
End Try
Dim featureChain As Esprit.FeatureChain = chainToRestart
If chainToRestart.Count > originalCount Then
Dim endPoint As Esprit.Point = Nothing
For i As Integer = 1 To chainToRestart.Count
Select Case chainToRestart.Item(i).GraphicObjectType
Case EspritConstants.espGraphicObjectType.espArc
Dim arc As Esprit.Arc = chainToRestart.Item(i)
endPoint = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd)
Case EspritConstants.espGraphicObjectType.espSegment
Dim segment As Esprit.Segment = chainToRestart.Item(i)
endPoint = segment.Extremity(EspritConstants.espExtremityType.espExtremityEnd)
End Select
If Not endPoint Is Nothing AndAlso SamePoint(endPoint, splitPoint) Then
featureChain = document.FeatureChains.Add(splitPoint)
For j As Integer = i + 1 To chainToRestart.Count
Try
featureChain.Add(chainToRestart.Item(j))
Catch ex As Exception
End Try
Next
For j As Integer = 1 To i
Try
featureChain.Add(chainToRestart.Item(j))
Catch ex As Exception
End Try
Next
Exit For
End If
Next
End If
Return featureChain
End Function
Private Function SamePoint(point1 As Esprit.Point, point2 As Esprit.Point) As Boolean
Return ((point1.X = point2.X) And (point1.Y = point2.Y) And (point1.Z = point2.Z))
End Function
Function RestartFeatureChain(ChainToRestart As Esprit.FeatureChain, SplitPoint As Esprit.Point) As Esprit.FeatureChain
Dim OriginalCount As Long
OriginalCount = ChainToRestart.Count
On Error Resume Next
Call ChainToRestart.Split(SplitPoint)
If ChainToRestart.Count > OriginalCount Then
' successful split, loop through the feature and find the split location
Dim I As Long, A As Esprit.Arc, S As Esprit.Segment, EndP As Esprit.Point
For I = 1 To ChainToRestart.Count
Select Case ChainToRestart.Item(I).GraphicObjectType
Case espArc
Set A = ChainToRestart.Item(I)
Set EndP = A.Extremity(espExtremityEnd)
Case espSegment
Set S = ChainToRestart.Item(I)
Set EndP = S.Extremity(espExtremityEnd)
End Select
' start new feature at the split point
If SamePoint(EndP, SplitPoint) Then
Set RestartFeatureChain = Document.FeatureChains.Add(SplitPoint)
Exit For
End If
Next
' add the remaining elements onto the new chain
If Not (RestartFeatureChain Is Nothing) Then
Dim J As Long
For J = I + 1 To ChainToRestart.Count
Call RestartFeatureChain.Add(ChainToRestart.Item(J))
Next
For J = 1 To I
Call RestartFeatureChain.Add(ChainToRestart.Item(J))
Next
End If
End If
If RestartFeatureChain Is Nothing Then
Set RestartFeatureChain = ChainToRestart
End If
End Function
Function SamePoint(P1 As Esprit.Point, P2 As Esprit.Point) As Boolean
If (P1.X = P2.X) And (P1.Y = P2.Y) And (P1.Z = P2.Z) Then
SamePoint = True
Else
SamePoint = False
End If
End Function

public void ChainAllCWCloseGapsAndChangeStart()
{
var set = SelectionSetHelper.GetOrAddSelectionSet(Document.SelectionSets, "Temp");
set.RemoveAll();
var set2 = SelectionSetHelper.GetOrAddSelectionSet(Document.SelectionSets, "Temp2");
set2.RemoveAll();
foreach (Esprit.IGraphicObject graphicObject in Document.GraphicsCollection)
{
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espArc || graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espSegment)
{
set.Add(graphicObject);
}
}
if (set.Count == 0)
{
return;
}
var objects = Document.FeatureRecognition.CreateAutoChains(set);
for (var i = objects.GetLowerBound(0); i <= objects.GetUpperBound(0); i++)
{
var graphicObject = objects.GetValue(i) as Esprit.GraphicObject;
if (graphicObject.GraphicObjectType == EspritConstants.espGraphicObjectType.espFeatureChain)
{
var featureChain = graphicObject as Esprit.FeatureChain;
if (!featureChain.IsClosed)
{
var startPoint = featureChain.Extremity(EspritConstants.espExtremityType.espExtremityStart);
var endPoint = featureChain.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
if (Geometry.Distance(startPoint, endPoint) < _application.Configuration.ConfigurationFeatureRecognition.Tolerance)
{
featureChain.Add(startPoint);
}
}
if (featureChain.IsClosed) // check again
{
set2.AddCopiesToSelectionSet = true;
set2.RemoveAll();
set2.Add(featureChain);
try
{
set2.Offset(0.001, EspritConstants.espOffsetSide.espOffsetRight, false, EspritConstants.espLookAheadMode.espLookAheadOn, true);
var offsetFeatureChain = set2[2] as Esprit.FeatureChain;
if (offsetFeatureChain.Area > featureChain.Area)
{
featureChain.Reverse();
}
Document.FeatureChains.Remove(offsetFeatureChain.Key);
}
catch(Exception)
{ }
}
foreach(Esprit.Point point in Document.Points)
{
var newFC = RestartFeatureChain(Document, featureChain, point);
}
}
}
Document.SelectionSets.Remove(set.Name);
Document.SelectionSets.Remove(set2.Name);
}

The following function is used to calculate distance between two points:
public static double Distance(Point point1, Point point2 = null)
{
var x2 = point2?.X ?? 0;
var y2 = point2?.Y ?? 0;
var z2 = point2?.Z ?? 0;
return Distance(point1.X, point1.Y, point1.Z, x2, y2, z2);
}
public static double Distance(double x1, double y1, double z1, double x2, double y2, double z2)
{
var dx = x2 - x1;
var dy = y2 - y1;
var dz = z2 - z1;
return Math.Sqrt(dx * dx + dy * dy + dz * dz);
}

Public Function Distance(Point1 As Esprit.Point, Point2 As Esprit.Point) As Double
Dim DeltaX As Double, DeltaY As Double, DeltaZ As Double
DeltaX = Point2.X - Point1.X
DeltaY = Point2.Y - Point1.Y
DeltaZ = Point2.Z - Point1.Z
Distance = Math.Sqrt(DeltaX ^ 2 + DeltaY ^ 2 + DeltaZ ^ 2)
End Function
Sub ChainAllCWCloseGapsAndChangeStart()
Dim SS(1 To 2) As Esprit.SelectionSet
Set SS(1) = GetSelectionSet("Temp")
Set SS(2) = GetSelectionSet("Temp2")
SS(1).RemoveAll
' loop through the graphics and include all bound segments and arcs in the selection set
Dim G As Esprit.GraphicObject
For Each G In Document.GraphicsCollection
If (G.GraphicObjectType = espArc) Or (G.GraphicObjectType = espSegment) Then
Call SS(1).Add(G)
End If
Next
' create the feature chains
Dim GraphicObj() As Esprit.GraphicObject
If SS(1).Count > 0 Then
GraphicObj = Document.FeatureRecognition.CreateAutoChains(SS(1))
End If
' loop through all of the newly created features
For I = LBound(GraphicObj) To UBound(GraphicObj)
If GraphicObj(I).GraphicObjectType = espFeatureChain Then
Dim FC As Esprit.FeatureChain
Set FC = GraphicObj(I)
' check the end points and close a small gap with segment
If Not FC.IsClosed Then
Dim P(1 To 2) As Esprit.Point
Set P(1) = FC.Extremity(espExtremityStart)
Set P(2) = FC.Extremity(espExtremityEnd)
If Distance(P(1), P(2)) < Application.Configuration.ConfigurationFeatureRecognition.Tolerance Then
Call FC.Add(P(1))
End If
End If
' check again to see if it is closed and reverse to go CW if necessary
If FC.IsClosed Then
SS(2).AddCopiesToSelectionSet = True
SS(2).RemoveAll
' offset by a small distance and compare areas to determine direction
Call SS(2).Add(FC)
Call SS(2).Offset(0.001, espOffsetRight, False, espLookAheadOn, True)
Dim OffsetFC As Esprit.FeatureChain
Set OffsetFC = SS(2).Item(2)
If OffsetFC.Area > FC.Area Then
' original was counter-clockwise, so reverse
FC.Reverse
End If
Call Document.FeatureChains.Remove(OffsetFC.Key)
End If
' loop through all of the points and see if any lie on a feature element
' split and restart the feature at the first point that does
Dim Point As Esprit.Point
For Each Point In Document.Points
Dim NewFC As Esprit.FeatureChain
Set NewFC = RestartFeatureChain(FC, Point)
Next
End If
Next
' cleanup
Call Document.SelectionSets.Remove(SS(1).Name)
Call Document.SelectionSets.Remove(SS(2).Name)
End Sub
