Scan Dimension and Notes Text

In the Tutorial: Scan Annotations, we saw that the Annotations collection could contain both Notes and Dimension objects. The Dimension objects themselves, then, would contain a Notes object. Both the Notes objects within the Annotations collection and the Notes object within each Dimension contains individual Note objects. Each Note object inside a Notes belonging to Annotations corresponds to one line of the Note. The Note objects inside of the Notes belonging to the Dimension will contain the dimension value and its tolerances, if any.

This tutorial shows how to find and display the information inside of all of these Notes.

public void ScanDimensionAndNoteText()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
if (Document.Annotations.Count == 0)
{
MessageBox.Show("There are no Annotations", "ScanDimensionAndNotesTutorial");
return;
}
var index = 0;
foreach (Esprit.Annotation annotation in Document.Annotations)
{
switch (annotation.AnnotationType)
{
case EspritConstants.espAnnotationType.espAnnotationDimension:
index++;
var annotationDimension = annotation as Esprit.Dimension;
for (var i = 1; i <= annotationDimension.Notes.Count; i++)
{
var note = annotationDimension.Notes.Item[i];
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNotesTutorial", $"Dimension {annotationDimension.Key} line {i} is {note.String}");
}
break;
case EspritConstants.espAnnotationType.espAnnotationNote:
index++;
var annotationNotes = annotation as Esprit.Notes;
for (var i = 1; i <= annotationNotes.Count; i++)
{
var note = annotationNotes.Item[i];
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNotesTutorial", $"Note {annotationNotes.Key} line {i} is {note.String}");
}
break;
}
}
if (index == 0)
{
MessageBox.Show("There are no Dimension or Note Annotations", "ScanDimensionAndNotesTutorial");
}
}
Public Sub ScanDimensionAndNoteText()
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
If Document.Annotations.Count = 0 Then
MsgBox("There are no Annotations", vbOKOnly)
Return
End If
Dim index As Integer = 0
For Each annotation As Esprit.Annotation In Document.Annotations
Select Case annotation.AnnotationType
Case EspritConstants.espAnnotationType.espAnnotationDimension
index += 1
Dim annotationDimension As Esprit.Dimension = annotation
' loop through each Note contained in the dimension
For j As Integer = 1 To annotationDimension.Notes.Count
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNotesTutorial", "Dimension " & annotationDimension.Key & " line " & j & " is " & annotationDimension.Notes.Item(j).String)
Next
Case EspritConstants.espAnnotationType.espAnnotationNote
index += 1
Dim annotationNotes As Esprit.Notes = annotation
' loop through each Note line within the Notes
For j As Integer = 1 To annotationNotes.Count
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNotesTutorial", "Note " & annotationNotes.Key & " line " & j & " is " & annotationNotes.Item(j).String)
Next
End Select
Next
If index = 0 Then
MsgBox("There are no Dimension or Note Annotations", vbOKOnly)
End If
End Sub
Sub ScanDimensionAndNoteText()
Application.EventWindow.Clear
Application.EventWindow.Visible = True
Dim An As Esprit.Annotation
Dim I As Long, J As Long
If Document.Annotations.Count = 0 Then
Call MsgBox("There are no Annotations", vbOKOnly)
Exit Sub
End If
Dim Index As Integer
Index = 0
For I = 1 To Document.Annotations.Count
Set An = Document.Annotations.Item(I)
' type cast the annonation to a specific object based on the type
Select Case An.AnnotationType
Case espAnnotationDimension
Index = Index + 1
Dim D As Esprit.Dimension
Set D = An
' loop through each Note contained in the dimension
For J = 1 To D.Notes.Count
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNoteTextTutorial", "Dimension " & D.Key & " line " & I & " is " & D.Notes.Item(J).String)
Next
Case espAnnotationNote
Index = Index + 1
Dim N As Esprit.Notes
Set N = An
' loop through each Note line within the Notes
For J = 1 To N.Count
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanDimensionAndNoteTextTutorial", "Note " & N.Key & " line " & J & " is " & N.Item(J).String)
Next
End Select
Next
If Index = 0 Then
Call MsgBox("There are no Dimension or Note Annotations", vbOKOnly)
End If
End Sub
