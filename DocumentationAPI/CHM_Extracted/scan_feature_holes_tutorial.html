public void ScanFeatureHoles()
{
foreach(Esprit.IHolesFeature feature in Document.HolesFeatures)
{
var stringBuilder = new StringBuilder();
stringBuilder.AppendLine("Common data:");
stringBuilder.AppendLine($"\tFeature Hole name: {feature.Name}");
stringBuilder.AppendLine($"\tFeature Hole depth: {feature.Depth}");
stringBuilder.AppendLine($"\tFeature Hole diameter: {feature.Diameter}");
stringBuilder.AppendLine($"\tFeature Hole throughAll: {feature.ThroughAll}");
stringBuilder.AppendLine($"\tFeature Hole Plane Name: {feature.Plane.Name}");
stringBuilder.AppendLine($"\tFeature Hole WorkCoordinate Name: {feature.WorkCoordinate.Name}");
var comFeature = feature.Object as EspritFeatures.ComFeatureHoles;
stringBuilder.AppendLine($"\tFeature Holes count: {comFeature.Count}");
var holes = comFeature.GetHoles();
for (var i = holes.GetLowerBound(0); i <= holes.GetUpperBound(0); i++)
{
var hole = holes.GetValue(i) as EspritFeatures.ComFeatureHole;
stringBuilder.AppendLine($"Hole {i+1} data:");
stringBuilder.AppendLine($"\tHole diameter: {hole.HoleDiameter}");
stringBuilder.AppendLine($"\tHole depth: {hole.HoleDepth}");
stringBuilder.AppendLine($"\tHole throughAll: {hole.ThroughAll}");
}
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "HolesFeature", stringBuilder.ToString());
}
}

Sub ScanFeatureHoles()
Dim feature As Esprit.IHolesFeature
For Each feature In Document.HolesFeatures
Dim Message As String
Message = "Common data:" & vbNewLine
Message = Message & Chr(9) & "Feature Hole name: " & feature.Name & vbNewLine
Message = Message & Chr(9) & "Feature Hole depth: " & feature.Depth & vbNewLine
Message = Message & Chr(9) & "Feature Hole diameter: " & feature.Diameter & vbNewLine
Message = Message & Chr(9) & "Feature Hole throughAll: " & feature.ThroughAll & vbNewLine
Message = Message & Chr(9) & "Feature Hole Plane Name: " & feature.Plane.Name & vbNewLine
Message = Message & Chr(9) & "Feature Hole WorkCoordinate Name: " & feature.WorkCoordinate.Name & vbNewLine
Dim comFeature As EspritFeatures.ComFeatureHoles
Set comFeature = feature.Object
Message = Message & Chr(9) & "Feature Holes count: " & comFeature.Count & vbNewLine
Dim holes() As ComFeatureHole
holes = comFeature.GetHoles()
Dim i As Integer
For i = 0 To UBound(holes)
Dim hole As EspritFeatures.ComFeatureHole
Set hole = holes(i)
Message = Message & "Hole " & i + 1 & " data:" & vbNewLine
Message = Message & Chr(9) & "Hole diameter: " & hole.HoleDiameter & vbNewLine
Message = Message & Chr(9) & "Hole depth: " & hole.HoleDepth & vbNewLine
Message = Message & Chr(9) & "Hole throughAll: " & hole.ThroughAll & vbNewLine
Next
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "HolesFeature", Message)
Next
End Sub
