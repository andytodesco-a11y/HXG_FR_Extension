public void Load()
{
var configuration = _application.Configuration;
var tools = Document.Tools as EspritTools.Tools;
var toolPageFolder = configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage);
var dirInfo = new DirectoryInfo(toolPageFolder);
var filePaths = dirInfo.GetFiles("*.test");
if (filePaths.Length > 0)
{
foreach (var filePath in filePaths)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FilesAndFoldersLocationTutorial", filePath.FullName);
}
System.Windows.MessageBox.Show($"Loaded tools from {filePaths[0].FullName}", "FilesAndFoldersLocationTutorial");
tools.LoadTools(filePaths[0].FullName);
}
}
Public Sub Load()
Dim Configuration As Esprit.Configuration = _application.Configuration
Dim tools As EspritTools.Tools = Document.Tools
Dim toolPageFolder As String = Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeToolPage)
Dim dirInfo = New DirectoryInfo(toolPageFolder)
Dim filePaths = dirInfo.GetFiles("*.test")
If (filePaths.Length > 0) Then
For Each filePath In filePaths
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FilesAndFoldersLocationTutorial", filePath.FullName)
Next
System.Windows.MessageBox.Show($"Loaded tools from {filePaths(0).FullName}", "FilesAndFoldersLocationTutorial")
tools.LoadTools(filePaths(0).FullName)
End If
End Sub
Sub Load()
Dim MyTools As EspritTools.Tools
Set MyTools = Document.Tools
Dim ToolPageFolder As String
Dim FirstTool As String
FirstTool = ""
ToolPageFolder = Application.Configuration.GetFileDirectory(espFileTypeToolPage)
Dim ToolPageFile As String
ToolPageFile = Dir(ToolPageFolder & "\*.test")
Do Until ToolPageFile = ""
' Note how the Dir() function is used to advance to the next matching file
If Len(FirstTool) = 0 Then
FirstTool = ToolPageFile
Call MyTools.LoadTools(ToolPageFolder & "\" & ToolPageFile)
Call LineOut("Loaded tools from " & ToolPageFolder)
Call MsgBox("Loaded tools from" & ToolPageFolder & "\" & ToolPageFile, vbOKOnly, "FilesAndFoldersLocationTutorial")
End If
ToolPageFile = Dir()
Loop
End Sub
