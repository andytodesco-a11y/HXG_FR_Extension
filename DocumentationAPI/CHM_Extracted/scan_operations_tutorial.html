Scan Operations

In this tutorial you will loop through all of the operations in the document and display certain information about them.

public void ScanOperations()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
foreach (Esprit.PartOperation partOperation in Document.PartOperations)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", $"OP {partOperation.Key} is a {partOperation.TypeName}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", $"OP {partOperation.Key} is on Layer {partOperation.Layer.Name}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", $"OP {partOperation.Key} Length of Feed is {partOperation.LengthOfFeed.ToString("F3")}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", $"OP {partOperation.Key} Length of Rapid is {partOperation.LengthOfRapid.ToString("F3")}");
}
}
Public Sub ScanOperations()
_application.EventWindow.Visible = True
_application.EventWindow.Clear()
For Each partOperation As Esprit.PartOperation In Document.PartOperations
' Strip the first four characters (Tech) off of the technology typename for display
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & partOperation.Key & " is a " & partOperation.TypeName)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & partOperation.Key & " is on Layer " & partOperation.Layer.Name)
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & partOperation.Key & " Length of Feed is " & partOperation.LengthOfFeed.ToString("F3"))
Call _application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & partOperation.Key & " Length of Rapid is " & partOperation.LengthOfRapid.ToString("F3"))
Next
End Sub
Sub ScanOperations()
' for any operations which are not in the library yet
' the following line of code is necessary to skip past them
On Error Resume Next
Application.EventWindow.Visible = True
Application.EventWindow.Clear
Dim OP As Esprit.PartOperation
For I = 1 To Document.PartOperations.Count
Set OP = Document.PartOperations.Item(I)
' Strip the first four characters (Tech) off of the technology typename for display
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " is a " & OP.TypeName)
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " is on Layer " & OP.Layer.Name)
If Document.SystemUnit = espMetric Then
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " Length of Feed is " & Format(OP.LengthOfFeed, "#.000") & " MM")
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " Length of Rapid is " & Format(OP.LengthOfRapid, "#.000") & " MM")
Else
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " Length of Feed is " & Format(OP.LengthOfFeed, "#.000") & " IN")
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanOperationsTutorial", "OP" & OP.Key & " Length of Rapid is " & Format(OP.LengthOfRapid, "#.000") & " IN")
End If
Next
End Sub
