Introduction to SelectionSet Manipulations: XY Symmetry

The primary use of the SelectionSet object is for manipulation, for example - Translation and Rotation.

Suppose that you often symmetry a group over the X axis, and then symmetry the original and its copy over the Y axis. Instead of doing this in two steps, you want to do it in one. What we will need to do is transfer the group into a selection set, then use the Symmetry method twice. Note the role that the AddCopiesToSelectionSet property plays.

public void XYSymmetry()
{
var groupName = "Temp";
var set = SelectionSetHelper.GetSelectionSet(Document, groupName);
set.RemoveAll();
for(var i = 1; i <= Document.Group.Count; i++)
{
set.Add(Document.Group[i]);
}
set.AddCopiesToSelectionSet = true;
set.Symmetry(Document.GetLine(Document.GetPoint(0, 0, 0), 1, 0, 0), true, true);
set.Symmetry(Document.GetLine(Document.GetPoint(0, 0, 0), 0, 1, 0), true, true);
Document.Refresh();
Document.SelectionSets.Remove(groupName);
}
public static Esprit.SelectionSet GetSelectionSet(Esprit.Document document, string name)
{
var set = document.SelectionSets[name];
if (set == null)
{
set = document.SelectionSets.Add(name);
}
return set;
}
Public Sub XYSymmetry()
Dim selectionSetName As String = "Temp"
Dim selectionSet As Esprit.SelectionSet = SelectionSetHelper.GetSelectionSet(Document, selectionSetName)
selectionSet.RemoveAll()
For i As Integer = 1 To Document.Group.Count
Call selectionSet.Add(Document.Group.Item(i))
Next
' perform the symmetry manipulations
selectionSet.AddCopiesToSelectionSet = True
Call selectionSet.Symmetry(Document.GetLine(Document.GetPoint(0, 0, 0), 1, 0, 0), True, True)
Call selectionSet.Symmetry(Document.GetLine(Document.GetPoint(0, 0, 0), 0, 1, 0), True, True)
Document.Refresh()
' clean up by removing Temp SelectionSet
Call Document.SelectionSets.Remove(selectionSetName)
End Sub
Public Function GetSelectionSet(document As Esprit.Document, name As String) As Esprit.SelectionSet
Dim selectionSet As Esprit.SelectionSet = document.SelectionSets.Item(name)
If selectionSet Is Nothing Then
selectionSet = document.SelectionSets.Add(name)
End If
Return selectionSet
End Function

Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)
On Error GoTo 0
If GetSelectionSet Is Nothing Then
Set GetSelectionSet = Document.SelectionSets.Add(SSName)
End If
End Function
