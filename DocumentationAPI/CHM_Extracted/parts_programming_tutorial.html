Introduction to Family of Parts Programming

Suppose we have a family of lathe parts with the same features but different dimensions. We want to prompt the user for the dimensions, and then draw the resulting geometry and features.

In this particular example, let us have a part that is threaded at a certain diameter for a proportion of its length. Then it has fillet radius onto a 45-degree chamfer out to another diameter. The 45-degree chamfer reaches the outside diameter at exactly half of the total part length. In the middle of that next section is a groove of a specific width and a depth equal to half of the difference between the outside and threaded diameters. Finally we will want to cutoff the part at its final length, and there happens to be a fillet around that corner of the same radius as the fillet between the threaded and chamfered sections.

We will want to prompt the user for the following key dimensions:

  • Part Length
  • Outside Diameter
  • Fillet Radius
  • Groove Width
  • Threaded Diameter

Those key dimension and the other dimensions relative to them are shown in the following diagram:

On this geometry there will be five features created for us to machine later. Notice that the threading will not be done as just two points, but rather a feature is made between those two thread points.

We can make a class for our Lathe Part Family which contains all of our key dimensions as properties and provides a method for creating the geometry and features:

: value;

public Esprit.SelectionSet DrawPart()
{
var document = _application.Document;
var set = SelectionSetHelper.AddUniqueSelectionSet(document.SelectionSets, "LatheFamilyPart");
var point1 = document.GetPoint(0, 0, 0);
var point2 = document.GetPoint(0, ThreadedDiameter / 2.0, 0);
var segment = document.Segments.Add(point1, point2);
set.Add(segment);
var localFaceContourFeature = document.FeatureChains.Add(point2);
localFaceContourFeature.Add(point1);
point1 = point2;
var radialDifference = (OutsideDiameter - ThreadedDiameter) / 2.0;
point2 = document.GetPoint(-(Length / 2) + radialDifference + FilletRadius * (1 - 2 * (Math.Sqrt(2) - 1) * Math.Cos(Geometry.ToRadians(45))), point1.Y, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
var localODRoughFeature = document.FeatureChains.Add(point1);
localODRoughFeature.Add(segment);
var localThreadFeature = document.FeatureChains.Add(point1);
localThreadFeature.Add(segment);
var point3 = document.GetPoint(point2.X, point2.Y + FilletRadius, 0);
var arc = document.Arcs.Add(point3, FilletRadius, Geometry.ToRadians(225), Geometry.ToRadians(270));
set.Add(arc);
arc.Reverse();
localODRoughFeature.Add(arc);
point1 = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
point2 = document.GetPoint(point1.X - ((OutsideDiameter / 2.0) - point1.Y), OutsideDiameter / 2.0, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
localODRoughFeature.Add(segment);
point1 = point2;
point2 = document.GetPoint(-(3 * Length / 4) + (GrooveWidth / 2), point1.Y, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
point1 = point2;
point2 = document.GetPoint(point1.X, point1.Y - radialDifference / 2.0, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
var localGrooveFeature = document.FeatureChains.Add(point1);
localGrooveFeature.Add(segment);
point1 = point2;
point2 = document.GetPoint(point1.X - GrooveWidth, point1.Y, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
localGrooveFeature.Add(segment);
point1 = point2;
point2 = document.GetPoint(point1.X, OutsideDiameter / 2.0, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
localGrooveFeature.Add(segment);
point1 = point2;
point2 = document.GetPoint(-Length + FilletRadius, point1.Y, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
point3 = document.GetPoint(point2.X, point2.Y - FilletRadius, 0);
arc = document.Arcs.Add(point3, FilletRadius, Geometry.ToRadians(90), Geometry.ToRadians(180));
set.Add(arc);
localODRoughFeature.Add(document.GetPoint(-Length, point1.Y, 0));
var localCutoffFeature = document.FeatureChains.Add(arc.Extremity(EspritConstants.espExtremityType.espExtremityStart));
localCutoffFeature.Add(arc);
point1 = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
point2 = document.GetPoint(-Length, 0, 0);
segment = document.Segments.Add(point1, point2);
set.Add(segment);
localCutoffFeature.Add(segment);
localODRoughFeature.Name = "1 - OD Rough";
localFaceContourFeature.Name = "2 - Face Contour";
localThreadFeature.Name = "3 - Thread";
localGrooveFeature.Name = "4 - Groove";
localCutoffFeature.Name = "5 - Cutoff";
set.Add(localODRoughFeature);
set.Add(localFaceContourFeature);
set.Add(localGrooveFeature);
set.Add(localThreadFeature);
set.Add(localCutoffFeature);
SelectionSetHelper.DefaultAttributesSet(set, _application.Configuration);
document.Refresh();
return set;
}

public static SelectionSet AddUniqueSelectionSet(ISelectionSets selectionSets, string baseName)
{
var suffix = 0;
foreach (SelectionSet set in selectionSets)
{
var setName = set.Name;
if (setName.Length > baseName.Length && setName.StartsWith(baseName) && setName[baseName.Length] == '_')
{
if (Int32.TryParse(setName.Substring(baseName.Length + 1), out var index))
{
if (index >= suffix)
{
suffix = index + 1;
}
}
}
}
var name = $"{baseName}_{suffix}";
return selectionSets.Add(name);
}

Public Function AddUniqueSelectionSet(document As Esprit.Document, baseName As String) As Esprit.SelectionSet
Dim baseNameLen As Long = Len(baseName)
Dim highestSSNumber As Long = 0
' loop through all SelectionSets to find ones with the same BaseName
For Each selectionSet As Esprit.SelectionSet In document.SelectionSets
' compare the position of the BaseName within the SelectionSet Name
If (InStr(1, selectionSet.Name, baseName) = 1) Then
' find the number at the end of the name
Dim sSNumber As Long = Val(Mid$(selectionSet.Name, baseNameLen + 1))
' keep track of the highest number so that it can be incremented later
If sSNumber > highestSSNumber Then highestSSNumber = sSNumber
End If
Next
' increment the highest number and return the resulting SelectionSet
baseName &= CStr(highestSSNumber + 1)
Return document.SelectionSets.Add(baseName)
End Function

Public Function DrawPart() As Esprit.SelectionSet
' set up the return selection set and other initial variables
Set DrawPart = AddUniqueSelectionSet("LatheFamilyPart")
Dim P(0 To 2) As Esprit.Point
Dim S As Esprit.Segment
Dim A As Esprit.Arc
Dim RadialDifference As Double
RadialDifference = (OutsideDiameter - ThreadedDiameter) / 2
' face segment
Set P(1) = Document.GetPoint(0, 0, 0)
Set P(2) = Document.GetPoint(0, ThreadedDiameter / 2, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Set localFaceContourFeature = Document.FeatureChains.Add(P(2))
Call localFaceContourFeature.Add(P(1))
' thread segment, note how X value of end point was trigged out
Set P(1) = P(2)
Set P(2) = Document.GetPoint(-(Length / 2) + RadialDifference + FilletRadius * (1 - 2 * (Sqr(2) - 1) * Cos(ToRadians(45))), P(1).Y, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Set localODRoughFeature = Document.FeatureChains.Add(P(1))
Call localODRoughFeature.Add(S)
Set localThreadFeature = Document.FeatureChains.Add(P(1))
Call localThreadFeature.Add(S)
' fillet radius (must define CCW, such that StartAngle < EndAngle)
Set P(0) = Document.GetPoint(P(2).X, P(2).Y + FilletRadius, 0)
Set A = Document.Arcs.Add(P(0), FilletRadius, ToRadians(225), ToRadians(270))
Call DrawPart.Add(A)
Call A.Reverse ' so that arc goes CW instead of CCW
Call localODRoughFeature.Add(A)
' large chamfer segment (45 degrees)
Set P(1) = A.Extremity(espExtremityEnd)
Set P(2) = Document.GetPoint(P(1).X - ((OutsideDiameter / 2) - P(1).Y), OutsideDiameter / 2, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Call localODRoughFeature.Add(S)
' OD segment ahead of groove
Set P(1) = P(2)
Set P(2) = Document.GetPoint(-(3 * Length / 4) + (GrooveWidth / 2), P(1).Y, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
' Groove wall
Set P(1) = P(2)
Set P(2) = Document.GetPoint(P(1).X, P(1).Y - RadialDifference / 2, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Set localGrooveFeature = Document.FeatureChains.Add(P(1))
Call localGrooveFeature.Add(S)
' Groove floor
Set P(1) = P(2)
Set P(2) = Document.GetPoint(P(1).X - GrooveWidth, P(1).Y, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Call localGrooveFeature.Add(S)
' opposite Groove wall
Set P(1) = P(2)
Set P(2) = Document.GetPoint(P(1).X, OutsideDiameter / 2, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Call localGrooveFeature.Add(S)
' OD segment in back of groove
Set P(1) = P(2)
Set P(2) = Document.GetPoint(-Length + FilletRadius, P(1).Y, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
' arc for Cutoff relief
Set P(0) = Document.GetPoint(P(2).X, P(2).Y - FilletRadius, 0)
Set A = Document.Arcs.Add(P(0), FilletRadius, ToRadians(90), ToRadians(180))
Call DrawPart.Add(A)
' so that it jumps over groove and past arc
Call localODRoughFeature.Add(Document.GetPoint(-Length, P(1).Y, 0))
Set localCutoffFeature = Document.FeatureChains.Add(A.Extremity(espExtremityStart))
Call localCutoffFeature.Add(A)
' Cutoff segment
Set P(1) = A.Extremity(espExtremityEnd)
Set P(2) = Document.GetPoint(-Length, 0, 0)
Set S = Document.Segments.Add(P(1), P(2))
Call DrawPart.Add(S)
Call localCutoffFeature.Add(S)
' set default feature names
localODRoughFeature.Name = "1 - OD Rough"
localFaceContourFeature.Name = "2 - Face Contour"
localThreadFeature.Name = "3 - Thread"
localGrooveFeature.Name = "4 - Groove"
localCutoffFeature.Name = "5 - Cutoff"
' add features to selection set, set default attributes and refresh graphics
Call DrawPart.Add(localODRoughFeature)
Call DrawPart.Add(localFaceContourFeature)
Call DrawPart.Add(localGrooveFeature)
Call DrawPart.Add(localThreadFeature)
Call DrawPart.Add(localCutoffFeature)
Call DefaultAttributesSet(DrawPart)
Document.Refresh
End Function

Public Function AddUniqueSelectionSet(BaseName As String) As Esprit.SelectionSet
Dim BaseNameLen As Long, HighestSSNumber As Long
BaseNameLen = Len(BaseName)
HighestSSNumber = 0
' loop through all SelectionSets to find ones with the same BaseName
Dim SS As Esprit.SelectionSet, SSNumber As Long
For Each SS In Document.SelectionSets
' compare the position of the BaseName within the SelectionSet Name
If (InStr(1, SS.Name, BaseName) = 1) Then
' find the number at the end of the name
SSNumber = Val(Mid$(SS.Name, BaseNameLen + 1))
' keep track of the highest number so that it can be incremented later
If SSNumber > HighestSSNumber Then HighestSSNumber = SSNumber
End If
Next
' increment the highest number and return the resulting SelectionSet
BaseName = BaseName & CStr(HighestSSNumber + 1)
