public static Esprit.Plane AddUniquePlane(Esprit.IPlanes planes, string baseName)
{
var suffix = 0;
foreach (Esprit.Plane plane in planes)
{
var planeName = plane.Name;
if (planeName.Length > baseName.Length && planeName.StartsWith(baseName) && planeName[baseName.Length] == '_')
{
if (int.TryParse(planeName.Substring(baseName.Length + 1), out var index))
{
if (index >= suffix)
{
suffix = index + 1;
}
}
}
}
var name = $"{baseName}_{suffix}";
return planes.Add(name);
}

Public Function AddUniquePlane(BaseName As String) As Esprit.Plane
Dim BaseNameLen As Long, HighestNumber As Long
BaseNameLen = Len(BaseName)
HighestNumber = 0
' loop through all Planes to find ones with the same BaseName
Dim P As Esprit.Plane, PNumber As Long
For Each P In Document.Planes
' compare the position of the BaseName within the Plane Name
If (InStr(1, P.Name, BaseName) = 1) Then
' find the number at the end of the name
PNumber = Val(Mid$(P.Name, BaseNameLen + 2))
' keep track of the highest number so that it can be incremented later
If PNumber > HighestNumber Then HighestNumber = PNumber
End If
Next
' increment the highest number and return the resulting Plane
BaseName = BaseName & "_" & CStr(HighestNumber + 1)
Set AddUniquePlane = Document.Planes.Add(BaseName)
End Function
