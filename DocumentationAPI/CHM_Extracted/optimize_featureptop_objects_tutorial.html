Optimizing FeaturePtop Objects

This tutorial shows how to optimize a FeaturePtop using the Optimize method which takes two arguments. The first argument is set to espPtopOptimizeTypeExtended - a true shortest path algorithm (for the chosen start element) and second argument is the start item index.

The code below loops through all elements of the chosen FeaturePtop, picks that item as a start point and compares the results of optimization. Note that each time we optimize a FeaturePtop, those items are rearranged and re-indexed. What we need to do to be correct is make a copy of FeaturePtop and then optimize that copy.

public void FullyOptimizeSelectedFeaturePtop()
{
Esprit.FeaturePtop featurePtop = null;
try
{
featurePtop = Document.GetAnyElement("Pick Feature Ptop", EspritConstants.espGraphicObjectType.espFeaturePtop) as Esprit.FeaturePtop;
}
catch(System.Runtime.InteropServices.COMException)
{
// Element was not set
return;
}
if (featurePtop == null)
{
return;
}
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
FeatureHelper.FullyOptimizeFeaturePtop(ref featurePtop, Document, _application.EventWindow);
}
public static void FullyOptimizeFeaturePtop(ref Esprit.FeaturePtop featurePtop, Esprit.Document document, Esprit.EventWindow outputWindow)
{
var set = SelectionSetHelper.GetOrAddSelectionSet(document.SelectionSets, "Temp");
set.RemoveAll();
set.Add(featurePtop);
set.AddCopiesToSelectionSet = true;
var shortestItemIndex = 0;
var length = 0.0;
for (var i = 1; i <= featurePtop.Count; i++)
{
set.Translate(0, 0, 0, 1);
var featurePtopCopy = set[2] as Esprit.FeaturePtop;
featurePtopCopy.Optimize(EspritConstants.espPtopOptimizeType.espPtopOptimizeTypeExtended, i);
outputWindow?.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", $"Length starting at Item {i} is {Math.Round(featurePtopCopy.Length, 5)}");
if (i == 1)
{
shortestItemIndex = i;
length = featurePtopCopy.Length;
}
else if (featurePtopCopy.Length < length)
{
length = featurePtopCopy.Length;
shortestItemIndex = i;
}
set.Remove(2);
document.FeaturePtops.Remove(featurePtopCopy.Key);
}
if (shortestItemIndex > 0)
{
featurePtop.Optimize(EspritConstants.espPtopOptimizeType.espPtopOptimizeTypeExtended, shortestItemIndex);
outputWindow?.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", $"Shortest path is {Math.Round(featurePtop.Length, 5)} starting at what was item {shortestItemIndex}");
document.Refresh();
}
}

Public Sub FullyOptimizeFeaturePtop(ByRef featurePtop As Esprit.FeaturePtop, document As Esprit.Document, outputWindow As Esprit.EventWindow)
Dim selectionSet = SelectionSetHelper.GetOrAddSelectionSet(document, "Temp")
selectionSet.RemoveAll()
selectionSet.Add(featurePtop)
selectionSet.AddCopiesToSelectionSet = True
Dim shortestItemIndex As Integer = 0
Dim length As Double = 0.0
For i As Integer = 1 To featurePtop.Count
selectionSet.Translate(0, 0, 0, 1)
Dim featurePtopCopy As Esprit.FeaturePtop = selectionSet(2)
featurePtopCopy.Optimize(EspritConstants.espPtopOptimizeType.espPtopOptimizeTypeExtended, i)
outputWindow?.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", $"Length starting at Item {i} is {Math.Round(featurePtopCopy.Length, 5)}")
If i = 1 Then
shortestItemIndex = i
length = featurePtopCopy.Length
ElseIf (featurePtopCopy.Length < length) Then
length = featurePtopCopy.Length
shortestItemIndex = i
End If
selectionSet.Remove(2)
document.FeaturePtops.Remove(featurePtopCopy.Key)
Next
If shortestItemIndex > 0 Then
featurePtop.Optimize(EspritConstants.espPtopOptimizeType.espPtopOptimizeTypeExtended, shortestItemIndex)
outputWindow?.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", $"Shortest path is {Math.Round(featurePtop.Length, 5)} starting at what was item {shortestItemIndex}")
document.Refresh()
End If
End Sub

Sub FullyOptimizeFeaturePtop(PtopToOptimize As Esprit.FeaturePtop)
Application.EventWindow.Clear
Application.EventWindow.Visible = True
Dim Shortest As Long, Length As Double
' use a selection set to create a copy of the original
Dim SS As Esprit.SelectionSet
On Error Resume Next
Set SS = Document.SelectionSets.Item("Temp")
On Error GoTo 0
If SS Is Nothing Then Set SS = Document.SelectionSets.Add("Temp")
SS.RemoveAll
Call SS.Add(PtopToOptimize)
SS.AddCopiesToSelectionSet = True
' loop through all of the points in the feature ptop
Dim PointNumber As Long
For PointNumber = 1 To PtopToOptimize.Count
' create copy by translating in place
Dim PtopCopy As Esprit.FeaturePtop
Call SS.Translate(0, 0, 0, 1)
Set PtopCopy = SS.Item(2)
' optimize the copy at the current location
Call PtopCopy.Optimize(espPtopOptimizeTypeExtended, PointNumber)
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", "Length starting at Item " & PointNumber & " is " & Round(PtopCopy.Length, DECIMAL_POINTS))
' keep track of the location that gives the shortest results
If PointNumber = 1 Then
Length = PtopCopy.Length
Shortest = 1
Else
If PtopCopy.Length < Length Then
Shortest = PointNumber
Length = PtopCopy.Length
End If
End If
' remove the copy
Call SS.Remove(2)
Call Document.FeaturePtops.Remove(PtopCopy.Key)
Next
' optimize the original feature ptop starting at the location that gave the shortest
Call PtopToOptimize.Optimize(espPtopOptimizeTypeExtended, Shortest)
Call Application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FullyOptimizeFeaturePtop", "Shortest path is " & Round(PtopToOptimize.Length, DECIMAL_POINTS) & " starting at what was item " & Shortest)
Document.Refresh
End Sub
