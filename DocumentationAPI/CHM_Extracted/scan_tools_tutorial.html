Scan Tools

In this tutorial you will loop through all of the tools in the document and display certain information about them.

In the Tutorial: Create Tools we worked with a couple of specific tool technology objects, ToolMillEndMill and ToolMillDrill. There is only one Tools collection, however. We do not have a separate collection for just the end mills and a separate collection for just the drills.

The Tools collection is similar to the GraphicsCollection that we looked at in Tutorial: Scan the Graphics Collection. The GraphicsCollection contains GraphicObject objects. Those objects are generic objects that are themselves a more specific type of object such as an Arc, Circle, or Point.

So just like there is a generic GraphicObject that has just those properties common to all graphic objects, so too there is a generic Tool object that has just those properties common to all tool technologies. All tools, for example, have a ToolID, as well as a ToolNumber, and a Comment. They also have a ToolStyle property to specify exactly what type of tool technology that it is. If you need to see values that are only on specific tools, use the ToolStyle value which is provided by the espToolType enumeration. We can use this property to type cast our generic Tool objects as specific tool technologies.

public void ScanTools()
{
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
var tools = Document.Tools as EspritTools.Tools;
foreach (EspritTechnology.ITechnology technology in tools)
{
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolId = {technology.FindParameter("ToolID").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolNumber = {technology.FindParameter("ToolNumber").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"Comment = {technology.FindParameter("Comment").Value}");
var toolStyle = (EspritConstants.espToolType)technology.FindParameter("ToolStyle").Value;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolStyle = {toolStyle}");
switch (toolStyle)
{
case EspritConstants.espToolType.espMillToolDrill:
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ShankDiameter = {technology.FindParameter("ShankDiameter").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"TipAngle = {technology.FindParameter("TipAngle").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolDiameter = {technology.FindParameter("ToolDiameter").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolLength = {technology.FindParameter("ToolLength").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolTip = {technology.FindParameter("ToolTip").Value}");
break;
case EspritConstants.espToolType.espMillToolEndMill:
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ShankDiameter = {technology.FindParameter("ShankDiameter").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolDiameter = {technology.FindParameter("ToolDiameter").Value}");
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", $"ToolLength = {technology.FindParameter("ToolLength").Value}");
break;
}
}
}
Public Sub ScanTools()
' prepare the output window for display
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
' note that the generic Tool technology is from the EspritTechnology library
For Each tool As EspritTechnology.Tool In Document.Tools
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolID = " & tool.ToolID)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolNumber = " & tool.ToolNumber)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "Comment = " & tool.Comment)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolStyle = " & tool.ToolStyle)
' now type cast the generic tool to a specific one
Select Case tool.ToolStyle
Case EspritConstants.espToolType.espMillToolDrill
Dim drill As EspritTechnology.ToolMillDrill = tool
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ShankDiameter = " & drill.ShankDiameter)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "TipAngle = " & drill.TipAngle)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolDiameter = " & drill.ToolDiameter)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolLength = " & drill.ToolLength)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolTip = " & drill.ToolTip)
Case EspritConstants.espToolType.espMillToolEndMill
Dim endMill As EspritTechnology.ToolMillEndMill = tool
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ShankDiameter = " & endMill.ShankDiameter)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolDiameter = " & endMill.ToolDiameter)
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolLength = " & endMill.ToolLength)
End Select
' This outputs a blank line between tools
Next
End Sub
Sub ScanTools()
' prepare the output window for display
Application.EventWindow.Clear
Application.EventWindow.Visible = True
' note that the generic Tool technology is from the EspritTechnology library
Dim TL As EspritTechnology.Tool
For Each TL In Document.Tools
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolID = " & TL.ToolID)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolNumber = " & TL.ToolNumber)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "Comment = " & TL.Comment)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolStyle = " & TL.ToolStyle)
' now type cast the generic tool to a specific one
Select Case TL.ToolStyle
Case espMillToolDrill
Dim DR As EspritTechnology.ToolMillDrill
Set DR = TL
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ShankDiameter = " & DR.ShankDiameter)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "TipAngle = " & DR.TipAngle)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolDiameter = " & DR.ToolDiameter)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolLength = " & DR.ToolLength)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolTip = " & DR.ToolTip)
Case espMillToolEndMill
Dim EM As EspritTechnology.ToolMillEndMill
Set EM = TL
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ShankDiameter = " & EM.ShankDiameter)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolDiameter = " & EM.ToolDiameter)
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "ScanToolsTutorial", "ToolLength = " & EM.ToolLength)
End Select
End With
Next
End Sub
