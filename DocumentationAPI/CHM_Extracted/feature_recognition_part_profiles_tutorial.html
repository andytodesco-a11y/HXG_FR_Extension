Feature Recognition: Part Profiles

Altogether there are six methods for automatically creating features with FeatureRecognition. This tutorial will look at CreatePartProfileShadow, which will create geometry or features around the shadow that a part casts onto a plane. We need to provide a SelectionSet as an argument, along with just one plane in this case. As with the other feature recognition commands, an array of GraphicObject objects is returned.

Suppose that you load solids for your lathe machining, and you always want to get a shadow of the part in the XYZ plane to get geometry that you can use to create your rough, finish, and cutoff features.

The code below shows the example of accessing the objects returned by CreatePartProfileShadow. Suppose that we want to find the maximum diameter of our part, to know what size stock to use. We could make a feature profile on the YZ plane, and then loop through the elements of the feature and find which one was the farthest away from the origin point at the center.

public void FindMaximumDiameter()
{
if (Document.Solids.Count == 0)
{
MessageBox.Show("There are no solids in this file.", "FeatureRecognitionPartProfilesTutorial", MessageBoxButton.OK);
return;
}
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
var plane = PlaneHelper.GetPlane(Document, "Temp XYZ");
PlaneHelper.SetPlaneVectors(ref plane, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1);
var set = SelectionSetHelper.GetSelectionSet(Document, "Temp");
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = Document.SystemUnit == EspritConstants.espUnitType.espInch ? 0.001 : 0.025;
foreach (Esprit.Solid solid in Document.Solids)
{
set.RemoveAll();
set.Add(solid);
Array shadows = null;
try
{
shadows = Document.FeatureRecognition.CreatePartProfileShadow(set, plane, EspritConstants.espGraphicObjectReturnType.espFeatureChains);
}
catch(Exception)
{
MessageBox.Show("Fail on creating part profile shadow.", "FeatureRecognitionPartProfilesTutorial", MessageBoxButton.OK);
return;
}
var maxDiameter = 0.0;
for (var i = shadows.GetLowerBound(0); i <= shadows.GetUpperBound(0); i++)
{
var graphicObject = shadows.GetValue(i) as Esprit.IGraphicObject;
if (graphicObject.GraphicObjectType != EspritConstants.espGraphicObjectType.espFeatureChain)
{
continue;
}
var featureChain = graphicObject as Esprit.FeatureChain;
for (var j = 1; j <= featureChain.Count; j++)
{
Esprit.Point point1 = null;
Esprit.Point point2 = null;
switch (featureChain.Item[j].GraphicObjectType)
{
case EspritConstants.espGraphicObjectType.espArc:
var arc = featureChain.Item[j] as Esprit.Arc;
point1 = arc.Extremity(EspritConstants.espExtremityType.espExtremityStart);
point2 = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
break;
case EspritConstants.espGraphicObjectType.espSegment:
var segment = featureChain.Item[j] as Esprit.Segment;
point1 = segment.Extremity(EspritConstants.espExtremityType.espExtremityStart);
point2 = segment.Extremity(EspritConstants.espExtremityType.espExtremityEnd);
break;
}
var distance = Math.Sqrt(point1.Y * point1.Y + point1.Z * point1.Z);
if (distance > maxDiameter / 2)
{
maxDiameter = distance * 2;
}
distance = Math.Sqrt(point2.Y * point2.Y + point2.Z * point2.Z);
if (distance > maxDiameter / 2)
{
maxDiameter = distance * 2;
}
}
Document.FeatureChains.Remove(featureChain.Key);
}
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FeatureRecognitionPartProfilesTutorial", $"The Max YZ Diameter of Solid {solid.Key} is {maxDiameter}");
}
}

System.Windows.MessageBox.Show("There are no solids in this file.", "FeatureRecognitionPartProfilesTutorial", vbOK)
Return
End If
_application.EventWindow.Clear()
_application.EventWindow.Visible = True
Dim plane As Esprit.Plane = PlaneHelper.GetPlane(Document, "Temp XYZ")
PlaneHelper.SetPlaneVectors(plane, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1)
Dim selectionSet = SelectionSetHelper.GetSelectionSet(Document, "Temp")
_application.Configuration.ConfigurationFeatureRecognition.Tolerance = IIf(Document.SystemUnit = EspritConstants.espUnitType.espInch, 0.001, 0.025)
For Each solid As Esprit.Solid In Document.Solids
selectionSet.RemoveAll()
selectionSet.Add(solid)
Dim shadow As Array = Nothing
Try
shadow = Document.FeatureRecognition.CreatePartProfileShadow(selectionSet, plane, EspritConstants.espGraphicObjectReturnType.espFeatureChains)
Catch ex As Exception
System.Windows.MessageBox.Show("Fail on creating part profile shadow.", "FeatureRecognitionPartProfilesTutorial", vbOK)
Return
End Try
Dim maxDiameter As Double = 0.0
For i As Integer = shadow.GetLowerBound(0) To shadow.GetUpperBound(0)
Dim graphicObject As Esprit.IGraphicObject = shadow.GetValue(i)
If graphicObject.GraphicObjectType <> EspritConstants.espGraphicObjectType.espFeatureChain Then
Continue For
End If
Dim featureChain As Esprit.FeatureChain = graphicObject
For j As Integer = 1 To featureChain.Count
Dim point1 As Esprit.Point = Nothing
Dim point2 As Esprit.Point = Nothing
Select Case featureChain.Item(j).GraphicObjectType
Case EspritConstants.espGraphicObjectType.espArc
Dim arc As Esprit.Arc = featureChain.Item(j)
point1 = arc.Extremity(EspritConstants.espExtremityType.espExtremityStart)
point2 = arc.Extremity(EspritConstants.espExtremityType.espExtremityEnd)
Case EspritConstants.espGraphicObjectType.espSegment
Dim segment As Esprit.Segment = featureChain.Item(j)
point1 = segment.Extremity(EspritConstants.espExtremityType.espExtremityStart)
point2 = segment.Extremity(EspritConstants.espExtremityType.espExtremityEnd)
End Select
Dim dist = Math.Sqrt(point1.Y * point1.Y + point1.Z * point1.Z)
If (dist > maxDiameter / 2) Then
maxDiameter = dist * 2
End If
dist = Math.Sqrt(point2.Y * point2.Y + point2.Z * point2.Z)
If (dist > maxDiameter / 2) Then
maxDiameter = dist * 2
End If
Next
Document.FeatureChains.Remove(featureChain.Key)
Next
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FeatureRecognitionPartProfilesTutorial", $"The Max YZ Diameter of Solid {solid.Key} is {maxDiameter}")

Public Function GetOrAddSelectionSet(document As Esprit.Document, name As String) As Esprit.SelectionSet
Dim selSet As Esprit.SelectionSet = document.SelectionSets.Item(name)
If selSet Is Nothing Then
selSet = document.SelectionSets.Add(name)
End If
Return selSet
End Function
Sub FindMaximumDiameter()
' check to make sure we have some solids
If Document.Solids.Count = 0 Then
Call MsgBox("There are no solids in this file. This macro is only for solids.", vbOKOnly, "Error")
Exit Sub
End If
' make a plane parallel to YZX
Dim PL As Esprit.Plane
Set PL = GetPlane("Temp XYZ")
' set the appropriate work plane vectors and do not have it as a view plane
SetPlaneVectors(PL, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1).IsWork = False
' get ready to display results
Application.EventWindow.Clear
Application.EventWindow.Visible = True
' loop through all of the solids in the file, add each to the selection set individually
Dim SL As Esprit.Solid
Dim SS As Esprit.SelectionSet
Set SS = GetSelectionSet("Temp")
For Each SL In Document.Solids
SS.RemoveAll
Call SS.Add(SL)
' perform part profile shadow creation with an appropriate tolerance
If Document.SystemUnit = espInch Then
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.001
Else
Application.Configuration.ConfigurationFeatureRecognition.Tolerance = 0.025
End If
Dim Shadows() As Esprit.GraphicObject
Shadows = Document.FeatureRecognition.CreatePartProfileShadow(SS, PL, espFeatureChains)
' loop through the resulting graphic objects
Dim MaxDiameter As Double
MaxDiameter = 0
Dim I As Long
For I = LBound(Shadows) To UBound(Shadows)
If Shadows(I).GraphicObjectType = espFeatureChain Then
Dim FC As Esprit.FeatureChain
Set FC = Shadows(I)
' loop through all of the elements in the feature, determine maximum diameter
Dim J As Long
For J = 1 To FC.Count
Dim P(1 To 2) As Esprit.Point
Select Case FC.Item(J).GraphicObjectType
Case espArc
Dim A As Esprit.Arc
Set A = FC.Item(J)
Set P(1) = A.Extremity(espExtremityStart)
Set P(2) = A.Extremity(espExtremityEnd)
Case espSegment
Dim S As Esprit.Segment
Set S = FC.Item(J)
Set P(1) = S.Extremity(espExtremityStart)
Set P(2) = S.Extremity(espExtremityEnd)
End Select
' find the distance to each end point
Dim Distance As Double
Dim K As Long
For K = 1 To 2
Distance = Sqr(P(K).Y ^ 2 + P(K).Z ^ 2)
If Distance > (MaxDiameter / 2) Then
MaxDiameter = 2 * Distance
End If
Next
Next
Call Document.FeatureChains.Remove(FC.Key)
End If
Next
' display the results
With Application.EventWindow
Call .AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FeatureRecognitionPartProfilesTutorial", "The Max YZ Diameter of Solid " & SL.Key & " is " & MaxDiameter & vbCrLf)
End With
Next
End Sub
Function GetSelectionSet(SSName As String) As Esprit.SelectionSet
On Error Resume Next
Set GetSelectionSet = Document.SelectionSets.Item(SSName)
On Error GoTo 0
If GetSelectionSet Is Nothing Then
Set GetSelectionSet = Document.SelectionSets.Add(SSName)
End If
End Function

Before running the code open a file with a lathe type solid in it, or create one yourself by using the Solid Modeler to rotate a feature.
