Create Hole Processing Info

This tutorial shows how to create Leader objects and Notes, combine those two and place notes that have hole processing information near chosen circles or arcs with leaders pointing from the notes to the circle or arc.

C#
VB.NET
VBA
public void HoleProcessingInfo()
{
var text = "Describe Center Drill / Drill / Tap or Any Other Info Here";
if (!RequestUserInput("Enter Processing Info", "Processing Info", ref text))
{
return;
}
var configuration = _application.Configuration;
foreach (Esprit.Circle circule in Document.Circles)
{
if (!circule.Grouped)
{
continue;
}
var point1 = Document.GetPoint(circule.X + circule.Radius * Math.Cos(Math.PI / 4), circule.Y + circule.Radius * Math.Sin(Math.PI / 4), circule.Z);
var point2 = Document.GetPoint(circule.X + 1.5 * circule.Radius, circule.Y + 1.5 * circule.Radius, circule.Z);
var point3 = Document.GetPoint(circule.X + 1.75 * circule.Radius, circule.Y + 1.5 * circule.Radius, circule.Z);
var leader = Document.Annotations.AddLeader(point1, point2);
leader.LeaderPoints.Add(point3);
leader.Color = configuration.ConfigurationDimensions.Color;
leader.LineType = configuration.ConfigurationDimensions.LineType;
var notes = Document.Annotations.AddNotes();
notes.Color = configuration.ConfigurationNotes.Color;
var note = notes.Add();
note.Font.Name = "Arial";
note.Font.Size = 6;
note.Font.Bold = true;
note.RotationAngle = configuration.ConfigurationAnnotation.RotationAngle;
note.String = $"R{circule.Radius}";
note.X = point3.X + 0.1 * circule.Radius;
note.Y = point3.Y;
note.Z = point3.Z;
note = notes.Add();
note.Font.Name = "Arial";
note.Font.Size = 6;
note.Font.Bold = true;
note.RotationAngle = configuration.ConfigurationAnnotation.RotationAngle;
note.String = text;
note.X = point3.X + 0.1 * circule.Radius;
note.Y = point3.Y - note.Height;
note.Z = point3.Z;
}
foreach (Esprit.Arc arc in Document.Arcs)
{
if (!arc.Grouped || Math.Abs(arc.EndAngle - arc.StartAngle) != 2 * Math.PI)
{
continue;
}
var point1 = Document.GetPoint(arc.X + arc.Radius * Math.Cos(Math.PI / 4), arc.Y + arc.Radius * Math.Sin(Math.PI / 4), arc.Z);
var point2 = Document.GetPoint(arc.X + 1.5 * arc.Radius, arc.Y + 1.5 * arc.Radius, arc.Z);
var point3 = Document.GetPoint(arc.X + 1.75 * arc.Radius, arc.Y + 1.5 * arc.Radius, arc.Z);
var leader = Document.Annotations.AddLeader(point1, point2);
leader.LeaderPoints.Add(point3);
leader.Color = configuration.ConfigurationDimensions.Color;
leader.LineType = configuration.ConfigurationDimensions.LineType;
var notes = Document.Annotations.AddNotes();
notes.Color = configuration.ConfigurationNotes.Color;
var note = notes.Add();
note.Font.Name = "Arial";
note.Font.Size = 6;
note.Font.Bold = true;
note.RotationAngle = configuration.ConfigurationAnnotation.RotationAngle;
note.String = $"R{arc.Radius}";
note.X = point3.X + 0.1 * arc.Radius;
note.Y = point3.Y + 0.5 * configuration.ConfigurationAnnotation.HeightSpacing * note.Height;
note.Z = point3.Z;
note = notes.Add();
note.Font.Name = "Arial";
note.Font.Size = 6;
note.Font.Bold = true;
note.RotationAngle = configuration.ConfigurationAnnotation.RotationAngle;
note.String = text;
note.X = point3.X + 0.1 * arc.Radius;
note.Y = point3.Y - note.Height - 0.5 * configuration.ConfigurationAnnotation.HeightSpacing * note.Height;
note.Z = point3.Z;
}
}

Sub HoleProcessingInfo()
Dim Circl As Esprit.Circle
Dim Arc As Esprit.Arc
Dim PI As Double
PI = 4 * Atn(1)
Dim Points(1 To 3) As Esprit.Point
Dim Leader As Esprit.Leader
Dim Notes As Esprit.Notes
Dim ProcessingInfo As String
ProcessingInfo = InputBox("Enter Processing Info", "Processing Info", "Describe Center Drill / Drill / Tap or Any Other Info Here")
' loop through all the circles
For Each Circl In Document.Circles
If Circl.Grouped Then
' create the leader from a position at 45 degrees on the circle
Set Points(1) = Document.GetPoint(Circl.X + Circl.Radius * Cos(PI / 4), Circl.Y + Circl.Radius * Sin(PI / 4), Circl.Z)
Set Points(2) = Document.GetPoint(Circl.X + 1.5 * Circl.Radius, Circl.Y + 1.5 * Circl.Radius, Circl.Z)
Set Points(3) = Document.GetPoint(Circl.X + 1.75 * Circl.Radius, Circl.Y + 1.5 * Circl.Radius, Circl.Z)
Set Leader = Document.Annotations.AddLeader(Points(1), Points(2))
Call Leader.LeaderPoints.Add(Points(3))
' set the leader to default attributes
Leader.Color = Application.Configuration.ConfigurationDimensions.Color
Leader.LineType = Application.Configuration.ConfigurationDimensions.LineType
' create a two line note, use default attributes for the note
Set Notes = Document.Annotations.AddNotes
Notes.Color = Application.Configuration.ConfigurationNotes.Color
Notes.Add
Notes.Item(1).Font = Application.Configuration.ConfigurationAnnotation.Font
Notes.Item(1).RotationAngle = Application.Configuration.ConfigurationAnnotation.RotationAngle
' first line of the note will describe the radius
Notes.Item(1).String = "R" & Circl.Radius
Notes.Item(1).X = Points(3).X + 0.1 * Circl.Radius
Notes.Item(1).Y = Points(3).Y
Notes.Item(1).Z = Points(3).Z
Notes.Add
Notes.Item(2).Font = Application.Configuration.ConfigurationAnnotation.Font
Notes.Item(2).RotationAngle = Application.Configuration.ConfigurationAnnotation.RotationAngle
' second line of the note will be the processing info that the user entered
Notes.Item(2).String = ProcessingInfo
Notes.Item(2).X = Points(3).X + 0.1 * Circl.Radius
Notes.Item(2).Y = Points(3).Y - Notes.Item(1).Height
Notes.Item(2).Z = Points(3).Z
End If
Next
' loop through all the arcs
For Each Arc In Document.Arcs
If Arc.Grouped Then
' Test to see if the arc is 360 degrees.
If (Abs(Arc.EndAngle - Arc.StartAngle) = 2 * PI) Then
' create the leader from a position at 45 degrees on the arc
Set Points(1) = Document.GetPoint(Arc.X + Arc.Radius * Cos(PI / 4), Arc.Y + Arc.Radius * Sin(PI / 4), Arc.Z)
Set Points(2) = Document.GetPoint(Arc.X + 1.5 * Arc.Radius, Arc.Y + 1.5 * Arc.Radius, Arc.Z)
Set Points(3) = Document.GetPoint(Arc.X + 1.75 * Arc.Radius, Arc.Y + 1.5 * Arc.Radius, Arc.Z)
Set Leader = Document.Annotations.AddLeader(Points(1), Points(2))
Call Leader.LeaderPoints.Add(Points(3))
' set the leader to default attributes
Leader.Color = Application.Configuration.ConfigurationDimensions.Color
Leader.LineType = Application.Configuration.ConfigurationDimensions.LineType
' create a two line note, use default attributes for the note
Set Notes = Document.Annotations.AddNotes
Notes.Add
Notes.Color = Application.Configuration.ConfigurationNotes.Color
Notes.Item(1).Font = Application.Configuration.ConfigurationAnnotation.Font
Notes.Item(1).RotationAngle = Application.Configuration.ConfigurationAnnotation.RotationAngle
' first line of the note will describe the radius
Notes.Item(1).String = "R" & Arc.Radius
Notes.Item(1).X = Points(3).X + 0.1 * Arc.Radius
Notes.Item(1).Y = Points(3).Y
Notes.Item(1).Z = Points(3).Z
Notes.Add
Notes.Item(2).Font = Application.Configuration.ConfigurationAnnotation.Font
Notes.Item(2).RotationAngle = Application.Configuration.ConfigurationAnnotation.RotationAngle
' second line of the note will be the processing info that the user entered
Notes.Item(2).String = ProcessingInfo
Notes.Item(2).X = Points(3).X + 0.1 * Arc.Radius
Notes.Item(2).Y = Points(3).Y - Notes.Item(1).Height
Notes.Item(2).Z = Points(3).Z
End If
End If
Next
Document.Refresh
End Sub

Be sure to group some circles or 360 degree arcs before running this code, then enter an appropriate description for a hole processing task, and examine the results.
