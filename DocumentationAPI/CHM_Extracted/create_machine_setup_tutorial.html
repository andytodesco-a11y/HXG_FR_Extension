private void CreateMachineSetup()
{
Document.InitialMachineSetup.OnModify += _machineSetup_OnModify;
Document.InitialMachineSetup.BeginEdit();
for(var i = Document.InitialMachineSetup.MachineItems.Count; i > 0; i--)
{
Document.InitialMachineSetup.MachineItems.Remove(i);
}
try
{
Document.InitialMachineSetup.MachineFileName = _application.Configuration.GetFileDirectory(EspritConstants.espFileType.espFileTypeMachineSetup) + "/Mori_Seiki_NLX2500/Mori_Seiki_NLX2500.mprj";
}
catch(System.Runtime.InteropServices.COMException)

return;
}

try
{
var fixture = Document.InitialMachineSetup.MachineItems.AddFixture(fixtureResourceDirectoryPath + "\\Chuck_OD_D5toD200.gdml") as Esprit.IFixture;
fixture.Name = "Fixture";

If the workpiece item does not have a reference to stock or it cannot be clamped then Machine setup tree informs user about it. To avoid such notification the following code adds a bar stock:

C#
VB.NET
VBA
private void AddBarStock()
{
Document.PartSetup.BeginEdit();
Esprit.Workpiece workpiece = null;
if (Document.Workpieces.Count > 0)
{
workpiece = Document.Workpieces[1];
}
else
{
workpiece = Document.Workpieces.Add();
}
var barStock = workpiece.Stocks.AddBarStock();
barStock.Tolerance = 0.1;
barStock.OutsideDiameter = 50;
var alignment = new EspritConstants.STOCKALIGNMENT { AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered, Size = 200 };
barStock.set_LengthAlignment(alignment);
Document.PartSetup.EndEdit();
}
Private Sub AddBarStock()
Document.PartSetup.BeginEdit()
Dim workpiece As ESPRIT.Workpiece = Nothing
If Document.Workpieces.Count > 0 Then
workpiece = Document.Workpieces(1)
Else
workpiece = Document.Workpieces.Add()
End If
Dim barStock As ESPRIT.BarStock = workpiece.Stocks.AddBarStock()
barStock.Tolerance = 0.1
barStock.OutsideDiameter = 50
Dim alignment = New EspritConstants.STOCKALIGNMENT With {.AlignmentType = EspritConstants.espStockAlignmentType.espStockAlignmentTypeCentered, .Size = 200}
barStock.LengthAlignment = alignment
Document.PartSetup.EndEdit()
End Sub
Sub AddBarStock()
Document.PartSetup.BeginEdit
Dim wp As Workpiece
If Document.Workpieces.Count > 0 Then
Set wp = Document.Workpieces(1)
Else
Set wp = Document.Workpieces.Add
End If
Dim bs As BarStock
Set bs = wp.Stocks.AddBarStock
bs.Tolerance = 0.1
bs.OutsideDiameter = 50
Dim Alignment As STOCKALIGNMENT
Alignment.AlignmentType = espStockAlignmentTypeCentered
Alignment.Size = 200
bs.LengthAlignment = Alignment
Document.PartSetup.EndEdit
End Sub

To add a fixture with jaws run following code:

C#
VB.NET
VBA
private void AddFixtureWithJaws()
{
try
{
var fixtureFilePath = GetFile("Fixture with jaw adaptors", "Fixture Files (*.jdml)|*.gdml");
if (fixtureFilePath != string.Empty)
{
Document.InitialMachineSetup.BeginEdit();
var fixture = Document.InitialMachineSetup.MachineItems.AddFixture(fixtureFilePath) as Esprit.IFixture;
fixture.Name = "Fixture with Jaw adaptor";
Document.InitialMachineSetup.EndEdit();
var jawAdaptors = new List<Esprit.IAdaptor>();
foreach (Esprit.IAdaptor adaptor in fixture.ChildAdaptors)
{
if (adaptor.AdaptorType == EspritConstants.espAdaptorType.espAdaptorTypeJaw)
{
jawAdaptors.Add(adaptor);
}
}
Document.InitialMachineSetup.BeginEdit();
if (jawAdaptors.Any())
{
var jawFilePath = GetFile("Jaw", "Jaw Files (*.jdml)|*.gdml");
if (jawFilePath != string.Empty)
{
fixture.MachineItems.AddJaw(jawFilePath, 100, jawAdaptors.FirstOrDefault().Key);
if (jawAdaptors.Count > 1)
{
fixture.MachineItems.AddJaws(jawFilePath, 150);
}
}
}
Document.InitialMachineSetup.EndEdit();
}
}
catch (System.Runtime.InteropServices.COMException)
{
}
}
Private Sub AddFixtureWithJaws()
Try
Dim fixtureFilePath = GetFile("Fixture with jaw adaptors", "Fixture Files (*.jdml)|*.gdml")
If Not fixtureFilePath Is String.Empty Then
Document.InitialMachineSetup.BeginEdit()
Dim fixture As ESPRIT.IFixture = Document.InitialMachineSetup.MachineItems.AddFixture(fixtureFilePath)
fixture.Name = "Fixture with Jaw adaptor"
Document.InitialMachineSetup.EndEdit()
Dim jawAdaptors = New List(Of ESPRIT.IAdaptor)
For Each adaptor As ESPRIT.IAdaptor In fixture.ChildAdaptors
If adaptor.AdaptorType = EspritConstants.espAdaptorType.espAdaptorTypeJaw Then
jawAdaptors.Add(adaptor)
End If
Next
Document.InitialMachineSetup.BeginEdit()
If jawAdaptors.Any() Then
Dim jawFilePath = GetFile("Jaw", "Jaw Files (*.jdml)|*.gdml")
If Not jawFilePath Is String.Empty Then
fixture.MachineItems.AddJaw(jawFilePath, 100, jawAdaptors.FirstOrDefault().Key)
If jawAdaptors.Count > 1 Then
fixture.MachineItems.AddJaws(jawFilePath, 150)
End If
End If
End If
Document.InitialMachineSetup.EndEdit()
End If
Catch ex As System.Runtime.InteropServices.COMException
Return
End Try
End Sub
Sub AddFixtureWithJaws()
Call Document.InitialMachineSetup.BeginEdit
Dim mFixture As Esprit.IFixture
Set mFixture = Document.InitialMachineSetup.MachineItems.AddFixture("C:\tmp\BuckB8126.gdml")
mFixture.Name = "Fixture with Jaw adaptor"
Call Document.InitialMachineSetup.EndEdit
Dim jawAdaptors() As Esprit.IAdaptor
Dim adaptor As Esprit.IAdaptor
Dim jawAdaptorsCount As Long
For Each adaptor In mFixture.ChildAdaptors
If adaptor.AdaptorType = EspritConstants.espAdaptorType.espAdaptorTypeJaw Then
jawAdaptorsCount = jawAdaptorsCount + 1
ReDim Preserve jawAdaptors(jawAdaptorsCount) As Esprit.IAdaptor
Set jawAdaptors(jawAdaptorsCount) = adaptor
End If
Next
Call Document.InitialMachineSetup.BeginEdit
If jawAdaptorsCount > 0 Then
Dim jawFilePath As String
jawFilePath = "C:\tmp\ME-Jaw.gdml"
Call mFixture.MachineItems.AddJaw(jawFilePath, 100, jawAdaptors(1).Key)
If jawAdaptorsCount > 1 Then
Call mFixture.MachineItems.AddJaws(jawFilePath, 150)
End If
End If
Call Document.InitialMachineSetup.EndEdit
End Sub
