FeatureChain Visible Circles And Arcs

This tutorial will loop through all of the Circle objects in the Document and create features on all of the visible ones at a start angle entered by the user. It will also loop through all Arc objects in the Document and check if they are 360 degrees and visible, and if so it will create a feature on the arc at its start point. These features will be created on the same layer as the original circles and arcs.

This tutorial brings up an important topic in programming: how do you test for certain values or conditions, and how close do you have to be to satisfy that requirement? In this case we will need to test if an arc is 360 degrees. But computer languages do not think in terms of degrees, they use radians. 360 degrees is 2 * pi radians. Pi is a transcendental number with digits going out to infinity after the decimal. At some point the computer must round off 2 * pi, so our calculation will never be exact. The only way to satisfy a condition like this is to introduce some kind of tolerance, and see if what you are testing is close enough (within tolerance). There can be many approaches to this method. This tutorial demonstrates one possible solution out of many alternatives.

public void FeatureChainVisibleCirclesAndArcs()
{
var angleDegree = 90.0;
if (!RequestUserInput("Enter Starting Angle", "Starting Angle", ref angleDegree))
{
return;
}
var angleRad = angleDegree * Math.PI / 180;
var previousActiveColor = Document.ActiveColor;
Document.ActiveColor = ColorHelper.ColorToUInt(Color.Blue);
//Because you cannot add a circle to a FeatureChain, make temporary arc instead
foreach (Esprit.Circle circle in Document.Circles)
{
if (circle.Layer.Visible)
{
var featureChain = Document.FeatureChains.Add(
Document.GetPoint(circle.X + circle.Radius * Math.Cos(angleRad), circle.Y + circle.Radius * Math.Sin(angleRad), circle.Z));
featureChain.Layer = circle.Layer;
featureChain.Add(Document.GetArc(circle.CenterPoint, circle.Radius, angleRad, 2 * Math.PI + angleRad));
}
}
foreach (Esprit.Arc arc in Document.Arcs)
{
if (arc.Layer.Visible)
{
var tolerance = Document.SystemUnit == espUnitType.espInch
? 0.00001
: 0.00025;
if (Math.Abs(arc.EndAngle - arc.StartAngle - 2 * Math.PI) < tolerance)
{
var featureChain = Document.FeatureChains.Add(arc.Extremity(EspritConstants.espExtremityType.espExtremityStart));
featureChain.Add(arc);
featureChain.Layer = arc.Layer;
}
}
}
Document.ActiveColor = previousActiveColor;
WindowHelper.FitAllWindows(Document.Windows);
}

End If
Next

tolerance = 0.00001

Public Function ColorToUInt(color As System.Drawing.Color) As UInteger
Dim a As UInteger = color.A
Dim r As UInteger = color.R
Dim g As UInteger = color.G
Dim b As UInteger = color.B
Dim resuls As UInteger = ((a << 24) Or (b << 16) Or (g << 8) Or (r << 0))
Return resuls
End Function

' loop through all the circles

Set FC = Document.FeatureChains.Add(Document.GetPoint(C.X + C.Radius * Cos(Angle), C.Y + C.Radius * Sin(Angle), C.Z))
FC.Layer = C.Layer
' because you cannot add a circle to a FeatureChain, make temporary arc instead
Call FC.Add(Document.GetArc(C.CenterPoint, C.Radius, Angle, 2 * PI + Angle))
End If
Next
' loop through all the arcs
For Each A In Document.Arcs
If A.Layer.Visible Then
' Test to see if the arc is 360 degrees.

If Document.SystemUnit = espInch Then
Tolerance = 0.00001
Else
Tolerance = 0.00025
End If
If (Abs(A.EndAngle - A.StartAngle - 2 * PI) < Tolerance) Then
' create the feature chain
