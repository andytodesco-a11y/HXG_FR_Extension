private static uint GetLayerCount(Esprit.Document document, Esprit.Layer layer)
{
uint count = 0;
for (var i = 1; i <= document.GraphicsCollection.Count; i++)
{
var graphicObject = document.GraphicsCollection[i] as Esprit.GraphicObject;
if (graphicObject.Layer.Name == layer.Name && graphicObject.TypeName != "Unknown")
{
count++;
}
}
return count;
}
Private Function GetLayerCount(document As Esprit.Document, layer As Esprit.Layer) As UInteger
Dim count As UInteger = 0
For i As Integer = 1 To document.GraphicsCollection.Count
Dim graphicObject As Esprit.GraphicObject = document.GraphicsCollection.Item(i)
If graphicObject.Layer.Name = layer.Name AndAlso graphicObject.TypeName <> "Unknown" Then
count += 1
End If
Next
Return count
End Function
Public Function GetLayerCount(LayerObj As Esprit.Layer) As Long
GetLayerCount = 0
Dim G As Esprit.GraphicObject
Dim I As Long
' loop through the all the graphics
For I = 1 To Document.GraphicsCollection.Count
Set G = Document.GraphicsCollection.Item(I)
If G.Layer.Name = LayerObj.Name And G.TypeName <> "Unknown" Then
' GraphicObject is on that Layer, so increment the count
GetLayerCount = GetLayerCount + 1
End If
Next
End Function

private void MidPointDemo()
{
Esprit.Point point1 = null;
Esprit.Point point2 = null;
_form.Hide();
try
{
point1 = Document.GetPoint("DateTimePicker Point 1");
point2 = Document.GetPoint("DateTimePicker Point 2");
}
catch (System.Runtime.InteropServices.COMException)
{
// Element was not set
}
if (point1 != null && point2 != null)
{
var midPoint = AddPoint(Document, GetMidPoint(Document, point1, point2));
Document.Refresh();
_application.EventWindow.Clear();
_application.EventWindow.Visible = true;
_application.EventWindow.AddMessage(EspritConstants.espMessageType.espMessageTypeInformation, "FunctionsTutorial", $"Midpoint is {midPoint.X}, {midPoint.Y}, {midPoint.Z}");
}
_form.Show();
}
private static Esprit.Point AddPoint(Esprit.Document document, Esprit.Point point)
{
return document.Points.Add(point.X, point.Y, point.Z);
}
private static Esprit.Point GetMidPoint(Esprit.Document document, Esprit.Point point1, Esprit.Point point2)
{
return document.GetPoint((point1.X + point2.X) / 2.0, (point1.Y + point2.Y) / 2.0, (point1.Z + point2.Z) / 2.0);
}

Sub MidPointDemo()
Dim P(1 To 3) As Esprit.Point
Set P(1) = Nothing
Set P(2) = Nothing
On Error Resume Next
Set P(1) = Document.GetPoint("Pick Point 1")
Set P(2) = Document.GetPoint("Pick Point 2")

Exit Sub

Document.Refresh
End Sub

Public Function GetMidPoint(Point1 As Esprit.Point, Point2 As Esprit.Point) As Esprit.Point
Dim MidX As Double, MidY As Double, MidZ As Double
MidX = (Point1.X + Point2.X) / 2
MidY = (Point1.Y + Point2.Y) / 2
MidZ = (Point1.Z + Point2.Z) / 2
Set GetMidPoint = Document.GetPoint(MidX, MidY, MidZ)
End Function

Public Function LayerExists(LayerName As String) As Boolean
LayerExists = False
Dim L As Esprit.Layer
For Each L In Document.Layers
' because layer names are not case sensitive
' must do a text string comparison
If StrComp(L.Name, LayerName, vbTextCompare) = 0 Then
LayerExists = True

Exit Function
End If
Next
End Function

private static Esprit.Line Line2(Esprit.Document document, Esprit.Point point1, Esprit.Point point2)
{
var deltaX = point2.X - point1.X;
var deltaY = point2.Y - point1.Y;
var deltaZ = point2.Z - point1.Z;
var distance = Geometry.Distance(point1, point2);
return document.Lines.Add(point1, deltaX / distance, deltaY / distance, deltaZ / distance);
}
calculate distance between two point
public static double Distance(Point point1, Point point2 = null)
{
var x2 = point2?.X ?? 0;
var y2 = point2?.Y ?? 0;
var z2 = point2?.Z ?? 0;
return Distance(point1.X, point1.Y, point1.Z, x2, y2, z2);
}
public static double Distance(double x1, double y1, double z1, double x2, double y2, double z2)
{
var dx = x2 - x1;
var dy = y2 - y1;
var dz = z2 - z1;
return Math.Sqrt(dx * dx + dy * dy + dz * dz);
}
