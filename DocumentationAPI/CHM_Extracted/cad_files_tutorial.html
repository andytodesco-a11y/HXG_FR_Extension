Cad Files

An advantage of using classes is that they can also contain event procedures. Event procedures are predefined for certain objects and are triggered when something happens to or with that object. The code inside the event procedure is what responds to the occurrence of that event.

For example, for the Application object, some of the events are AfterDocumentClose, AfterDocumentOpen and BeforeQuit.

The Tutorial demostrates the CadFiles object events.

Define a class and add the event procedures:

VB.NET
VBA
class CadFilesEventsHandler
{
private readonly Esprit.Application _application;
private readonly string _messageBoxTitle;
public CadFilesEventsHandler(Esprit.Application app)
{
_application = app;
_messageBoxTitle = "CadFilesEventsTutorial";
var cadFiles = _application.Document.CadFiles;
cadFiles.AfterRemove += AfterRemove;
cadFiles.AfterUpdate += AfterUpdate;
cadFiles.BeforeRemove += BeforeRemove;
cadFiles.BeforeUpdate += BeforeUpdate;
cadFiles.OnCreate += OnCreate;
cadFiles.OnModify += OnModify;
MessageBox.Show($"Document contains {cadFiles.Count} files", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnModify(object CadFile)
{
MessageBox.Show("OnModify called on cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK);
}
private void OnCreate(object CadFile)
{
MessageBox.Show("OnCreate called on import a Cad file or calling Document.MergeFile", _messageBoxTitle, MessageBoxButton.OK);
}
private void BeforeUpdate(object CadFile, ref bool Override)
{
MessageBox.Show("BeforeUpdate called before cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK);
}
private void BeforeRemove(object CadFile, ref bool Override)
{
MessageBox.Show("BeforeRemove called before removing cadfile via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterUpdate(object CadFile)
{
MessageBox.Show("AfterUpdate called after cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK);
}
private void AfterRemove()
{
MessageBox.Show("AfterRemove called after removing cadfile via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK);
}
}
Public Class CadFilesEventsHandler
Private ReadOnly _application As ESPRIT.Application
Private ReadOnly _messageBoxTitle As String
Public Sub New(app As ESPRIT.Application)
_application = app
_messageBoxTitle = "CadFilesEventsTutorial"
With _application.Document.CadFiles
AddHandler .AfterRemove, AddressOf Me.AfterRemove
AddHandler .AfterUpdate, AddressOf Me.AfterUpdate
AddHandler .BeforeRemove, AddressOf Me.BeforeRemove
AddHandler .BeforeUpdate, AddressOf Me.BeforeUpdate
AddHandler .OnCreate, AddressOf Me.OnCreate
AddHandler .OnModify, AddressOf Me.OnModify
End With
MessageBox.Show($"Document contains {_application.Document.CadFiles.Count} files", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub AfterRemove()
MessageBox.Show("AfterRemove called after removing cadfile via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub AfterUpdate(CadFile As Object)
MessageBox.Show("AfterUpdate called after cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub BeforeRemove(CadFile As Object, ByRef Override As Boolean)
MessageBox.Show("BeforeRemove called before removing cadfile via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub BeforeUpdate(CadFile As Object, ByRef Override As Boolean)
MessageBox.Show("BeforeUpdate called before cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub OnCreate(CadFile As Object)
MessageBox.Show("OnCreate called on import a Cad file or calling Document.MergeFile", _messageBoxTitle, MessageBoxButton.OK)
End Sub
Private Sub OnModify(CadFile As Object)
MessageBox.Show("OnModify called on cadfile updated via Reference Files status dialog or API", _messageBoxTitle, MessageBoxButton.OK)
End Sub
End Class
Dim WithEvents MyAppCadFiles As Esprit.CadFiles
